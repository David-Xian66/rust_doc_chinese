1//! 捕获主循环：把屏幕抓下来 + 编码成 H.264
2//!
3//! 数据流（一帧）：
4//! ```text
5//! DXGI Desktop Duplication (dup.acquire_next_frame)
6//!   └─ ID3D11Texture2D (BGRA, 显存里)
7//!       └─ encoder.convert_bgra_to_nv12:  BGRA → NV12（GPU VideoProcessor）
8//!       └─ encoder.build_sample:          NV12 → IMFSample
9//!       └─ encoder.process_input:         IMFSample → MFT
10//!       └─ encoder.process_output:        MFT → IMFSample（含 H.264 字节）
11//!       └─ avcc_to_annex_b_into → EncodedFrame → mpsc → 网络线程切片发 QUIC
12//! ```
13//!
14//! # 重要术语
15//! - **DXGI Desktop Duplication**: Windows 7+ 提供的桌面复制 API，
16//!   应用层可以把"屏幕正在显示的画面"整帧拷到自己的显存里。
17//! - **HNS (100-nanosecond units)**: Media Foundation 时间单位，1 HNS = 100ns。
18//!   时间戳和帧时长都用这个单位，方便和 MF 内部时间轴对齐。
19

43/// RAII guard for `CoUninitialize`。`CoInitializeEx` 没有自动对偶，必须手动调。
44struct CoUninitGuard;
45impl Drop for CoUninitGuard {
46    fn drop(&mut self) {
47        unsafe {
48            // CoUninitialize 是 O(1) 且 idempotent，调用失败也不影响 cleanup
49            let _ = CoUninitialize();
50        }
51    }
52}

54/// DXGI `AcquireNextFrame` 的等待超时（毫秒）。
55///
56/// 不是"每帧间隔"，而是"如果这么久都没新帧就返回 Timeout 让你继续循环"。
57/// 乘 1.5 是经验值，给屏幕没变化时一个不会立刻返回、又能及时退出的等待时间。
58fn get_acquire_next_frame_time(fps_: u32) -> u32 {
59    (1000.0 / fps_ as f32 * 1.5) as u32
60}

62/// 一帧的时长（HNS，100ns 单位）。
63/// 比如 30fps 就是 1/30 秒 = 3,333,333 HNS。
64fn get_frame_duration_hns(fps_: u32) -> i64 {
65    10_000_000 / fps_ as i64

73/// 已经编码完的一帧 H.264，尚未切片 / 尚未拼 QUIC datagram。
74///
75/// ## 数据流
76///
77/// ```text
78/// capture 线程 (sync)               网络线程 (async)
79/// ─────────────────────             ────────────────
80/// H264Encoder.process_output()
81///   └─ IMFSample (H.264 bytes)
82///       └─ avcc_to_annex_b_into → annex_b: Bytes
83///           └─ 构造 EncodedFrame ──── mpsc::Sender ───▶  接收
84///              { is_keyframe,                                ▼
85///                annex_b,           切片 / 拼 FragmentHeader
86///                frame_seq,         + conn.send_datagram()
87///                keyframe_index }
88/// ```
89///
90/// ## 字段选择理由
91///
92/// - `annex_b: Bytes` —— 整片 Annex-B（含 `00 00 00 01` 起始码 + SPS/PPS 注入）。
93///   网络线程收到后按 `conn.max_datagram_size()` 动态 MTU 切 datagram 子片，
94///   内部 `whole.slice(r)` 零拷贝。
95///
96/// - `is_keyframe: bool` —— 来自 `avcc_to_annex_b_into` 的 NALU type == 5 判定，
97///   网络层据此决定切片后走可靠 stream 还是不可靠 datagram。
98///
99/// - `frame_seq: u32` —— capture 端单调递增的帧序号，给客户端用来检测丢帧 /
100///   重排序。u32 在 130 FPS 下连续跑 1.36 年才溢出，demo 阶段够用。
101///
102/// - `keyframe_index: u8` —— 当前 GOP 编号（每发一个 IDR 自增，u8 自然 wrap）。
103///   网络层据此填 fragment header 里的 kf_idx 字段。
104///
105/// - `delta_index: u16` —— 当前 GOP 内的 P 帧序号（每发一个 P 帧自增，u16 自然 wrap；
106///   遇 IDR 时由 `H264EncodedState::delta_index_add` 之前的逻辑置 0）。
107///   网络层据此填 fragment header 里的 delta_idx 字段。
108pub(crate) struct EncodedFrame {
109    pub is_keyframe: bool,
110    pub annex_b: Bytes,
111    pub frame_seq: u32,
112    pub keyframe_index: u8,
113    pub delta_index: u16,
114}

116/// 用户活动模式。
117///
118/// - **Active**：用户"正在操作"（拖动、输入、滚动）。FPS 拉满到配置值（如 59 fps），
119///   方案 B 启用（GPU 合成复用），方案 A **禁用**（不能漏关键帧）。
120/// - **Idle**：用户静止超过 N 秒。FPS 降到 IDLE_FPS，方案 B 仍启用，
121///   方案 A 启用（脏面积 < SKIP_THRESHOLD 时跳过整帧，进一步省带宽）。
122#[derive(Debug, Clone, Copy, PartialEq, Eq)]
123enum ActivityMode {
124    Active,
125    Idle,
126}
127
128/// 状态机参数
129///
130/// - `ACTIVE_DIRTY_THRESHOLD` (0.5%)：脏面积低于此值算"静止信号"
131/// - `IDLE_ENTRY_FRAMES` (30 帧 ≈ 0.5 秒 @ 59fps)：连续 N 帧都低于阈值才进入 Idle
132/// - `IDLE_FPS` (5 fps)：Idle 模式 FPS（人眼看静止画面 5 fps 察觉不到）
133/// - `SKIP_THRESHOLD` (0.5%)：Idle 模式下脏面积低于此值 → 跳过整帧
134const ACTIVE_DIRTY_THRESHOLD: f64 = 0.005;
135const IDLE_ENTRY_FRAMES: u32 = 30;
136const IDLE_FPS: u32 = 5;
137const SKIP_THRESHOLD: f64 = 0.005;
138
139/// 用户活动状态机。
140struct ActivityState {
141    mode: ActivityMode,
142    /// 连续 dirty 面积 <= ACTIVE_DIRTY_THRESHOLD 的帧数（Active → Idle 判定）
143    below_threshold_count: u32,
144    /// 累计统计（给 release-log 用）
145    total_active_frames: u64,
146    total_idle_frames: u64,
147    total_skipped_frames: u64,
148    total_active_to_idle: u64,
149    total_idle_to_active: u64,
150}
151
152impl ActivityState {
153    fn new() -> Self {
154        Self {
155            mode: ActivityMode::Active, // 启动默认 Active
156            below_threshold_count: 0,
157            total_active_frames: 0,
158            total_idle_frames: 0,
159            total_skipped_frames: 0,
160            total_active_to_idle: 0,
161            total_idle_to_active: 0,
162        }
163    }
164
165    /// 每帧调一次，输入 dirty_ratio，返回当前模式（可能发生切换）。
166    ///
167    /// `dirty_ratio` ∈ [0.0, 1.0]，表示脏矩形占屏幕面积的比例。
168    fn update(&mut self, dirty_ratio: f64) -> ActivityMode {
169        let prev_mode = self.mode;
170        match self.mode {
171            ActivityMode::Active => {
172                self.total_active_frames += 1;
173                if dirty_ratio <= ACTIVE_DIRTY_THRESHOLD {
174                    self.below_threshold_count += 1;
175                    if self.below_threshold_count >= IDLE_ENTRY_FRAMES {
176                        self.mode = ActivityMode::Idle;
177                        self.total_active_to_idle += 1;
178                    }
179                } else {
180                    self.below_threshold_count = 0;
181                }
182            }
183            ActivityMode::Idle => {
184                self.total_idle_frames += 1;
185                if dirty_ratio > ACTIVE_DIRTY_THRESHOLD {
186                    self.mode = ActivityMode::Active;
187                    self.below_threshold_count = 0;
188                    self.total_idle_to_active += 1;
189                }
190            }
191        }
192        #[cfg(any(debug_assertions, feature = "release-log"))]
193        if self.mode != prev_mode {
194            let mode_str = match self.mode {
195                ActivityMode::Active => "Active",
196                ActivityMode::Idle => "Idle",
197            };
198            let from_str = match prev_mode {
199                ActivityMode::Active => "Active",
200                ActivityMode::Idle => "Idle",
201            };
202            println!(
203                "🔀 [activity] 状态切换: {} -> {} (连续 {} 帧 < {:.2}%, dirty_ratio={:.4}%)",
204                from_str,
205                mode_str,
206                self.below_threshold_count,
207                ACTIVE_DIRTY_THRESHOLD * 100.0,
208                dirty_ratio * 100.0,
209            );
210        }
211        self.mode
212    }
213
214    /// Idle 模式 + 脏面积 < SKIP_THRESHOLD → 跳过整帧（A 启用条件）
215    fn should_skip(&self, dirty_ratio: f64) -> bool {
216        self.mode == ActivityMode::Idle && dirty_ratio < SKIP_THRESHOLD
217    }
218}

220/// 把 dirty_rects 列表的覆盖面积占屏幕的百分比算出来
221fn compute_dirty_ratio(dirty_rects: &[RECT], width: u32, height: u32) -> f64 {
222    if width == 0 || height == 0 || dirty_rects.is_empty() {
223        return 0.0;
224    }
225    let total_pixels = dirty::dirty_total_area(dirty_rects);
226    let screen_pixels = (width as u64) * (height as u64);
227    if screen_pixels == 0 {
228        return 0.0;
229    }
230    total_pixels as f64 / screen_pixels as f64
231}

233/// 主入口：选屏幕 → 起 DXGI 会话 → 死循环抓帧 + 编码。
234///
235/// `enumerate_index`: 屏幕索引（来自 `screen::screen_list_all` 的 HashMap key）
236/// `fps_`: 目标帧率（会和屏幕刷新率取较小值）
237/// `fps_screen`: 当前屏幕的刷新率
238/// `bitrate_bps`: 目标平均码率（bps 单位，1 Mbps = 1_000_000）
239/// `profile`: H.264 Profile（High/Main/Baseline）
240/// `slice_config`: H.264 Slice 切分配置（透传给编码器）
241/// `shutdown`: 关闭信号 flag。Ctrl+C handler 在 main.rs 里设，主循环检查。
242/// `lockscreen`: 锁屏监听器（可选）。传 None 时禁用锁屏暂停特性。
243/// `frame_tx`: 编码后的 EncodedFrame 通过这个 sender 推给网络线程切片 + 发 QUIC。
244#[allow(clippy::too_many_arguments)]
245pub fn capture_main(
246    enumerate_index: usize,
247    fps_: u32,
248    fps_screen: u32,
249    bitrate_bps: u32,
250    profile: H264Profile,
251    slice_config: H264SliceConfig,
252    shutdown: Arc<AtomicBool>,
253    lockscreen: Option<LockscreenWatcher>,
254    frame_tx: tokio::sync::mpsc::Sender<EncodedFrame>
255) -> Result<(), Box<dyn std::error::Error>> {
256    unsafe {
257        // COINIT_MULTITHREADED = 公寓模型为 MTA（多线程），所有 COM 对象跨线程访问安全。
258        CoInitializeEx(None, COINIT_MULTITHREADED).ok()?;
259    }
260    // RAII 配对：函数返回（包括 panic / 早返回）自动调 CoUninitialize
261    let _co_uninit = CoUninitGuard;
262
263    let mut session = CaptureSession::new(
264        enumerate_index,
265        fps_ as u32,
266        bitrate_bps,
267        profile,
268        slice_config,
269        frame_tx
270    )?;
271    let mut recover_state = RecoverState::new();
272    #[cfg(any(debug_assertions, feature = "release-log"))]
273    println!("✅ 会话已建立: {}x{} @{}fps, 码率 {} bps",
274        session.width(), session.height(), session.fps(), session.bitrate());
275
276    // 连续 N 次 AccessLost / SessionError 视为"持续失败"，自动进入 fallback
277    // (释放 session + 阻塞等 WTS 解锁事件 + recreate)。
278    const DXGI_FALLBACK_THRESHOLD: u32 = 3;
279    let mut dxgi_consecutive_failures: u32 = 0;
280
281    let mut activity_state = ActivityState::new();
282    #[cfg(any(debug_assertions, feature = "release-log"))]
283    println!(
284        "🎚️ [activity] 状态机已启动: 阈值={}%, 进入 Idle 需 {} 帧, Idle FPS={}, 跳过阈值={}%",
285        ACTIVE_DIRTY_THRESHOLD * 100.0,
286        IDLE_ENTRY_FRAMES,
287        IDLE_FPS,
288        SKIP_THRESHOLD * 100.0,
289    );
290
291    let mut first_valid_frame = true;
292
293    #[cfg(any(debug_assertions, feature = "release-log"))]
294    let mut frame_count: u64 = 0;
295    #[cfg(any(debug_assertions, feature = "release-log"))]
296    let mut output_count: u64 = 0;
297    #[cfg(any(debug_assertions, feature = "release-log"))]
298    let mut timeout_count: u64 = 0;
299
300    let mut wire_frame_index: u64 = 0;
301    #[cfg(any(debug_assertions, feature = "release-log"))]
302    let mut first_frame_logged = false;
303
304    #[cfg(any(debug_assertions, feature = "release-log"))]
305    println!(
306        "📡 进入捕获主循环, 屏幕刷新率={}fps, 配置 FPS={}",
307        fps_screen, fps_
308    );
309
310    // OnceLock 保证整个进程里只 init 一次。start 用作时间戳的基准点（进程启动时刻）
311    static CAPTURE_START: OnceLock<Instant> = OnceLock::new();
312    let start = *CAPTURE_START.get_or_init(Instant::now);
313    let mut hns_timestamp: i64;
314
315    let mut acquire_next_frame_time;
316    let mut frame_duration_hns: i64;
317
318    let mut H264_struct = H264EncodedState::new();
319
320    loop {
321        // 循环顶部 —— 每次循环先检查 shutdown flag
322        if shutdown.load(Ordering::Relaxed) {
323            #[cfg(any(debug_assertions, feature = "release-log"))]
324            println!("🛑 主循环退出 (shutdown flag set)");
325            break;
326        }
327
328        // session 已 release 但当前不在 paused 分支时，立即重建。
329        // 触发场景：DXGI fallback 释放 session 后 `continue`，但此刻用户没锁屏
330        // （is_paused() = false），会跳过下面的 paused 分支直接走主路径，
331        // 不重建会走到 session.dup() 触发 expect panic。
332        if session.is_released()
333            && (lockscreen.is_none() || !lockscreen.as_ref().unwrap().is_paused())
334        {
335            #[cfg(any(debug_assertions, feature = "release-log"))]
336            println!("🏗️ session 已 release 但未进入 paused 分支，立即重建");
337            if let Err(e) = session.recreate() {
338                return Err(format!("立即重建 session 失败: {}", e).into());
339            }
340            dxgi_consecutive_failures = 0;
341        }
342
343        // 锁屏 / 用户切换暂停分支：
344        //   触发：LockscreenWatcher 检测到 WTS_LOCK / WTS_LOGOFF
345        //   行为：release session → 阻塞等解锁 → recreate + force IDR
346        if let Some(lw) = &lockscreen
347            && lw.is_paused()
348        {
349            if !session.is_released() {
350                session.release();
351                #[cfg(any(debug_assertions, feature = "release-log"))]
352                println!(
353                    "🔒 进入暂停 (会话状态={:?}), session 已释放",
354                    lw.current_state()
355                );
356            }
357            lw.wait_until_active(&shutdown)?;
358            if shutdown.load(Ordering::Relaxed) {
359                break;
360            }
361            if let Err(e) = session.recreate() {
362                return Err(format!("解锁后 recreate 失败: {}", e).into());
363            }
364            dxgi_consecutive_failures = 0;
365            #[cfg(any(debug_assertions, feature = "release-log"))]
366            println!("🔓 解锁/切换完成, session 重建, 继续捕获");
367            continue;
368        }
369
370        // 根据当前 mode 动态调整 acquire 超时：
371        //   Active: 短超时；Idle: 长超时（降低 CPU 占用）
372        acquire_next_frame_time = match activity_state.mode {
373            ActivityMode::Active => get_acquire_next_frame_time(fps_screen),
374            ActivityMode::Idle => (1000.0 / IDLE_FPS as f32 * 1.5) as u32,
375        };
376        unsafe {
377            // acquire_next_frame: 阻塞等待下一帧，最多等 acquire_next_frame_time 毫秒
378            // 返回值：
379            //   Ok(frame)         - 拿到一帧
380            //   Err(Timeout)      - 超时但没新帧，屏幕没变化
381            //   Err(AccessLost)   - 桌面切换/锁屏/DXGI 被其他程序抢走，需要 recreate
382            match session.dup().acquire_next_frame(acquire_next_frame_time) {
383                Ok(frame) => {
384                    // 拿到帧之后 —— 覆盖最常见路径
385                    // （DXGI 在 acquire 时阻塞，Ctrl+C 后立即返回，立刻检查 flag）
386                    if shutdown.load(Ordering::Relaxed) { break; }
387
388                    #[cfg(any(debug_assertions, feature = "release-log"))]
389                    {
390                        frame_count += 1;
391                        if !first_frame_logged {
392                            first_frame_logged = true;
393                            println!("✅ 首帧已到达, 进入编码流程");
394                        } else if frame_count % 30 == 0 {
395                            println!("📊 已处理 {} 帧, 累计 Timeout {} 次", frame_count, timeout_count);
396                        }
397                    }
398
399                    let elapsed = start.elapsed();
400                    // 把 elapsed 纳秒转成 HNS（100ns）
401                    hns_timestamp = elapsed.as_nanos() as i64 / 100;
402
403                    // 把 frame 的所有使用收敛到一个内层 scope。
404                    // 目的：让 frame 在 scope 结束时 drop（释放 session.dup() 的借用），
405                    // 这样后续就能 mut borrow session.encoder()。
406                    // borrowed data → owned data 一一对应：
407                    //   - frame.texture()        → ID3D11Texture2D (COM clone = AddRef)
408                    //   - frame.texture_desc()   → (w, h) u32 (Copy)
409                    //   - frame.frame_info()     → LastPresentTime u32 (Copy)
410                    //   - dirty::extract(...)    → Vec<RECT> (owned)
411                    let (bgra_texture, new_w, new_h, last_present_time, dirty_rects) = {
412                        let bgra_texture = frame.texture().clone();
413                        let desc = frame.texture_desc();
414                        let new_w = desc.Width;
415                        let new_h = desc.Height;
416                        let last_present_time = frame.frame_info().LastPresentTime;
417                        let dirty_rects = dirty::extract_dirty_reacts(&frame);
418                        (bgra_texture, new_w, new_h, last_present_time, dirty_rects)
419                    };
420
421                    // 分辨率变化检测：用户改了显示设置 / 切了屏幕时，DXGI 给的 frame 尺寸
422                    // 会和编码器初始化的尺寸不一致。必须在喂帧前重协商，否则 MFT 后续会报
423                    // MF_E_TRANSFORM_STREAM_CHANGE。
424                    {
425                        let enc = session.encoder_ref();
426                        if new_w != enc.width || new_h != enc.height {
427                            #[cfg(any(debug_assertions, feature = "release-log"))]
428                            println!(
429                                "🔄 分辨率变化: {}x{} -> {}x{}, 重新协商编码器 + 重建 compositor...",
430                                enc.width, enc.height, new_w, new_h
431                            );
432                            session.encoder().renegotiate(new_w, new_h)?;
433                            session.resize_compositor(new_w, new_h)?;
434                            // 强制 IDR：让下一帧是关键帧，客户端能立刻看到新分辨率的画面
435                            session.encoder().request_keyframe();
436                        }
437                    }
438
439                    // 跳过未实际呈现的帧：LastPresentTime == 0 表示这一帧其实没有显示过
440                    if !first_valid_frame && last_present_time == 0 {
441                        continue;
442                    }
443                    first_valid_frame = false;
444
445                    // dirty rect debug 日志（仅 debug / release-log 模式有）：
446                    //   前 5 帧每帧打详细；之后每 30 帧打一次摘要
447                    #[cfg(any(debug_assertions, feature = "release-log"))]
448                    if !dirty_rects.is_empty() {
449                        let n_rects = dirty_rects.len();
450                        let total_area = dirty::dirty_total_area(&dirty_rects);
451                        let screen_pixels = (new_w as u64) * (new_h as u64);
452                        let ratio_pct = if screen_pixels > 0 {
453                            total_area as f64 / screen_pixels as f64 * 100.0
454                        } else {
455                            0.0
456                        };
457                        if frame_count <= 5 || frame_count % 30 == 0 {
458                            println!(
459                                "📐 帧 {}: dirty={}个, 面积={}px ({:.3}%), 屏幕={}x{}",
460                                frame_count, n_rects, total_area, ratio_pct, new_w, new_h,
461                            );
462                            if frame_count <= 5 {
463                                for (i, r) in dirty_rects.iter().take(3).enumerate() {
464                                    let rw = r.right - r.left;
465                                    let rh = r.bottom - r.top;
466                                    println!(
467                                        "   rect[{}]: ({},{}) - ({},{}), {}x{}",
468                                        i, r.left, r.top, r.right, r.bottom, rw, rh,
469                                    );
470                                }
471                            }
472                        }
473                    }
474
475                    if dirty_rects.is_empty() {
476                        continue;
477                    }
478
479                    let dirty_ratio = compute_dirty_ratio(&dirty_rects, new_w, new_h);
480                    let _current_mode = activity_state.update(dirty_ratio);
481                    // 状态切换时 frame_duration_hns 可能要变（Active 用屏幕 fps，Idle 用 IDLE_FPS）
482                    frame_duration_hns = match activity_state.mode {
483                        ActivityMode::Active => get_frame_duration_hns(fps_screen),
484                        ActivityMode::Idle => get_frame_duration_hns(IDLE_FPS),
485                    };
486
487                    // Idle 模式 + 脏面积 < SKIP_THRESHOLD → 跳过整帧
488                    if activity_state.should_skip(dirty_ratio) {
489                        activity_state.total_skipped_frames += 1;
490                        #[cfg(any(debug_assertions, feature = "release-log"))]
491                        if activity_state.total_skipped_frames <= 5
492                            || activity_state.total_skipped_frames % 30 == 0
493                        {
494                            println!(
495                                "⏭️ [activity] Idle 跳过第 {} 帧 (dirty_ratio={:.4}% < {}%)",
496                                activity_state.total_skipped_frames,
497                                dirty_ratio * 100.0,
498                                SKIP_THRESHOLD * 100.0,
499                            );
500                        }
501                        continue;
502                    }
503
504                    // GPU 合成：把 prev_bgra + 当前帧脏矩形合并到 current_bgra
505                    // 编码器看到的输入 = "和上一帧几乎一样"，H.264 出极小 delta frame
506                    let composited_bgra = session.composite(&bgra_texture, &dirty_rects)?;
507
508                    #[cfg(any(debug_assertions, feature = "release-log"))]
509                    if frame_count % 30 == 0 {
510                        println!(
511                            "⏱️ [compositor] 上次 composit 耗时: {} us",
512                            session.last_composite_us()
513                        );
514                    }
515
516                    // 颜色转换：BGRA → NV12（喂的是合成后的 current_bgra）
517                    session.encoder().convert_bgra_to_nv12(&composited_bgra)?;
518
519                    // 把 NV12 数据打包成 IMFSample，准备喂给 MFT
520                    let imfsample = session.encoder().build_sample(hns_timestamp, frame_duration_hns)?;
521
522                    // 喂输入：硬件 MFT 是异步的，会在内部等 NeedInput 事件
523                    session.encoder().process_input(imfsample)?;
524
525                    // 取输出：如果 MFT 还没攒出一帧，会返回 Err(NEED_MORE_INPUT)
526                    match session.encoder().process_output() {
527                        Ok(out_sample_opt) => {
528                            #[cfg(any(debug_assertions, feature = "release-log"))]
529                            {
530                                output_count += 1;
531                                if output_count % 30 == 0 {
532                                    let enc = session.encoder_ref();
533                                    println!("📊 已编码输出 {} 帧 ({}x{}@{}fps)",
534                                        output_count, enc.width, enc.height, fps_);
535                                }
536                            }
537                            if let Some(out_sample) = out_sample_opt {
538                                // avcc_to_annex_b_into 自动检测 MFT 输出的格式：
539                                //   - NVENC 走 Annex-B（00 00 00 01 起始码），avcc_to_annex_b 透传 + IDR 前注入 SPS/PPS
540                                //   - 软编走 AVCC（4 字节 BE 长度前缀），avcc_to_annex_b 转成起始码 + 注入
541                                // ConvertToContiguousBuffer 强制拷出 sample 的真实内容，避免依赖
542                                // buffer_count=1 时 GetBufferByIndex(0) 拿 MFT 内部引用的风险。
543                                let contiguous_result = out_sample.ConvertToContiguousBuffer();
544                                if let Ok(buffer) = contiguous_result {
545                                    let mut data_ptr: *mut u8 = std::ptr::null_mut();
546                                    let mut max_len: u32 = 0;
547                                    let mut cur_len: u32 = 0;
548                                    if buffer.Lock(&mut data_ptr, Some(&mut max_len as *mut _), Some(&mut cur_len as *mut _)).is_ok() {
549                                        let avcc_bytes = std::slice::from_raw_parts(data_ptr, cur_len as usize);
550
551                                        // 硬件 MFT（D3D11 Video Encoder / NVENC）不把 SPS/PPS 写进 bitstream，
552                                        // 而是写到 output MediaType 的 MF_MT_MPEG_SEQUENCE_HEADER 属性里——
553                                        // 但要等编出首帧之后才填。create_hw 末尾的 extract 拿不到 blob，
554                                        // 这里在 sps_pps_annex_b 为空时懒提取一次：调 GetOutputCurrentType(0)
555                                        // 拿当前 output MediaType 再提取。
556                                        // 软编走 create_sw 路径，extract 已经拿到 blob，这条分支进不来。
557                                        if session.encoder().sps_pps_annex_b().is_empty() {
558                                            session.encoder().lazy_extract_sps_pps_from_output_type();
559                                        }
560
561                                        let sps_pps = session.encoder().sps_pps_annex_b();
562
563                                        // avcc → Annex-B 写到预分配 BytesMut，这是整条热路径唯一的 memcpy
564                                        // （起始码字节是新生成的），之后 freeze → Bytes 跨 mpsc 推网络线程，全程 refcount++ 零拷贝
565                                        let mut annex_b_buf = BytesMut::with_capacity(
566                                            avcc_bytes.len() + sps_pps.len() + 16,
567                                        );
568                                        let is_keyframe =
569                                            avcc_to_annex_b_into(&mut annex_b_buf, avcc_bytes, sps_pps);
570
571
572
573                                        if is_keyframe {
574                                            H264_struct.keyframe_index_add();
575                                        } else {
576                                            H264_struct.delta_index_add();
577                                        }
578
579
580                                        let annex_b_bytes = annex_b_buf.freeze();
581
582                                        let frame_seq = wire_frame_index as u32;
583                                        let annex_b_len = annex_b_bytes.len();
584                                        let frame = EncodedFrame {
585                                            is_keyframe,
586                                            annex_b: annex_b_bytes,
587                                            frame_seq,
588                                            keyframe_index: H264_struct.keyframe_index,
589                                            delta_index: H264_struct.delta_index,
590                                        };
591
592                                        // try_send：满载时返回 Err 走"丢老帧"语义。
593                                        // 对实时屏幕捕获这是正确的背压策略（旧帧到对端也用不上），
594                                        // 而 blocking_send / send().await 都会卡住 capture 主循环。
595                                        session.frame_tx.try_send(frame).ok();
596                                        wire_frame_index += 1;
597
598                                        #[cfg(any(debug_assertions, feature = "release-log"))]
599                                        {
600                                            if frame_count == 1 || frame_count % 30 == 0 {
601                                                println!(
602                                                    "📦 frame #{}: is_keyframe={}, annex_b {} 字节, frame_seq={}",
603                                                    frame_count, is_keyframe,
604                                                    annex_b_len,
605                                                    frame_seq,
606                                                );
607                                            }
608                                        }
609
610                                        let _ = buffer.Unlock();
611                                    }
612                                }
613                                session.encoder().recycle_output_sample(Some(out_sample));
614                            }
615                            // 回收用过的 NV12 纹理
616                            session.encoder().recycle_texture();
617                        }
618                        Err(err) if err.code() == MF_E_TRANSFORM_NEED_MORE_INPUT => {
619                            // NVENC 类硬件编码器是 1:1 的：收到 NEED_MORE_INPUT 表示 MFT 没消费任何输入，
620                            // 刚才 push 进 pending_textures 的那一帧其实 MFT 没吃，应该立刻还回池子。
621                            // 否则 pending_textures 单调增长、nv12_texture_pool 单调缩水，跑久了池子空了崩。
622                            // 软编走不到这条分支（软编同步消费，不会报 NEED_MORE_INPUT）。
623                            session.encoder().recycle_texture();
624                            continue;
625                        }
626                        Err(err) if err.code() == MF_E_TRANSFORM_STREAM_CHANGE => {
627                            // MFT 要求重新协商输出格式（屏幕分辨率变化等）。这次循环的帧已经走过
628                            // convert_bgra_to_nv12，旧尺寸的 NV12 纹理会在 renegotiate 里被 drain 掉重建，
629                            // 下一帧直接用新尺寸。
630                            session.encoder().recycle_texture();
631                            let enc = session.encoder_ref();
632                            if new_w != enc.width || new_h != enc.height {
633                                #[cfg(any(debug_assertions, feature = "release-log"))]
634                                println!(
635                                    "🔄 MFT 要求重协商 ({}x{} -> {}x{}), 重新协商编码器...",
636                                    enc.width, enc.height, new_w, new_h
637                                );
638                                session.encoder().renegotiate(new_w, new_h)?;
639                                // 强制 IDR：和 proactive 分支对齐
640                                session.encoder().request_keyframe();
641                            } else {
642                                #[cfg(any(debug_assertions, feature = "release-log"))]
643                                println!(
644                                    "⚠️ MFT 要求重协商但尺寸未变 ({}x{}), skipping",
645                                    enc.width, enc.height
646                                );
647                            }
648                            continue;
649                        }
650                        Err(err) => {
651                            match SessionError::classify(&err) {
652                                SessionError::Recoverable(reason) => {
653                                    #[cfg(any(debug_assertions, feature = "release-log"))]
654                                    println!("⚠️ 编码器错误（可恢复）: {}, 尝试 recover...", reason);
655                                    match recover_state.record_failure() {
656                                        RecoverAction::Backoff(ms) => {
657                                            std::thread::sleep(std::time::Duration::from_millis(ms));
658                                        }
659                                        RecoverAction::GiveUp => {
660                                            return Err(format!("30 秒内失败 {} 次, 放弃", recover_state.consecutive_failures()).into());
661                                        }
662                                    }
663                                    session.recover()?;
664                                    recover_state.record_success();
665                                    #[cfg(any(debug_assertions, feature = "release-log"))]
666                                    println!("✅ Session 恢复成功 (累计 {} 次)",
667                                        recover_state.total_recoveries());
668                                }
669                                SessionError::Transient(_) => {
670                                    // 业务流错误，不该到这里（已经在前面 match MF_E_TRANSFORM_NEED_MORE_INPUT 处理）
671                                    return Err(err.into());
672                                }
673                                SessionError::Fatal(reason) => {
674                                    return Err(format!("编码器致命错误: {} (HRESULT 0x{:08X})",
675                                        reason, err.code().0 as u32).into());
676                                }
677                            }
678                        }
679                    }
680                }
681                Err(DupError::Timeout) => {
682                    #[cfg(any(debug_assertions, feature = "release-log"))]
683                    {
684                        timeout_count += 1;
685                        if timeout_count == 1 || timeout_count % 30 == 0 {
686                            println!(
687                                "⏳ 等待帧超时 ({}/{} 次, 屏幕可能无变化 — 这是正常状态)",
688                                timeout_count, frame_count + timeout_count
689                            );
690                        }
691                    }
692                    continue;
693                }
694                Err(DupError::AccessLost) => {
695                    // 桌面切换/锁屏/全屏独占程序抢走了 DXGI 句柄。走完整的 recover 路径
696                    // （D3D device / DXGI dup / Encoder 全部重建），能 cover 笔记本睡眠唤醒等
697                    // 同时让 D3D device 也失效的场景。
698                    dxgi_consecutive_failures += 1;
699
700                    // 连续 N 次失败（不限 AccessLost 也包括下面的 DXGI 错误），
701                    // 如果锁屏 watcher 存在，模拟锁屏暂停：release session → 等解锁事件 → recreate。
702                    // 没有 watcher 时走原来的 recover_state 节流路径。
703                    if dxgi_consecutive_failures >= DXGI_FALLBACK_THRESHOLD
704                        && lockscreen.is_some()
705                    {
706                        #[cfg(any(debug_assertions, feature = "release-log"))]
707                        println!(
708                            "🔁 DXGI 连续失败 {} 次 >= 阈值 {}, 切换到 fallback: release + 等解锁事件",
709                            dxgi_consecutive_failures, DXGI_FALLBACK_THRESHOLD
710                        );
711                        session.release();
712                        // 不进入 wait_until_active（让外层 paused 分支统一处理）
713                        // 通过持续循环让 is_paused() 检查进入 paused 分支
714                        std::thread::sleep(std::time::Duration::from_millis(500));
715                        recover_state = RecoverState::new();
716                        continue;
717                    }
718
719                    match recover_state.record_failure() {
720                        RecoverAction::Backoff(ms) => {
721                            std::thread::sleep(std::time::Duration::from_millis(ms));
722                        }
723                        RecoverAction::GiveUp => {
724                            return Err(format!("30 秒内失败 {} 次, 放弃", recover_state.consecutive_failures()).into());
725                        }
726                    }
727                    match session.recover() {
728                        Ok(()) => {
729                            recover_state.record_success();
730                            dxgi_consecutive_failures = 0; // recover 成功 → 重置失败计数
731                            #[cfg(any(debug_assertions, feature = "release-log"))]
732                            println!("✅ Session 恢复成功 (累计 {} 次)",
733                                recover_state.total_recoveries());
734                        }
735                        Err(e) => {
736                            return Err(format!("Session recover 失败: {}", e).into());
737                        }
738                    }
739                }
740                Err(e) => {
741                    // 其他 windows-capture 错误：分类后看是否可恢复。
742                    // 走 SessionError::classify 分类（直接当 Recoverable 太乐观：某些 DXGI 错误码是 Fatal，
743                    // 参数错、驱动不支持，硬 recover 浪费 1 秒 +）。
744                    #[cfg(any(debug_assertions, feature = "release-log"))]
745                    println!("❌ DXGI 错误: {}, 分类: AccessLost 之外的未知", e);
746                    dxgi_consecutive_failures += 1;
747
748                    if dxgi_consecutive_failures >= DXGI_FALLBACK_THRESHOLD
749                        && lockscreen.is_some()
750                    {
751                        #[cfg(any(debug_assertions, feature = "release-log"))]
752                        println!(
753                            "🔁 DXGI 连续失败 {} 次 >= 阈值 {}, 切换到 fallback",
754                            dxgi_consecutive_failures, DXGI_FALLBACK_THRESHOLD
755                        );
756                        session.release();
757                        std::thread::sleep(std::time::Duration::from_millis(500));
758                        recover_state = RecoverState::new();
759                        continue;
760                    }
761
762                    // 分类：尝试 downcast 到 windows::core::Error，
763                    // 成功就调 SessionError::classify，失败兜底 Fatal（和 classify() 的未知错误兜底对齐）。
764                    let classification = {
765                        let mut src: Option<&(dyn std::error::Error + 'static)> = Some(&e);
766                        let mut found_win_err: Option<&windows::core::Error> = None;
767                        while let Some(s) = src {
768                            if let Some(we) = s.downcast_ref::<windows::core::Error>() {
769                                found_win_err = Some(we);
770                                break;
771                            }
772                            src = s.source();
773                        }
774                        match found_win_err {
775                            Some(we) => SessionError::classify(we),
776                            None => SessionError::Fatal(
777                                format!("DupError 无法 downcast 到 windows::core::Error (DupError msg: {})", e)
778                            ),
779                        }
780                    };
781                    match classification {
782                        SessionError::Fatal(reason) => {
783                            return Err(format!(
784                                "DXGI 致命错误: {} (DupError 链: {})",
785                                reason, e
786                            ).into());
787                        }
788                        SessionError::Recoverable(_) | SessionError::Transient(_) => {
789                            match recover_state.record_failure() {
790                                RecoverAction::Backoff(ms) => {
791                                    std::thread::sleep(std::time::Duration::from_millis(ms));
792                                }
793                                RecoverAction::GiveUp => {
794                                    return Err(format!("30 秒内失败 {} 次, 放弃", recover_state.consecutive_failures()).into());
795                                }
796                            }
797                            if let Err(rerr) = session.recover() {
798                                return Err(format!("DXGI 错误且 recover 失败: orig={}, recover={}", e, rerr).into());
799                            }
800                            recover_state.record_success();
801                            dxgi_consecutive_failures = 0;
802                            #[cfg(any(debug_assertions, feature = "release-log"))]
803                            println!("✅ Session 恢复成功 (累计 {} 次)", recover_state.total_recoveries());
804                        }
805                    }
806                }
807            }
808        }
809    }
810
811    Ok(())
