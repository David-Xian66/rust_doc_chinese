1//! 捕获主循环：把屏幕抓下来 + 编码成 H.264
2//!
3//! 数据流（一帧）：
4//! ```text
5//! DXGI Desktop Duplication (dup.acquire_next_frame)
6//!   └─ ID3D11Texture2D (BGRA, 显存里)
7//!       └─ encoder.convert_bgra_to_nv12:  BGRA → NV12（GPU VideoProcessor）
8//!       └─ encoder.build_sample:          NV12 → IMFSample
9//!       └─ encoder.process_input:         IMFSample → MFT
10//!       └─ encoder.process_output:        MFT → IMFSample（含 H.264 字节）
11//!       └─ 这里把 H.264 字节读出来（目前只是 println，没真的发到网络）
12//! ```
13//!
14//! # 重要术语
15//! - **DXGI Desktop Duplication**: Windows 7+ 提供的桌面复制 API，
16//!   应用层可以把"屏幕正在显示的画面"整帧拷到自己的显存里。
17//! - **HNS (100-nanosecond units)**: Media Foundation 时间单位，1 HNS = 100ns。
18//!   时间戳和帧时长都用这个单位，方便和 MF 内部时间轴对齐。

48/// [C1] RAII guard for `CoUninitialize`。
49/// `CoInitializeEx` 没有自动对偶（不像 `Box` 之于 `new`），必须手动调。
50/// 用 RAII 保证 panic / 早返回 / 正常退出都走 cleanup。
51struct CoUninitGuard;
52impl Drop for CoUninitGuard {
53    fn drop(&mut self) {
54        unsafe {
55            // CoUninitialize 是 O(1) 且 idempotent，调用失败也不影响 cleanup
56            let _ = CoUninitialize();
57        }
58    }
59}

61/// DXGI `AcquireNextFrame` 的等待超时（毫秒）。
62///
63/// 不是"每帧间隔"，而是"如果这么久都没新帧就返回 Timeout 让你继续循环"。
64/// 乘 1.5 是经验值，给屏幕没变化时一个不会立刻返回、又能及时退出的等待时间。
65fn get_acquire_next_frame_time(fps_: u32) -> u32 {
66    (1000.0 / fps_ as f32 * 1.5) as u32
67}

69/// 一帧的时长（HNS，100ns 单位）。
70/// 比如 30fps 就是 1/30 秒 = 3,333,333 HNS。
71fn get_frame_duration_hns(fps_: u32) -> i64 {

79/// 用户活动模式。
80///
81/// - **Active**：用户"正在操作"（拖动、输入、滚动）。FPS 拉满到配置值（如 59 fps），
82///   方案 B 启用（GPU 合成复用），方案 A **禁用**（不能漏关键帧）。
83/// - **Idle**：用户静止超过 N 秒。FPS 降到 IDLE_FPS，方案 B 仍启用，
84///   方案 A 启用（脏面积 < SKIP_THRESHOLD 时跳过整帧，进一步省带宽）。
85#[derive(Debug, Clone, Copy, PartialEq, Eq)]
86enum ActivityMode {
87    Active,
88    Idle,
89}
90
91/// 状态机参数（按 v3 报告 §4.1 默认值，实测后调）
92///
93/// - `ACTIVE_DIRTY_THRESHOLD` (0.5%)：脏面积低于此值算"静止信号"
94/// - `IDLE_ENTRY_FRAMES` (30 帧 ≈ 0.5 秒 @ 59fps)：连续 N 帧都低于阈值才进入 Idle
95/// - `IDLE_FPS` (5 fps)：Idle 模式 FPS（人眼看静止画面 5 fps 察觉不到）
96/// - `SKIP_THRESHOLD` (0.5%)：Idle 模式下脏面积低于此值 → 跳过整帧
97const ACTIVE_DIRTY_THRESHOLD: f64 = 0.005;
98const IDLE_ENTRY_FRAMES: u32 = 30;
99const IDLE_FPS: u32 = 5;
100const SKIP_THRESHOLD: f64 = 0.005;
101
102/// 用户活动状态机。
103struct ActivityState {
104    mode: ActivityMode,
105    /// 连续 dirty 面积 <= ACTIVE_DIRTY_THRESHOLD 的帧数（Active → Idle 判定）
106    below_threshold_count: u32,
107    /// 累计统计（给 release-log 用）
108    total_active_frames: u64,
109    total_idle_frames: u64,
110    total_skipped_frames: u64,
111    total_active_to_idle: u64,
112    total_idle_to_active: u64,
113}
114
115impl ActivityState {
116    fn new() -> Self {
117        Self {
118            mode: ActivityMode::Active, // 启动默认 Active
119            below_threshold_count: 0,
120            total_active_frames: 0,
121            total_idle_frames: 0,
122            total_skipped_frames: 0,
123            total_active_to_idle: 0,
124            total_idle_to_active: 0,
125        }
126    }
127
128    /// 每帧调一次，输入 dirty_ratio，返回当前模式（可能发生切换）。
129    ///
130    /// `dirty_ratio` ∈ [0.0, 1.0]，表示脏矩形占屏幕面积的比例。
131    fn update(&mut self, dirty_ratio: f64) -> ActivityMode {
132        let prev_mode = self.mode;
133        match self.mode {
134            ActivityMode::Active => {
135                self.total_active_frames += 1;
136                if dirty_ratio <= ACTIVE_DIRTY_THRESHOLD {
137                    self.below_threshold_count += 1;
138                    if self.below_threshold_count >= IDLE_ENTRY_FRAMES {
139                        self.mode = ActivityMode::Idle;
140                        self.total_active_to_idle += 1;
141                    }
142                } else {
143                    self.below_threshold_count = 0;
144                }
145            }
146            ActivityMode::Idle => {
147                self.total_idle_frames += 1;
148                if dirty_ratio > ACTIVE_DIRTY_THRESHOLD {
149                    self.mode = ActivityMode::Active;
150                    self.below_threshold_count = 0;
151                    self.total_idle_to_active += 1;
152                }
153            }
154        }
155        // release-log：状态切换时打日志
156        #[cfg(any(debug_assertions, feature = "release-log"))]
157        if self.mode != prev_mode {
158            let mode_str = match self.mode {
159                ActivityMode::Active => "Active",
160                ActivityMode::Idle => "Idle",
161            };
162            let from_str = match prev_mode {
163                ActivityMode::Active => "Active",
164                ActivityMode::Idle => "Idle",
165            };
166            println!(
167                "🔀 [activity] 状态切换: {} -> {} (连续 {} 帧 < {:.2}%, dirty_ratio={:.4}%)",
168                from_str,
169                mode_str,
170                self.below_threshold_count,
171                ACTIVE_DIRTY_THRESHOLD * 100.0,
172                dirty_ratio * 100.0,
173            );
174        }
175        self.mode
176    }
177
178    /// Idle 模式 + 脏面积 < SKIP_THRESHOLD → 跳过整帧（A 启用条件）
179    fn should_skip(&self, dirty_ratio: f64) -> bool {
180        self.mode == ActivityMode::Idle && dirty_ratio < SKIP_THRESHOLD
181    }
182}

184/// 把 dirty_rects 列表的覆盖面积占屏幕的百分比算出来
185fn compute_dirty_ratio(dirty_rects: &[RECT], width: u32, height: u32) -> f64 {
186    if width == 0 || height == 0 || dirty_rects.is_empty() {
187        return 0.0;
188    }
189    let total_pixels = dirty::dirty_total_area(dirty_rects);
190    let screen_pixels = (width as u64) * (height as u64);
191    if screen_pixels == 0 {
192        return 0.0;
193    }
194    total_pixels as f64 / screen_pixels as f64
195}

197/// 主入口：选屏幕 → 起 DXGI 会话 → 死循环抓帧 + 编码。
198///
199/// `enumerate_index`: 屏幕索引（来自 `screen::screen_list_all` 的 HashMap key）
200/// `fps_`: 目标帧率（会和屏幕刷新率取较小值）
201/// `fps_screen`: 当前屏幕的刷新率
202/// `bitrate_bps`: 目标平均码率（bps 单位，1 Mbps = 1_000_000）
203/// `profile`: [B3] H.264 Profile（High/Main/Baseline）
204/// `shutdown`: [C1] 关闭信号 flag。Ctrl+C handler 在 main.rs 里设，主循环检查。
205/// `lockscreen`: [L1] 锁屏监听器（可选）。传 None 时禁用锁屏暂停特性。
206pub fn capture_main(
207    enumerate_index: usize,
208    fps_: usize,
209    fps_screen: u32,
210    bitrate_bps: u32,
211    profile: H264Profile,
212    shutdown: Arc<AtomicBool>,
213    lockscreen: Option<LockscreenWatcher>,
214) -> Result<(), Box<dyn std::error::Error>> {
215    unsafe {
216        // CoInitializeEx: 初始化当前线程的 COM 运行时。
217        // COINIT_MULTITHREADED = 公寓模型为 MTA（多线程），所有 COM 对象跨线程访问安全。
218        // 必须先调一次，后面所有 COM 调用才能用。
219        CoInitializeEx(None, COINIT_MULTITHREADED).ok()?;
220    }
221    // [C1] RAII 配对：函数返回（包括 panic / 早返回）自动调 CoUninitialize
222    let _co_uninit = CoUninitGuard;
223
224    // [H1 修复] 删调试残留：原来这里有一行 `println!("fffff");`，明显是 dev 阶段
225    //   调试用的占位字符串（应该测帧号 / 首帧日志等），写死了 "fffff"。
226    //   release profile 也会打，每进程启动 1 次抢 stdout 锁 + 拼字符串 —— 没有任何信息量。
227    //   V2 审查时 grep 出来删除。
228
229    // [C2] 创建会话（一次性把 D3D11 device + DXGI duplication + H.264 编码器都建好）
230    // session 把这三者打包成一个可恢复的单元，错误时可以整个重建。
231    let mut session = CaptureSession::new(
232        enumerate_index,
233        fps_ as u32,
234        bitrate_bps,
235        profile,
236    )?;
237    let mut recover_state = RecoverState::new();
238    // [H2 修复] 加 cfg gate（首次启动只打印一次，开销可忽略，但保持风格一致）
239    #[cfg(any(debug_assertions, feature = "release-log"))]
240    println!("✅ 会话已建立: {}x{} @{}fps, 码率 {} bps",
241        session.width(), session.height(), session.fps(), session.bitrate());
242
243    // [L1] DXGI 持续失败计数器。
244    // 连续 N 次 AccessLost / SessionError 视为"持续失败"，自动进入 fallback
245    // (释放 session + 阻塞等 WTS 解锁事件 + recreate)。
246    const DXGI_FALLBACK_THRESHOLD: u32 = 3;
247    let mut dxgi_consecutive_failures: u32 = 0;
248
249    // [D1 Phase 2] 用户活动状态机（Active/Idle 模式切换）
250    let mut activity_state = ActivityState::new();
251    // [H2 修复] 加 cfg gate（启动时只打一次）
252    #[cfg(any(debug_assertions, feature = "release-log"))]
253    println!(
254        "🎚️ [activity] 状态机已启动: 阈值={}%, 进入 Idle 需 {} 帧, Idle FPS={}, 跳过阈值={}%",
255        ACTIVE_DIRTY_THRESHOLD * 100.0,
256        IDLE_ENTRY_FRAMES,
257        IDLE_FPS,
258        SKIP_THRESHOLD * 100.0,
259    );
260
261    let mut first_valid_frame = true;
262
263    // [release-log] liveness 统计变量。开了 release-log 才有，不开连变量都不存在。
264    #[cfg(any(debug_assertions, feature = "release-log"))]
265    let mut frame_count: u64 = 0;
266    #[cfg(any(debug_assertions, feature = "release-log"))]
267    let mut output_count: u64 = 0;
268    #[cfg(any(debug_assertions, feature = "release-log"))]
269    let mut timeout_count: u64 = 0;
270
271    // [V6-R2] wire format 计数器（无条件存在，release 模式也用）
272    let mut wire_frame_index: u64 = 0;
273    #[cfg(any(debug_assertions, feature = "release-log"))]
274    let mut first_frame_logged = false;
275
276    // [release-log] 进入主循环的提示。生产构建（不开 release-log）不会有这行。
277    //   acquire 超时现在是 mode 相关的（Active ≈ 25ms / Idle ≈ 300ms），不在启动日志打固定值
278    #[cfg(any(debug_assertions, feature = "release-log"))]
279    println!(
280        "📡 进入捕获主循环, 屏幕刷新率={}fps, 配置 FPS={}",
281        fps_screen, fps_
282    );
283
284    // [MEMORY_OPTIMIZATION_V2.md §5] 死变量注释：原 capture.rs 死代码 output_buffer
285    //   注释掉后 input_ids / output_ids / h264_status 再也没有任何引用，跟着一起注释。
286    // // MFT 的 stream id 都用 0（只有一个输入流、一个输出流）
287    // let input_ids = vec![0u32];
288    // let output_ids = vec![0u32];
289    // let mut h264_status = 0u32;
290
291    // OnceLock 保证整个进程里只 init 一次。
292    // start 用作时间戳的基准点（进程启动时刻）
293    static CAPTURE_START: OnceLock<Instant> = OnceLock::new();
294    let start = *CAPTURE_START.get_or_init(Instant::now);
295    let mut hns_timestamp: i64;
296
297    let mut acquire_next_frame_time;
298    // frame_duration_hns 在 dirty_rects 非空后赋（依赖 activity_state.mode），不在这儿赋
299    let mut frame_duration_hns: i64;
300    // [L1 + H5 修复] 不再用 `session_released` 本地变量，统一用 `session.is_released()` 作为
301    //   single source of truth。两条状态一旦走偏会触发 panic（session.dup() expect panic）。
302    //   删掉本地变量后所有"是否已 release"的判断都走 session 自己。
303
304    loop {
305        // [C1] 检查点 1: 循环顶部 —— 每次循环先检查 shutdown flag
306        if shutdown.load(Ordering::Relaxed) {
307            // [H2 修复] 加 cfg gate
308            #[cfg(any(debug_assertions, feature = "release-log"))]
309            println!("🛑 主循环退出 (shutdown flag set)");
310            break;
311        }
312
313        // [H5 修复] 检查点 1.5: session 已 release 但当前不在 paused 分支时，
314        //   立即重建（避免主循环走到 session.dup() 触发 expect panic）。
315        //   触发场景：DXGI fallback 释放 session 后 `continue`，但此刻用户没锁屏
316        //   （is_paused() = false），会跳过下面的 paused 分支直接走主路径。
317        if session.is_released()
318            && (lockscreen.is_none() || !lockscreen.as_ref().unwrap().is_paused())
319        {
320            // [H2 修复] 加 cfg gate（H5 触发是异常路径，但仍要保持 release 0 开销）
321            #[cfg(any(debug_assertions, feature = "release-log"))]
322            println!("🏗️ [H5] session 已 release 但未进入 paused 分支，立即重建");
323            if let Err(e) = session.recreate() {
324                return Err(format!("H5 立即重建 session 失败: {}", e).into());
325            }
326            dxgi_consecutive_failures = 0;
327        }
328
329        // [L1] 锁屏 / 用户切换暂停分支
330        //   触发：LockscreenWatcher 检测到 WTS_LOCK / WTS_LOGOFF
331        //   行为：release session → 阻塞等解锁 → recreate + force IDR
332        //   不传 lockscreen watcher 时跳过这个分支
333        if let Some(lw) = &lockscreen
334            && lw.is_paused()
335        {
336            // session 还没 release → release 一次
337            if !session.is_released() {
338                session.release();
339                // [H2 修复] 加 cfg gate（锁屏事件偶尔触发，开销可忽略但保持一致）
340                #[cfg(any(debug_assertions, feature = "release-log"))]
341                println!(
342                    "🔒 [L1] 进入暂停 (会话状态={:?}), session 已释放",
343                    lw.current_state()
344                );
345            }
346            // 真阻塞等到 is_paused() 变 false（解锁事件触发）
347            lw.wait_until_active(&shutdown)?;
348            if shutdown.load(Ordering::Relaxed) {
349                break;
350            }
351            // 重建 session + force IDR
352            // [H5 修复] recreate 失败时显式处理：返回错误而不是用 `?` 早返回。
353            //   recreate 失败说明环境真的坏了，留着半完成状态会越陷越深。
354            if let Err(e) = session.recreate() {
355                return Err(format!("L1 解锁后 recreate 失败: {}", e).into());
356            }
357            // 重置 DXGI 失败计数（重建后从干净状态开始）
358            dxgi_consecutive_failures = 0;
359            // [H2 修复] 加 cfg gate
360            #[cfg(any(debug_assertions, feature = "release-log"))]
361            println!("🔓 [L1] 解锁/切换完成, session 重建, 继续捕获");
362            continue;
363        }
364        // 离开 paused 分支时，session 应该是 active 状态
365        // （首次循环 / 解锁后到这里时，session 必须非 release）
366
367        // [D1 Phase 2] 根据当前 mode 动态调整 acquire 超时
368        //   - Active: 短超时（25ms @ 59fps）
369        //   - Idle:   长超时（300ms @ 5fps），降低 CPU 占用
370        //   frame_duration_hns 在 dirty_rects 非空后用 activity_state.mode 算（在下面）
371        acquire_next_frame_time = match activity_state.mode {
372            ActivityMode::Active => get_acquire_next_frame_time(fps_screen),
373            ActivityMode::Idle => (1000.0 / IDLE_FPS as f32 * 1.5) as u32,
374        };
375        unsafe {
376            // acquire_next_frame: 阻塞等待下一帧，最多等 acquire_next_frame_time 毫秒
377            // 返回值：
378            //   Ok(frame)         - 拿到一帧
379            //   Err(Timeout)      - 超时但没新帧，屏幕没变化
380            //   Err(AccessLost)   - 桌面切换/锁屏/DXGI 被其他程序抢走，需要 recreate
381            match session.dup().acquire_next_frame(acquire_next_frame_time) {
382                Ok(frame) => {
383                    // [C1] 检查点 2: 拿到帧之后 —— 覆盖最常见路径
384                    // （DXGI 在 acquire 时阻塞，Ctrl+C 后立即返回，立刻检查 flag）
385                    if shutdown.load(Ordering::Relaxed) { break; }
386
387                    // [release-log] 首帧提示 + 每 30 帧进度
388                    #[cfg(any(debug_assertions, feature = "release-log"))]
389                    {
390                        frame_count += 1;
391                        if !first_frame_logged {
392                            first_frame_logged = true;
393                            println!("✅ 首帧已到达, 进入编码流程");
394                        } else if frame_count % 30 == 0 {
395                            println!("📊 已处理 {} 帧, 累计 Timeout {} 次", frame_count, timeout_count);
396                        }
397                    }
398
399                    let elapsed = start.elapsed();
400                    // 把 elapsed 纳秒转成 HNS（100ns）
401                    hns_timestamp = elapsed.as_nanos() as i64 / 100;
402
403                    // [C2] 把 frame 的所有使用收敛到一个内层 scope。
404                    //   目的：让 frame 在 scope 结束时 drop（释放 session.dup() 的借用），
405                    //   这样后续就能 mut borrow session.encoder()。
406                    //   borrowed data → owned data 一一对应：
407                    //     - frame.texture()        → ID3D11Texture2D (COM clone = AddRef)
408                    //     - frame.texture_desc()   → (w, h) u32  (Copy)
409                    //     - frame.frame_info()     → LastPresentTime u32 (Copy)
410                    //     - dirty::extract(...)    → Vec<RECT> (owned)
411                    let (bgra_texture, new_w, new_h, last_present_time, dirty_rects) = {
412                        let bgra_texture = frame.texture().clone();  // AddRef COM 对象，独立于 frame
413                        let desc = frame.texture_desc();
414                        let new_w = desc.Width;
415                        let new_h = desc.Height;
416                        let last_present_time = frame.frame_info().LastPresentTime;
417                        let dirty_rects = dirty::extract_dirty_reacts(&frame);
418                        (bgra_texture, new_w, new_h, last_present_time, dirty_rects)
419                    };
420                    // frame 在这里 drop，session.dup() 的借用释放
421
422                    // 分辨率变化检测：用户改了显示设置 / 切了屏幕时，DXGI 给的 frame 尺寸
423                    // 会和编码器初始化的尺寸不一致。必须在喂帧前重协商，否则 MFT 后续会报
424                    // MF_E_TRANSFORM_STREAM_CHANGE。proactive 检查在 STREAM_CHANGE 之前抓到。
425                    {
426                        let enc = session.encoder_ref();
427                        if new_w != enc.width || new_h != enc.height {
428                            // [H2 修复] 加 cfg gate：release 模式 0 开销（避免每秒抢 stdout 锁）
429                            #[cfg(any(debug_assertions, feature = "release-log"))]
430                            println!(
431                                "🔄 分辨率变化: {}x{} -> {}x{}, 重新协商编码器 + 重建 compositor...",
432                                enc.width, enc.height, new_w, new_h
433                            );
434                            session.encoder().renegotiate(new_w, new_h)?;
435                            // [D1 Phase 1] 同步重建 compositor（prev_bgra / current_bgra 都失效）
436                            session.resize_compositor(new_w, new_h)?;
437                            // [H6 修复] 强制 IDR：让下一帧是关键帧，客户端能立刻看到新分辨率的画面。
438                            //   原代码没调，客户端会先看到 P 帧混乱（参考帧尺寸已经变了但 P 帧还引用旧尺寸）。
439                            session.encoder().request_keyframe();
440                        }
441                    }
442
443                    // 跳过未实际呈现的帧
444                    // LastPresentTime == 0 表示这一帧其实没有显示过（GPU 还没把它送到屏幕）
445                    if !first_valid_frame && last_present_time == 0 {
446                        continue;
447                    }
448                    first_valid_frame = false;
449
450                    // 检查变化区域（首帧必须编码，作为 I 帧）
451
452                    // [D1 准备] dirty rect debug 日志 —— 观察实际数据，为方案 B 设计提供依据。
453                    //   之前 MEMORY_OPTIMIZATION_V2.md §3.2 把所有热路径 println! 删了，
454                    //   现在恢复 dirty rect 相关的日志（仅 debug / release-log 模式有，release 模式 0 开销）。
455                    //
456                    //   打印规则：
457                    //     - 前 5 帧每帧打：详细（数量、面积、占比、前 3 个 rect 位置/尺寸）
458                    //     - 之后每 30 帧打一次摘要：数量、面积、占比
459                    //
460                    //   不打 dirty_rects.is_empty() 的情况（屏幕完全静止），因为那种 case 已经在前面 continue 了。
461                    #[cfg(any(debug_assertions, feature = "release-log"))]
462                    if !dirty_rects.is_empty() {
463                        let n_rects = dirty_rects.len();
464                        let total_area = dirty::dirty_total_area(&dirty_rects);
465                        let screen_pixels = (new_w as u64) * (new_h as u64);
466                        let ratio_pct = if screen_pixels > 0 {
467                            total_area as f64 / screen_pixels as f64 * 100.0
468                        } else {
469                            0.0
470                        };
471                        if frame_count <= 5 || frame_count % 30 == 0 {
472                            println!(
473                                "📐 帧 {}: dirty={}个, 面积={}px ({:.3}%), 屏幕={}x{}",
474                                frame_count, n_rects, total_area, ratio_pct, new_w, new_h,
475                            );
476                            // 前 5 帧打前 3 个 rect 的具体位置/尺寸（观察脏矩形分布规律）
477                            if frame_count <= 5 {
478                                for (i, r) in dirty_rects.iter().take(3).enumerate() {
479                                    let rw = r.right - r.left;
480                                    let rh = r.bottom - r.top;
481                                    println!(
482                                        "   rect[{}]: ({},{}) - ({},{}), {}x{}",
483                                        i, r.left, r.top, r.right, r.bottom, rw, rh,
484                                    );
485                                }
486                            }
487                        }
488                    }
489
490                    if dirty_rects.is_empty() {
491                        continue;
492                    }
493
494                    // [D1 Phase 2] 计算脏面积占比，更新状态机
495                    let dirty_ratio = compute_dirty_ratio(&dirty_rects, new_w, new_h);
496                    let _current_mode = activity_state.update(dirty_ratio);
497                    // 状态切换时 frame_duration_hns 可能要变（Active 用屏幕 fps，Idle 用 IDLE_FPS）
498                    frame_duration_hns = match activity_state.mode {
499                        ActivityMode::Active => get_frame_duration_hns(fps_screen),
500                        ActivityMode::Idle => get_frame_duration_hns(IDLE_FPS),
501                    };
502
503                    // [D1 Phase 2] Idle 模式 + 脏面积 < SKIP_THRESHOLD → 跳过整帧（A 启用）
504                    if activity_state.should_skip(dirty_ratio) {
505                        activity_state.total_skipped_frames += 1;
506                        #[cfg(any(debug_assertions, feature = "release-log"))]
507                        if activity_state.total_skipped_frames <= 5
508                            || activity_state.total_skipped_frames % 30 == 0
509                        {
510                            println!(
511                                "⏭️ [activity] Idle 跳过第 {} 帧 (dirty_ratio={:.4}% < {}%)",
512                                activity_state.total_skipped_frames,
513                                dirty_ratio * 100.0,
514                                SKIP_THRESHOLD * 100.0,
515                            );
516                        }
517                        continue;
518                    }
519
520                    // [修复 §3.2] 热路径 println! 全部删除：
521                    //   每帧 4~6 次 println! = 每帧 4~6 次 stdout 锁 + 4~6 次小 String 堆分配。
522                    //   调试时手动加回，或者等接网络层时统一用 log crate 接日志方案。
523
524                    // [D1 Phase 1] GPU 合成：把 prev_bgra + 当前帧脏矩形合并到 current_bgra
525                    //   编码器看到的输入 = "和上一帧几乎一样"，H.264 出极小 delta frame
526                    let composited_bgra = session.composite(&bgra_texture, &dirty_rects)?;
527
528                    // [D1 Phase 1] 周期打印 composit 耗时（debug / release-log 模式）
529                    #[cfg(any(debug_assertions, feature = "release-log"))]
530                    if frame_count % 30 == 0 {
531                        println!(
532                            "⏱️ [compositor] 上次 composit 耗时: {} us",
533                            session.last_composite_us()
534                        );
535                    }
536
537                    // 颜色转换：BGRA → NV12（喂的是合成后的 current_bgra，不是原始 frame）
538                    session.encoder().convert_bgra_to_nv12(&composited_bgra)?;
539                    // composited_bgra 在这里 drop（Release）
540
541                    // 把 NV12 数据打包成 IMFSample，准备喂给 MFT
542                    let imfsample = session.encoder().build_sample(hns_timestamp, frame_duration_hns)?;
543
544                    // 喂输入：硬件 MFT 是异步的，会在内部等 NeedInput 事件
545                    // [修复 §3.3] 改 by value：process_input 内部把软编的 sample 还回池里
546                    session.encoder().process_input(imfsample)?;
547
548                    // 取输出：如果 MFT 还没攒出一帧，会返回 Err(NEED_MORE_INPUT)
549                    match session.encoder().process_output() {
550                        Ok(out_sample_opt) => {
551                            // [release-log] 每 30 帧输出一次进度
552                            #[cfg(any(debug_assertions, feature = "release-log"))]
553                            {
554                                output_count += 1;
555                                if output_count % 30 == 0 {
556                                    let enc = session.encoder_ref();
557                                    println!("📊 已编码输出 {} 帧 ({}x{}@{}fps)",
558                                        output_count, enc.width, enc.height, fps_);
559                                }
560                            }
561                            if let Some(out_sample) = out_sample_opt {
562                                // [P1-12 修复] MFT 自提供的 sample 通常本来就是单 contiguous buffer。
563                                //   ConvertToContiguousBuffer 在 buffer count == 1 时仍会做一次无效的合并检查
564                                //   + 临时分配，每帧 ~5-50μs CPU stall。130 fps 下累计 ~6.5ms/s。
565                                //   优化：先 GetBufferCount；==1 直接 GetBufferByIndex(0) 跳过合并。
566                                let buffer_count = out_sample.GetBufferCount().unwrap_or(1);
567                                let contiguous_result = if buffer_count == 1 {
568                                    // 快路径：直接拿首个 buffer（不调用 ConvertToContiguousBuffer）
569                                    out_sample.GetBufferByIndex(0)
570                                } else {
571                                    // 慢路径：多 buffer 合并（罕见，例如 MFT 跨 stride）
572                                    out_sample.ConvertToContiguousBuffer()
573                                };
574                                if let Ok(buffer) = contiguous_result {
575                                    let mut data_ptr: *mut u8 = std::ptr::null_mut();
576                                    let mut max_len: u32 = 0;
577                                    let mut cur_len: u32 = 0;
578                                    // Lock 拿到字节指针 + 当前长度
579                                    if buffer.Lock(&mut data_ptr, Some(&mut max_len as *mut _), Some(&mut cur_len as *mut _)).is_ok() {
580                                        // [V6-R1] MFT 给的字节是 AVCC 格式（4 字节大端长度前缀 NALU），不是 Annex-B。
581                                        //   直接拿字节走 ffmpeg / 浏览器 MSE / libavcodec 全解不开。
582                                        //   必须转 Annex-B（00 00 00 01 起始码） + IDR 帧前注入 SPS+PPS。
583                                        let avcc_bytes = std::slice::from_raw_parts(data_ptr, cur_len as usize);
584                                        let sps_pps = session.encoder().sps_pps_annex_b();
585                                        let (annex_b, is_keyframe) =
586                                            avcc_to_annex_b(avcc_bytes, sps_pps);
587
588                                        // [V6-R2] wire format：构造 WireFrame
589                                        //   - frame_index 自增（用于客户端检测丢帧 / 重排序）
590                                        //   - timestamp_hns 来自 input sample（MF HNS 单位）
591                                        //   - is_keyframe 来自 NALU type == 5 的检测
592                                        //   - payload 已经是 Annex-B 字节流
593                                        let payload_len = annex_b.len() as u32;
594                                        let wire = WireFrame {
595                                            frame_index: wire_frame_index,
596                                            timestamp_hns: hns_timestamp,
597                                            is_keyframe,
598                                            payload_len,
599                                            payload: annex_b,
600                                        };
601
602                                        // [V6-R3 占位] 把 WireFrame 推到 session.frame_buffer 等待网络层接走。
603                                        //   当前 sender thread 还没接（V5-B-3.3 D4 协议层未动工），
604                                        //   所以 wire 现在只进 buffer，不真的发到 socket。
605                                        //   网络层接上后用 mpsc::channel 替换 push_back。
606                                        session.push_frame(wire);
607                                        wire_frame_index += 1;
608
609                                        // [release-log] 首帧 + 每 30 帧打一次落地摘要
610                                        #[cfg(any(debug_assertions, feature = "release-log"))]
611                                        {
612                                            if frame_count == 1 || frame_count % 30 == 0 {
613                                                println!(
614                                                    "📦 [V6-R2] frame #{}: is_keyframe={}, payload {} 字节 (Annex-B), 累计 {} 帧在 buffer",
615                                                    frame_count, is_keyframe, payload_len,
616                                                    session.frame_buffer_len()
617                                                );
618                                            }
619                                        }
620
621                                        let _ = buffer.Unlock();
622                                    }
623                                }
624                                // 用完还回池里
625                                session.encoder().recycle_output_sample(Some(out_sample));
626                            }
627                            // 回收用过的 NV12 纹理
628                            session.encoder().recycle_texture();
629                        }
630                        Err(err) if err.code() == MF_E_TRANSFORM_NEED_MORE_INPUT => {
631                            // [修复 §2.2] NVENC 类硬件编码器是 1:1 的：
632                            //   收到 NEED_MORE_INPUT 表示 MFT 没消费任何输入，
633                            //   刚才 push 进 pending_textures 的那一帧其实 MFT 没吃，
634                            //   应该立刻还回池子。否则 pending_textures 单调增长、
635                            //   nv12_texture_pool 单调缩水，跑久了池子空了崩。
636                            //   软编走不到这条分支（软编同步消费，不会报 NEED_MORE_INPUT）。
637                            session.encoder().recycle_texture();
638                            continue;
639                        }
640                        Err(err) if err.code() == MF_E_TRANSFORM_STREAM_CHANGE => {
641                            // MFT 要求重新协商输出格式（屏幕分辨率变化等）。
642                            // 走和 proactive 分支一样的流程：拿当前帧的尺寸调 renegotiate。
643                            // 这次循环的帧已经走过 convert_bgra_to_nv12，旧尺寸的 NV12
644                            // 纹理会在 renegotiate 里被 drain 掉重建，下一帧直接用新尺寸。
645                            // [H6/H9 修复] 显式撤销 pending_textures（renegotiate 内部也会 clear，但显式更明确）：
646                            //   当前帧的 (tex, buf) 在 MFT 已经不引用了，立刻 pop 还回池子
647                            session.encoder().recycle_texture();
648                            let enc = session.encoder_ref();
649                            if new_w != enc.width || new_h != enc.height {
650                                // [H2 修复] 加 cfg gate
651                                #[cfg(any(debug_assertions, feature = "release-log"))]
652                                println!(
653                                    "🔄 MFT 要求重协商 ({}x{} -> {}x{}), 重新协商编码器...",
654                                    enc.width, enc.height, new_w, new_h
655                                );
656                                session.encoder().renegotiate(new_w, new_h)?;
657                                // [H6 修复] 强制 IDR：和 proactive 分支对齐
658                                session.encoder().request_keyframe();
659                            } else {
660                                // 尺寸没变但 MFT 还是要重协商 —— 罕见，先 log 观察
661                                // [H2 修复] 加 cfg gate
662                                #[cfg(any(debug_assertions, feature = "release-log"))]
663                                println!(
664                                    "⚠️ MFT 要求重协商但尺寸未变 ({}x{}), skipping",
665                                    enc.width, enc.height
666                                );
667                            }
668                            continue;
669                        }
670                        Err(err) => {
671                            // [C2] 未知 MFT 错误：分类后看是不是可恢复
672                            match SessionError::classify(&err) {
673                                SessionError::Recoverable(reason) => {
674                                    // [H2 修复] 加 cfg gate
675                                    #[cfg(any(debug_assertions, feature = "release-log"))]
676                                    println!("⚠️ 编码器错误（可恢复）: {}, 尝试 recover...", reason);
677                                    match recover_state.record_failure() {
678                                        RecoverAction::Backoff(ms) => {
679                                            std::thread::sleep(std::time::Duration::from_millis(ms));
680                                        }
681                                        RecoverAction::GiveUp => {
682                                            return Err(format!("30 秒内失败 {} 次, 放弃", recover_state.consecutive_failures()).into());
683                                        }
684                                    }
685                                    session.recover()?;
686                                    recover_state.record_success();
687                                    // [H2 修复] 加 cfg gate
688                                    #[cfg(any(debug_assertions, feature = "release-log"))]
689                                    println!("✅ Session 恢复成功 (累计 {} 次)",
690                                        recover_state.total_recoveries());
691                                }
692                                SessionError::Transient(_) => {
693                                    // 业务流错误，不该到这里（已经在前面 match MF_E_TRANSFORM_NEED_MORE_INPUT 处理）
694                                    // 真到这里说明有 bug，转 fatal
695                                    return Err(err.into());
696                                }
697                                SessionError::Fatal(reason) => {
698                                    return Err(format!("编码器致命错误: {} (HRESULT 0x{:08X})",
699                                        reason, err.code().0 as u32).into());
700                                }
701                            }
702                        }
703                    }
704                }
705                Err(DupError::Timeout) => {
706                    // 屏幕没变化，正常情况，继续等
707                    // [release-log] 累计超时统计。第一次 + 每 30 次报一次
708                    #[cfg(any(debug_assertions, feature = "release-log"))]
709                    {
710                        timeout_count += 1;
711                        if timeout_count == 1 || timeout_count % 30 == 0 {
712                            println!(
713                                "⏳ 等待帧超时 ({}/{} 次, 屏幕可能无变化 — 这是正常状态)",
714                                timeout_count, frame_count + timeout_count
715                            );
716                        }
717                    }
718                    continue;
719                }
720                Err(DupError::AccessLost) => {
721                    // [C2] 桌面切换/锁屏/全屏独占程序抢走了 DXGI 句柄
722                    // 走完整的 recover 路径（D3D device / DXGI dup / Encoder 全部重建），
723                    // 比原来的 dup.recreate() 更彻底——能 cover 笔记本睡眠唤醒等
724                    // 同时让 D3D device 也失效的场景。
725                    // [M5 修复] DupError::AccessLost 不带 HRESULT（windows-capture crate 是 typed enum
726                    //   把 DXGI_ERROR_ACCESS_LOST 收敛成一个变体），无法直接走 classify。
727                    //   但驱动可能扩展行为：新驱动可能把 DEVICE_REMOVED 包成 AccessLost，
728                    //   我们当前统一走"recover 3 次后 fallback"路径不会因此出问题，
729                    //   只是日志里少一行 "SessionError::classify" 输出。补一行提示：
730                    //   控制流不变（不影响 recover_state），仅作 diagnostic。
731                    #[cfg(any(debug_assertions, feature = "release-log"))]
732                    {
733                        // DupError::AccessLost 等价于 DXGI_ERROR_ACCESS_LOST，
734                        //   SessionError::classify 会返回 Recoverable("DXGI_ACCESS_LOST")。
735                        //   这里只是声明语义，未来真出现新 HRESULT 包成 AccessLost
736                        //   会在这里的日志里看到 "但 dup 已 wrap 成 AccessLost"
737                        let classified_reason = "DXGI_ACCESS_LOST (DupError enum 固定语义)";
738                        println!(
739                            "⚠️ DXGI AccessLost (classify: Recoverable / {}), 尝试 recover...",
740                            classified_reason
741                        );
742                    }
743                    dxgi_consecutive_failures += 1;
744
745                    // [L1 + 用户新要求] DXGI 持续失败 → 自动回退
746                    //   连续 N 次失败（不限 AccessLost 也包括下面的 DXGI 错误），
747                    //   如果锁屏 watcher 存在，模拟锁屏暂停：release session → 等解锁事件 → recreate。
748                    //   这是用户要求的"DXGI 失败时回退到锁屏方案"的实现。
749                    //   没有 watcher 时走原来的 C2 路径（recover_state 节流）。
750                    if dxgi_consecutive_failures >= DXGI_FALLBACK_THRESHOLD
751                        && lockscreen.is_some()
752                    {
753                        // [H2 修复] 加 cfg gate
754                        #[cfg(any(debug_assertions, feature = "release-log"))]
755                        println!(
756                            "🔁 [L1] DXGI 连续失败 {} 次 >= 阈值 {}, 切换到 fallback: release + 等解锁事件",
757                            dxgi_consecutive_failures, DXGI_FALLBACK_THRESHOLD
758                        );
759                        session.release();
760                        // [H5 修复] 不再写 session_released 本地变量。session.is_released() 是 single source of truth。
761                        //   下次循环顶部的 H5 检查会决定是进 paused 分支还是立即 recreate。
762                        // 不进入 wait_until_active（让外层 paused 分支统一处理）
763                        // 通过持续循环让 is_paused() 检查进入 paused 分支
764                        // 这里用 sleep 短暂等待，避免下一轮立刻再失败
765                        std::thread::sleep(std::time::Duration::from_millis(500));
766                        // 重置 recover_state（已经从 fallback 路径处理）
767                        recover_state = RecoverState::new();
768                        continue;
769                    }
770
771                    match recover_state.record_failure() {
772                        RecoverAction::Backoff(ms) => {
773                            std::thread::sleep(std::time::Duration::from_millis(ms));
774                        }
775                        RecoverAction::GiveUp => {
776                            return Err(format!("30 秒内失败 {} 次, 放弃", recover_state.consecutive_failures()).into());
777                        }
778                    }
779                    match session.recover() {
780                        Ok(()) => {
781                            recover_state.record_success();
782                            dxgi_consecutive_failures = 0; // recover 成功 → 重置失败计数
783                            // [H2 修复] 加 cfg gate
784                            #[cfg(any(debug_assertions, feature = "release-log"))]
785                            println!("✅ Session 恢复成功 (累计 {} 次)",
786                                recover_state.total_recoveries());
787                        }
788                        Err(e) => {
789                            return Err(format!("Session recover 失败: {}", e).into());
790                        }
791                    }
792                }
793                Err(e) => {
794                    // [C2] 其他 windows-capture 错误：分类后看是否可恢复
795                    // [1.8 修复] 走 SessionError::classify 分类（之前直接当 Recoverable 太乐观）：
796                    //   - 某些 DXGI 错误码是 Fatal（参数错、驱动不支持），硬 recover 浪费 1 秒 +
797                    //   - DupError 通常 wrap 了一个 windows::core::Error，downcast 出来给 classify
798                    // [H2 修复] 加 cfg gate
799                    #[cfg(any(debug_assertions, feature = "release-log"))]
800                    println!("❌ DXGI 错误: {}, 分类: AccessLost 之外的未知", e);
801                    dxgi_consecutive_failures += 1;
802                    // [H8 修复] dup() 入口已经 match 掉 DupError::AccessLost，这里走的是其他 DupError 分支，
803                    //   全走 classify 决策；和 entry 处保持一致
804
805                    // [L1] 同上：连续失败 → fallback 到锁屏等待
806                    if dxgi_consecutive_failures >= DXGI_FALLBACK_THRESHOLD
807                        && lockscreen.is_some()
808                    {
809                        // [H2 修复] 加 cfg gate
810                        #[cfg(any(debug_assertions, feature = "release-log"))]
811                        println!(
812                            "🔁 [L1] DXGI 连续失败 {} 次 >= 阈值 {}, 切换到 fallback",
813                            dxgi_consecutive_failures, DXGI_FALLBACK_THRESHOLD
814                        );
815                        session.release();
816                        // [H5 修复] 不再写 session_released 本地变量（同上注释）
817                        std::thread::sleep(std::time::Duration::from_millis(500));
818                        recover_state = RecoverState::new();
819                        continue;
820                    }
821
822                    // 分类：尝试 downcast 到 windows::core::Error，
823                    //   成功就调 SessionError::classify，失败兜底 Fatal（和 classify() 的未知错误兜底对齐）
824                    let classification = {
825                        let mut src: Option<&(dyn std::error::Error + 'static)> = Some(&e);
826                        let mut found_win_err: Option<&windows::core::Error> = None;
827                        while let Some(s) = src {
828                            if let Some(we) = s.downcast_ref::<windows::core::Error>() {
829                                found_win_err = Some(we);
830                                break;
831                            }
832                            src = s.source();
833                        }
834                        match found_win_err {
835                            Some(we) => SessionError::classify(we),
836                            // [H8 修复] 拿不到 HRESULT 时降级为 Fatal，**和 SessionError::classify 未知错误兜底对齐**：
837                            //   1. 之前这里返回 Recoverable 是乐观假定 —— 实际遇到"未知错误"里其实是 Fatal 的（参数错等）
838                            //      会陷入无限 recover 循环
839                            //   2. classify() 自己的未知错误兜底就是 Fatal
840                            //   3. 这里 downcast 失败 = 拿不到 HRESULT = 同样属于"未知错误"，语义和 classify() 兜底一致
841                            //   4. 但 DupError 还有原始消息可以参考，所以拼上 dup 的 msg 让日志有信息
842                            None => SessionError::Fatal(
843                                format!("DupError 无法 downcast 到 windows::core::Error (DupError msg: {})", e)
844                            ),
845                        }
846                    };
847                    match classification {
848                        SessionError::Fatal(reason) => {
849                            // [1.8 修复] 致命错误直接退出，不再无脑 recover
850                            return Err(format!(
851                                "DXGI 致命错误: {} (DupError 链: {})",
852                                reason, e
853                            ).into());
854                        }
855                        SessionError::Recoverable(_) | SessionError::Transient(_) => {
856                            // 可恢复 / 业务流错误：走原来的节流 + recover 流程
857                            match recover_state.record_failure() {
858                                RecoverAction::Backoff(ms) => {
859                                    std::thread::sleep(std::time::Duration::from_millis(ms));
860                                }
861                                RecoverAction::GiveUp => {
862                                    return Err(format!("30 秒内失败 {} 次, 放弃", recover_state.consecutive_failures()).into());
863                                }
864                            }
865                            if let Err(rerr) = session.recover() {
866                                return Err(format!("DXGI 错误且 recover 失败: orig={}, recover={}", e, rerr).into());
867                            }
868                            recover_state.record_success();
869                            dxgi_consecutive_failures = 0;
870                            // [H2 修复] 加 cfg gate
871                            #[cfg(any(debug_assertions, feature = "release-log"))]
872                            println!("✅ Session 恢复成功 (累计 {} 次)", recover_state.total_recoveries());
873                        }
874                    }
875                }
876            }
877        }
878    }
879
880    Ok(())
881}
