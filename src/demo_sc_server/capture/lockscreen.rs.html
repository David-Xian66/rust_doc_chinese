1//! [L1] 锁屏监听器：注册 WTSAPI 会话通知 → 主循环检测到锁屏 → release session → 阻塞等解锁
2//!
3//! ## 目的
4//!
5//! Win+L / 屏幕超时 / 用户注销时，DXGI Desktop Duplication 会持续报 `DXGI_ERROR_ACCESS_LOST`，
6//! C2 recover 路径会浪费 CPU 且 30 秒后 GiveUp 服务端退出。
7//!
8//! 本模块提供：
9//! 1. **WTSAPI 监听**：后台线程跑 Windows 消息循环，捕获 `WM_WTSSESSION_CHANGE` 事件
10//! 2. **状态机聚合**：把 WTS_LOCK / WTS_UNLOCK / WTS_LOGOFF / WTS_LOGON 转换为 `SessionState`
11//! 3. **Condvar 阻塞**：主循环在 paused 状态下真阻塞（不 sleep 轮询），解锁时 < 50ms 唤醒
12//! 4. **事件接口**（预留）：把事件流暴露给心跳占位等将来特性
13//!
14//! ## 数据流
15//!
16//! ```text
17//! Windows 内核 (会话状态变化)
18//!   └─ WM_WTSSESSION_CHANGE 消息
19//!       └─ 监听线程的消息循环
20//!           └─ 转换为 SessionEvent
21//!               └─ 状态机更新 SessionState (atomic)
22//!                   ├─ 主循环 is_paused() 检查 → 进入 release/wait 分支
23//!                   └─ 解锁时 notify_all Condvar → 主循环被唤醒
24//! ```
25//!
26//! ## 简化说明（vs v3 §11.3 的完整设计）
27//!
28//! - 实现了 WTSAPI 事件源（消息循环 + WM_WTSSESSION_CHANGE）
29//! - **未实现** PowerAPI（Modern Standby 监听）。原因：PowerRegisterSuspendResumeNotification 的
30//!   callback 模式要通过 raw pointer 跨线程传递 Sender，复杂度高。Modern Standby 用户量较少，
31//!   真遇到时 Win+L 也会触发 WTS_LOCK，可以兜底。
32//! - **未实现** 心跳占位。等 D4 协议层落地后做。
33//! - **未实现** UAC / 用户切换的可配置开关。v1 默认不启用 user switch pause。
34

51/// 会话状态机。
52#[derive(Debug, Clone, Copy, PartialEq, Eq)]
53pub enum SessionState {
54    /// 正常运行
55    Active,
56    /// 锁屏中（Win+L / 屏幕超时）
57    Locked,
58    /// 用户注销 / 切换中（LOGOFF → LOGON 间隙）
59    UserSwitching,
60}
61
62impl SessionState {
63    fn as_u8(self) -> u8 {
64        match self {
65            SessionState::Active => 0,
66            SessionState::Locked => 1,
67            SessionState::UserSwitching => 2,
68        }
69    }
70    fn from_u8(v: u8) -> Self {
71        match v {
72            1 => SessionState::Locked,
73            2 => SessionState::UserSwitching,
74            _ => SessionState::Active,
75        }
76    }
77    /// 当前状态是否需要暂停主循环
78    pub fn is_paused(&self) -> bool {
79        !matches!(self, SessionState::Active)
80    }
81}

84#[derive(Debug, Clone, Copy, PartialEq, Eq)]
85pub enum SessionEvent {
86    Locked,
87    Unlocked,
88    UserLoggedOff,
89    UserLoggedOn,
90    /// [预留] 远程桌面连接（WTS_CONSOLE_CONNECT / WTS_REMOTE_CONNECT）
91    #[allow(dead_code)]
92    RemoteConnected,
93    /// [预留] 远程桌面断开（WTS_CONSOLE_DISCONNECT / WTS_REMOTE_DISCONNECT）
94    #[allow(dead_code)]

96}

98/// 锁屏监听器。
99///
100/// 启动后会在后台跑一个线程，里面：
101/// 1. 注册 Windows 窗口类 + 创建 message-only 窗口
102/// 2. 注册 WTSAPI 会话通知
103/// 3. 跑消息循环 + 处理 WM_WTSSESSION_CHANGE
104/// 4. 转换 WTS 事件为 SessionEvent，更新共享状态 + Condvar 唤醒
105///
106/// 主循环通过 `is_paused()` 检查状态，通过 `wait_until_active()` 阻塞到解锁。
107pub struct LockscreenWatcher {
108    state: Arc<AtomicU8>,
109    /// [1.10 修复] 后台消息循环是否还活着。GetMessageA 失败时设为 false，
110    ///   `is_paused()` 检测到后会返回 false（视为 watcher 已死，不再 paused）。
111    healthy: Arc<AtomicBool>,
112    resume_signal: Arc<(Mutex<bool>, Condvar)>,
113    /// 后台线程 handle（drop 时自动 join）
114    _wts_thread: JoinHandle<()>,
115    /// message-only 窗口句柄。Drop 时调 DestroyWindow 让消息循环收到 WM_DESTROY
116    /// → wnd_proc 调 PostQuitMessage → GetMessageA 返回 0 → 后台线程退出。
117    _hwnd: SendHwnd,
118}
119
120/// HWND 是 raw pointer (*mut c_void)，默认不是 Send/Sync。
121/// 但 Windows 文档明确允许窗口句柄跨线程传递（不同线程用同一 HWND 调 API 是合法的，
122/// 只是不能两个线程同时操作同一窗口）。
123/// [P1-9 修复] 只声明 Send，不声明 Sync：
124///   - 跨线程调用同一个 HWND 的 API 是合法的
125///   - 但两个线程同时操作同一个 HWND 是非法的（线程亲和性约束）
126///   - 旧 unsafe impl Sync 给上层"安全"假象，未来若跨线程操作会撞 bug
127///   - 当前所有调用都在主线程，Send 足够跨线程 Move
128struct SendHwnd(HWND);
129unsafe impl Send for SendHwnd {}
130
131unsafe impl Send for LockscreenWatcher {}
132
133impl LockscreenWatcher {
134    /// 启动监听器。失败原因通常是窗口类冲突（重复启动 watcher）。
135    pub fn new(enable_user_switch: bool) -> Result<Self, Box<dyn std::error::Error>> {
136        let state = Arc::new(AtomicU8::new(SessionState::Active.as_u8()));
137        let healthy = Arc::new(AtomicBool::new(true));
138        let resume_signal = Arc::new((Mutex::new(false), Condvar::new()));
139        // HWND 不是 Send，需要用 SendHwnd 包装
140        let hwnd_holder: Arc<Mutex<Option<SendHwnd>>> = Arc::new(Mutex::new(None));
141        let err_holder: Arc<Mutex<Option<String>>> = Arc::new(Mutex::new(None));
142
143        let state_for_thread = state.clone();
144        let healthy_for_thread = healthy.clone();
145        let resume_for_thread = resume_signal.clone();
146        let hwnd_holder_for_thread = hwnd_holder.clone();
147        let err_holder_for_thread = err_holder.clone();
148
149        // 后台线程：注册窗口类 + 创建 message-only 窗口 + 注册 WTS + 消息循环
150        let wts_thread = std::thread::spawn(move || {
151            wts_thread_body(
152                &state_for_thread,
153                &healthy_for_thread,
154                &resume_for_thread,
155                &hwnd_holder_for_thread,
156                &err_holder_for_thread,
157                enable_user_switch,
158            );
159        });
160
161        // 等窗口创建好（最多 500ms，每 10ms 检查一次）
162        let mut attempts = 0;
163        let hwnd = loop {
164            std::thread::sleep(std::time::Duration::from_millis(10));
165            // 检查错误
166            if let Some(err) = err_holder.lock().unwrap_or_else(|p| p.into_inner()).take() {
167                return Err(err.into());
168            }
169            if let Some(h) = hwnd_holder.lock().unwrap_or_else(|p| p.into_inner()).take() {
170                break h;
171            }
172            attempts += 1;
173            if attempts > 50 {
174                return Err("LockscreenWatcher 窗口创建超时".into());
175            }
176        };
177
178        Ok(Self {
179            state,
180            healthy,
181            resume_signal,
182            _wts_thread: wts_thread,
183            _hwnd: hwnd,
184        })
185    }
186
187    /// 当前是否处于暂停状态（锁屏 / 用户切换）。
188    ///
189    /// [1.10 修复] watcher 失效时（GetMessageA 失败 / 线程 panic 等）返回 false，
190    ///   让主循环继续运行（即使 watcher 死了，捕获功能还能跑）。
191    /// [P1-7 修复] state 用 Acquire load（配对 Release store，在 handle_wts_event 里改），
192    ///   防止 Relaxed 在弱内存模型（ARM / 省电 C-state）下读不到最新状态。
193    ///   500ms fallback sleep 期间系统可能切状态，原 Relaxed 配 SeqCst 不一致 → 主循环
194    ///   读到陈旧值。Acquire/Release 一致配对保证 happens-before。
195    pub fn is_paused(&self) -> bool {
196        if !self.healthy.load(Ordering::Acquire) {
197            return false;
198        }
199        SessionState::from_u8(self.state.load(Ordering::Acquire)).is_paused()
200    }
201
202    /// 当前状态（给 release-log 日志用）
203    pub fn current_state(&self) -> SessionState {
204        SessionState::from_u8(self.state.load(Ordering::Acquire))
205    }
206
207    /// 真阻塞：等到状态变 Active 才返回。如果 shutdown 被 set 或 watcher 失效，立即返回。
208    ///
209    /// [1.1 修复] 用 `wait_timeout` 周期唤醒（500ms），即使 cvar 没人 notify 也能响应 shutdown flag。
210    ///   原实现 `cvar.wait` 永久阻塞（Ctrl+C 锁屏期间主线程永远醒不过来）。
211    pub fn wait_until_active(&self, shutdown: &Arc<AtomicBool>) -> Result<(), Box<dyn std::error::Error>> {
212        // 已经在 Active 或 watcher 失效 → 直接返回
213        if !self.is_paused() {
214            return Ok(());
215        }
216        let (lock, cvar) = &*self.resume_signal;
217        let mut woken = lock.lock().unwrap_or_else(|p| p.into_inner());
218        let wait_period = std::time::Duration::from_millis(500);
219        // 双检：condvar 被 spurious wakeup / timeout 唤醒时也要重新检查条件
220        // wait_timeout 即使没人 notify 也会超时返回 → 周期性检查 shutdown flag
221        while !*woken && !shutdown.load(Ordering::Relaxed) && self.healthy.load(Ordering::Acquire) {
222            let (guard, _) = cvar.wait_timeout(woken, wait_period)
223                .unwrap_or_else(|p| p.into_inner());
224            woken = guard;
225        }
226        if shutdown.load(Ordering::Relaxed) {
227            return Err("shutdown during pause".into());
228        }
229        if !self.healthy.load(Ordering::Acquire) {
230            // watcher 失效（GetMessageA 失败等）→ 返回 Ok 让主循环继续
231            // 不算错误，因为捕获功能还能跑
232            return Ok(());
233        }
234        *woken = false; // 重置标志，下次 wait 还能继续阻塞
235        Ok(())
236    }
237}

239/// [P1-8] wts_thread 主体（拆出来方便 catch_unwind 包一层）
240fn wts_thread_body(
241    state_for_thread: &Arc<AtomicU8>,
242    healthy_for_thread: &Arc<AtomicBool>,
243    resume_for_thread: &Arc<(Mutex<bool>, Condvar)>,
244    hwnd_holder_for_thread: &Arc<Mutex<Option<SendHwnd>>>,
245    err_holder_for_thread: &Arc<Mutex<Option<String>>>,
246    enable_user_switch: bool,
247) {
248    // 窗口类名（C 字符串以 \0 结尾）
249    let class_name = b"DemoScWtsSessionWatcher\0";
250
251    // 拿当前 module handle
252    let h_instance = unsafe { GetModuleHandleA(None) };
253
254    let h_instance = match h_instance {
255        Ok(h) => h,
256        Err(e) => {
257            *err_holder_for_thread.lock().unwrap_or_else(|p| p.into_inner()) =
258                Some(format!("GetModuleHandleA 失败: {}", e));
259            return;
260        }
261    };
262
263    // 注册窗口类
264    let wnd_class = WNDCLASSEXA {
265        cbSize: std::mem::size_of::<WNDCLASSEXA>() as u32,
266        style: WNDCLASS_STYLES(0),
267        lpfnWndProc: Some(wnd_proc),
268        cbClsExtra: 0,
269        cbWndExtra: 0,
270        hInstance: h_instance.into(),
271        hIcon: Default::default(),
272        hCursor: Default::default(),
273        hbrBackground: Default::default(),
274        lpszMenuName: PCSTR::null(),
275        lpszClassName: PCSTR(class_name.as_ptr()),
276        hIconSm: Default::default(),
277    };
278    let atom = unsafe { RegisterClassExA(&wnd_class) };
279    if atom == 0 {
280        // 1410 = ERROR_CLASS_ALREADY_EXISTS（重复启动）
281        *err_holder_for_thread.lock().unwrap_or_else(|p| p.into_inner()) =
282            Some("RegisterClassExA 失败（窗口类可能已注册）".into());
283        return;
284    }
285
286    // 创建 message-only 窗口
287    let hwnd_result = unsafe {
288        CreateWindowExA(
289            Default::default(),
290            PCSTR(class_name.as_ptr()),
291            PCSTR(class_name.as_ptr()),
292            Default::default(),
293            0,
294            0,
295            0,
296            0,
297            Some(HWND_MESSAGE),
298            None,
299            Some(h_instance.into()),
300            None,
301        )
302    };
303    let hwnd = match hwnd_result {
304        Ok(h) => h,
305        Err(e) => {
306            *err_holder_for_thread.lock().unwrap_or_else(|p| p.into_inner()) =
307                Some(format!("CreateWindowExA 失败: {}", e));
308            return;
309        }
310    };
311
312    // 注册 WTS 会话通知
313    if let Err(e) = unsafe { WTSRegisterSessionNotification(hwnd, NOTIFY_FOR_THIS_SESSION) } {
314        *err_holder_for_thread.lock().unwrap_or_else(|p| p.into_inner()) =
315            Some(format!("WTSRegisterSessionNotification 失败: {}", e));
316        return;
317    }
318
319    // 把 hwnd 暴露给主线程（用于 Drop 时清理）
320    *hwnd_holder_for_thread.lock().unwrap_or_else(|p| p.into_inner()) = Some(SendHwnd(hwnd));
321
322    // 消息循环
323    // [1.10 修复] 用 match ret.0 区分 0 (WM_QUIT) / -1 (错误) / >0 (正常消息)
324    //   GetMessageA 失败时：设 healthy=false + notify 主线程 + break
325    //   否则主线程一直以为 watcher 活着，session_released=true 状态下走编码会 panic
326    let mut msg = MSG::default();
327    loop {
328        let ret = unsafe { GetMessageA(&mut msg, Some(hwnd), 0, 0) };
329        match ret.0 {
330            0 => break, // WM_QUIT（Drop 时发的 PostQuitMessage）
331            -1 => {
332                #[cfg(any(debug_assertions, feature = "release-log"))]
333                eprintln!("⚠️ [L1] GetMessageA 失败 (ret=-1), 标记 watcher 失效并退出消息循环");
334                // 标记 watcher 失效
335                // [P1-7] Release store 配对 is_paused / wait_until_active 的 Acquire load
336                healthy_for_thread.store(false, Ordering::Release);
337                // 唤醒可能在 wait_until_active 里阻塞的主线程
338                let (lock, cvar) = resume_for_thread.as_ref();
339                if let Ok(mut g) = lock.lock() {
340                    *g = true;
341                    cvar.notify_all();
342                }
343                break;
344            }
345            _ => {} // 正常消息，继续往下处理
346        }
347        if msg.message == WM_WTSSESSION_CHANGE {
348            handle_wts_event(msg.wParam.0 as u32, &state_for_thread, &resume_for_thread, enable_user_switch);
349        }
350        unsafe {
351            let _ = TranslateMessage(&msg);
352            DispatchMessageA(&msg);
353        }
354    }
355}

357impl Drop for LockscreenWatcher {
358    fn drop(&mut self) {
359        // [H7 修复] 4 个 bug：
360        //   1. 原 `PostQuitMessage(0)` 投到**主线程**消息队列（当前线程），
361        //      后台线程的 GetMessageA 收不到，**永远不生效**。
362        //   2. 没调 DestroyWindow，后台线程靠 join 阻塞退出，主线程被 join 卡住。
363        //   3. 主线程此刻若在 `wait_until_active` 阻塞（用户锁屏了），
364        //      shutdown flag 没人 set，cvar 没人 notify → 主线程永久阻塞。
365        //   4. Ctrl+C 第一次只 set shutdown，但 join 阻塞 → 第二次 Ctrl+C 才 exit(2) 强退。
366        //
367        // 修复：DestroyWindow 触发 WM_DESTROY → wnd_proc 调 PostQuitMessage →
368        //       GetMessageA 返回 0 → 后台线程 break → join 立即返回。
369        //       额外：healthy=false + notify_all 唤醒可能在 wait_until_active 的主线程。
370        unsafe {
371            let _ = WTSUnRegisterSessionNotification(self._hwnd.0);
372            // 1. 销毁窗口 → 触发 WM_DESTROY → wnd_proc 调 PostQuitMessage → GetMessageA 返回 0
373            let _ = DestroyWindow(self._hwnd.0);
374        }
375        // 2. 标记 watcher 失效 + 唤醒可能在 wait_until_active 的主线程
376        //    （healthy 字段也是 Arc<AtomicBool>，但 Drop 时 self 还在 → 直接 store）
377        // [P1-7] Release store 配对 reader 的 Acquire load
378        self.healthy.store(false, Ordering::Release);
379        let (lock, cvar) = self.resume_signal.as_ref();
380        if let Ok(mut g) = lock.lock() {
381            *g = true;
382            cvar.notify_all();
383        }
384        // _wts_thread 在本函数返回后自动 Drop（join）
385        // 此时消息循环已收到 WM_DESTROY → break → 线程退出
386    }
387}

389/// 把 WM_WTSSESSION_CHANGE 的 wParam 值转换为 SessionEvent 并更新状态。
390///
391/// 注意：WTSAPI 的常量在 windows crate 里可能有变动，这里用魔法数字而不是常量名，
392/// 避免编译失败。原始值见 Microsoft Docs: WM_WTSSESSION_CHANGE 的 wParam 取值。
393fn handle_wts_event(
394    wparam: u32,
395    state: &Arc<AtomicU8>,
396    resume_signal: &Arc<(Mutex<bool>, Condvar)>,
397    enable_user_switch: bool,
398) {
399    // WTS API 常量值（参考 wtsapi32.h + Microsoft Docs）
400    //   WTS_SESSION_LOGON    = 1
401    //   WTS_SESSION_LOGOFF   = 2
402    //   WTS_REMOTE_CONNECT   = 3
403    //   WTS_REMOTE_DISCONNECT= 4
404    //   WTS_SESSION_LOGON    (重复了)
405    //   WTS_SESSION_LOGOFF   (重复了)
406    //   WTS_SESSION_LOCK     = 7
407    //   WTS_SESSION_UNLOCK   = 8
408    //
409    // 注意：实际值是 1/2/3/4/5/6/7/8（不要看错 MSDN 文档，wParam 是这些数字）
410    let event = match wparam {
411        7 => SessionEvent::Locked,
412        8 => SessionEvent::Unlocked,
413        5 => SessionEvent::UserLoggedOn, // 实际文档里 WTS_SESSION_LOGON = 5
414        6 => SessionEvent::UserLoggedOff, // WTS_SESSION_LOGOFF = 6
415        3 | 4 | 9 | 10 => {
416            // RemoteConnect/Disconnect 等：仅记录日志，不改状态
417            #[cfg(any(debug_assertions, feature = "release-log"))]
418            println!("📡 [session] 远程桌面会话切换 (wParam={})", wparam);
419            return;
420        }
421        _ => {
422            #[cfg(any(debug_assertions, feature = "release-log"))]
423            println!("⚠️ [session] 未知 WTS 事件 (wParam={})", wparam);
424            return;
425        }
426    };
427
428    let old_state = SessionState::from_u8(state.load(Ordering::Acquire));
429    let new_state = match (old_state, event) {
430        (SessionState::Active, SessionEvent::Locked) => SessionState::Locked,
431        (SessionState::Locked, SessionEvent::Unlocked) => SessionState::Active,
432        (SessionState::Active, SessionEvent::UserLoggedOff) if enable_user_switch => SessionState::UserSwitching,
433        (SessionState::UserSwitching, SessionEvent::UserLoggedOn) => SessionState::Active,
434        (SessionState::UserSwitching, SessionEvent::Unlocked) => {
435            // UserSwitching 期间收到 UNLOCK（罕见，但可能）：保持 UserSwitching
436            old_state
437        }
438        // 幂等性：保持原状态
439        (state, _) => state,
440    };
441
442    if new_state != old_state {
443        // [P1-7 修复] 用 Release store（配对 is_paused / current_state 的 Acquire load）。
444        //   比 SeqCst 弱但够用——只需要保证"store happens-before 后续 reader 的 Acquire load"，
445        //   不需要全局顺序。
446        state.store(new_state.as_u8(), Ordering::Release);
447        #[cfg(any(debug_assertions, feature = "release-log"))]
448        println!("🔀 [session] 状态切换: {:?} -> {:?} ({:?})", old_state, new_state, event);
449
450        // 切回 Active → 唤醒主循环
451        if new_state == SessionState::Active {
452            let (lock, cvar) = &**resume_signal;
453            let mut woken = lock.lock().unwrap_or_else(|p| p.into_inner());
454            *woken = true;
455            cvar.notify_all();
456        }
457    }
458}

460/// 窗口过程函数。
461///
462/// WM_DESTROY 时 PostQuitMessage 让消息循环退出。其他消息走默认处理。
463unsafe extern "system" fn wnd_proc(hwnd: HWND, msg: u32, wparam: WPARAM, lparam: LPARAM) -> LRESULT {
464    if msg == WM_DESTROY {
465        unsafe { PostQuitMessage(0) };
466        return LRESULT(0);
467    }
468    unsafe { DefWindowProcA(hwnd, msg, wparam, lparam) }
