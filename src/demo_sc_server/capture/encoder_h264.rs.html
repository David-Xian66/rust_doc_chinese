1//! H.264 视频编码器（基于 Windows Media Foundation）
2//!
3//! # 这个文件做什么
4//! 把 GPU 上的 BGRA 纹理（来自 DXGI 桌面复制）转成 H.264 码流。
5//! 优先用硬件编码器（NVENC / QuickSync / AMF 等），失败回退到微软自带的软编。
6//!
7//! # COM 概念速览（看这里能少踩 80% 的坑）
8//!
9//! Windows 的多媒体栈几乎全部用 COM（Component Object Model）写成。COM 对象的核心规则：
10//! - 每个接口都引用计数。`AddRef` +1，`Release` -1，归零时对象自我销毁。
11//! - Rust 这边 `windows` crate 用智能指针（如 `IMFTransform`、`IMFSample`）包装，
12//!   离开作用域时会自动调 `Release`，所以**正常情况下你不需要手动 Release**。
13//! - 所有跨线程/跨方法的 COM 对象都靠引用计数传递所有权。
14//!
15//! # Media Foundation 核心接口（这个文件会反复用到）
16//!
17//! | 接口                          | 作用                                                  |
18//! |-------------------------------|-------------------------------------------------------|
19//! | `IMFTransform` (MFT)          | 编/解码器本体，`ProcessInput` / `ProcessOutput` 是出入口 |
20//! | `IMFActivate`                 | MFT 的"工厂对象"，要先 `ActivateObject` 才能拿到 MFT    |
21//! | `IMFMediaType`                | 描述一种视频格式（分辨率、帧率、子类型 NV12/H264 等）  |
22//! | `IMFSample`                   | 一帧数据的容器，内部有 1 个或多个 `IMFMediaBuffer`       |
23//! | `IMFMediaBuffer`              | 一块连续的字节/显存                                     |
24//! | `IMFDXGIDeviceManager`        | 把 D3D11 设备"借"给 MFT，让它直接读写 GPU 显存          |
25//! | `IMFMediaEventGenerator`      | MFT 的事件来源，硬件 MFT 是异步的，要靠它推状态机        |
26//!
27//! # 数据流（一帧 H.264 输出的生命周期）
28//!
29//! ```text
30//! DXGI BGRA 纹理
31//!   └─ ID3D11Texture2D (bgra_texture)
32//!       └─ convert_bgra_to_nv12:   GPU VideoProcessor → NV12 纹理（零拷贝）
33//!       └─ build_sample:           把 NV12 纹理包成 IMFSample（硬件路径：包成 DXGI Surface Buffer）
34//!       └─ process_input:          IMFSample → MFT 输入
35//!       └─ process_output:         MFT 输出 → IMFSample（含一段 H.264 码流）
36//!       └─ 业务层读出 H.264 字节（这里只是 println，没真的发出去）
37//! ```

55/// [B3] H.264 编码 Profile。
56///
57/// 影响压缩率 / 兼容性 / 延迟：
58/// - Baseline: 无 B 帧，最广兼容（移动端 / 老硬件）
59/// - Main:     有 B 帧 + CABAC，中等兼容
60/// - High:     Main + 8x8 变换，最高压缩率，新硬件都支持
61///
62/// 注意：值对应 `eAVEncH264VProfile_*` 常量，直接传给 `MF_MT_MPEG2_PROFILE`。
63#[derive(Debug, Clone, Copy, PartialEq, Eq)]
64pub enum H264Profile {
65    Baseline,
66    Main,
67    High,
68}

75            H264Profile::Main => 77,       // eAVEncH264VProfile_Main

97    }

109/// H.264 Slice 切分模式（对应 `CODECAPI_AVEncSliceMode`）。
110///
111/// 数值与 mfapi.h `eAVEncSliceMode` 一致。0.62.2 没暴露这个枚举，要自己写常量。
112///
113/// 决定编码器内部把一帧切成多少个独立可解码片：
114/// - `WholeFrame`: 整帧 = 1 个 slice（最低延迟，最高压缩，单 NALU 可能很大）
115/// - `ArbitrarySlice`: 编码器按内部启发式切（NVENC 不一定支持）
116/// - `MbBased`: 按宏块（16x16）切，最细粒度
117///
118/// `WholeFrame` / `ArbitrarySlice` 当前 main.rs 默认配置未用到（用 `MbBased`），
119/// 但作为公开 API 保留供调用方切换，加 `#[allow(dead_code)]` 避免 unused 警告。
120#[allow(dead_code)]
121#[derive(Debug, Clone, Copy, PartialEq, Eq)]
122pub enum SliceMode {
123    /// 0: 整帧 = 1 个 slice
124    WholeFrame = 0,
125    /// 1: 编码器启发式切分
126    ArbitrarySlice = 1,
127    /// 2: 按宏块（16x16）切
128    MbBased = 2,
129}
130
131/// H.264 Slice 粒度单位（对应 `CODECAPI_AVEncSliceControlMode`）。
132///
133/// 数值与 mfapi.h `eAVEncSliceControlMode` 一致。决定 [`H264SliceConfig::count`]
134/// 字段按什么单位解读。
135///
136/// `Macroblocks` 当前未在 main.rs 使用（用 `MbRows`），作为 API 完整性保留。
137#[allow(dead_code)]
138#[derive(Debug, Clone, Copy, PartialEq, Eq)]
139pub enum SliceControlUnit {
140    /// 0: 按宏块数（每片多少 MB）
141    Macroblocks = 0,
142    /// 1: 按字节数（每片目标 NALU 字节数，常用于 MTU 限制）
143    Bytes = 1,
144    /// 2: 按 MB 行数（每片多少行 MB）
145    MbRows = 2,
146}

148/// H.264 Slice 配置（4 个 ICodecAPI 属性的聚合，方案见 `H264_SLICE_PARAMS.md`）。
149///
150/// 任何字段 `None` = 不动编码器默认。`Default` 全 None。
151///
152/// # 字段语义
153///
154/// - `mode` + `control_mode` + `count`: 主控制旋钮。比如 `mode=MbBased, control_mode=MbRows,
155///   count=16` 表示"按 MB 行切，每片 16 行"。
156/// - `max_bytes`: 限每片字节数。**会覆盖** `count` 的语义——`apply_slice_config` 在设完
157///   `count` 后再用 `Bytes` 单位 + `max_bytes` 再调一次 SetValue（强制按字节限大小）。
158///   单独使用时也是合法的（不依赖 `control_mode`）。
159/// - `max_count`: 每帧片数上限（独立于 `count`，对应 `AVEncSliceMaxNumber`）。
160///
161/// # 不支持的字段
162///
163/// 任何字段 SetValue 失败时**只 log 警告，不影响其他字段继续下发，也不影响编码器继续运行**。
164/// （和 `H264Profile` 行为对称。）
165#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
166pub struct H264SliceConfig {
167    pub mode: Option<SliceMode>,
168    pub control_mode: Option<SliceControlUnit>,
169    pub count: Option<u32>,
170    pub max_bytes: Option<u32>,
171    pub max_count: Option<u32>,
172}

188/// 构造一个 `VARIANT(VT_UI4, x)`。
189///
190/// `VARIANT_0` 是 union（`Anonymous: ManuallyDrop<VARIANT_0_0>` 或 `decVal: DECIMAL`），
191/// `VARIANT_0_0` 又是 `ManuallyDrop<VARIANT_0_0_0>`，需要：
192/// - `VARIANT_0` 的 union 字段访问 → `unsafe`
193/// - `ManuallyDrop` 字段赋值 → 解引用（`*`）显式说明不跑旧值 drop
194/// - `VARIANT_0_0_0` 的 union 字段访问 → `unsafe`
195///
196/// `vt = VT_UI4` 下所有 `ManuallyDrop` 字段都不会被读，零内存是合法的活跃变体。
197/// `VARIANT` 离开作用域无需 `VariantClear`（`vt != VT_DISPATCH/VT_UNKNOWN/VT_BSTR`）。
198fn variant_ui4(x: u32) -> VARIANT {
199    let mut v: VARIANT = unsafe { core::mem::zeroed() };
200    // `addr_of_mut!` 绕过 Rust 的自动 deref + borrow 规则，
201    // 拿到 `*mut ManuallyDrop<VARIANT_0_0>`。
202    // ManuallyDrop<T> 是 #[repr(transparent)]，指针 cast 到 `*mut T` 是 OK 的。
203    unsafe {
204        let inner: *mut VARIANT_0_0 =
205            core::ptr::addr_of_mut!(v.Anonymous.Anonymous).cast();
206        (*inner).vt = VT_UI4;
207        (*inner).wReserved1 = 0;
208        (*inner).wReserved2 = 0;
209        (*inner).wReserved3 = 0;
210        // VARIANT_0_0_0 是 union, ulVal 写入需要 unsafe
211        (*inner).Anonymous.ulVal = x;
212    }
213    v
214}

356impl H264Encoder {
357    /// [FIX-2026-06-24] `lazy_extract_sps_pps_from_output_type` 在 sps_pps_annex_b 仍空时
358    /// 每帧都会被 capture.rs 调一次。硬件 MFT 写 SPS/PPS 通常发生在第一帧 IDR 之后，
359    /// 之前 blob 是空的，这个过渡期 1-2 帧打一次正常，200+ 帧刷屏就成噪音。
360    /// 60 帧 ≈ 1 秒（@59fps），足够看出"持续拿不到"的异常，又不至于刷屏。
361    const SPARSE_LOG_EVERY: u64 = 60;

363    /// 构造一个编码器：先试硬件，失败回退到软件。
364    ///
365    /// `bitrate_bps` 用 bps 单位（1 Mbps = 1_000_000），会被写到
366    /// 输出 MediaType 的 `MF_MT_AVG_BITRATE` 字段，并在 `renegotiate` 中复用。
367    /// `slice_config` 决定 ICodecAPI 下发的 4 个 Slice 切分属性（详见 `H264SliceConfig`）。
368    pub fn new(
369        _device: &ID3D11Device,
370        width: u32, height: u32, fps: u32, bitrate_bps: u32,
371        profile: H264Profile,
372        slice_config: H264SliceConfig,
373    ) -> Result<Self> {

429    /// 分辨率变化时重协商 MediaType + 重建 NV12 纹理池。
430    ///
431    /// 触发场景：屏幕分辨率变了（用户改显示设置 / 切换屏幕），
432    /// 当前 MediaType 的 `MF_MT_FRAME_SIZE` 和新的帧尺寸对不上，
433    /// MFT 接下来会报 `MF_E_TRANSFORM_STREAM_CHANGE`。
434    ///
435    /// 调用方传入新尺寸（一般是 `frame.texture_desc()` 的 `Width`/`Height`），
436    /// 本方法会：
437    ///   1. 用新尺寸重新 SetInputType / SetOutputType（保留原 fps / bitrate / profile）
438    ///   2. 发 `NOTIFY_BEGIN_STREAMING` + `NOTIFY_START_OF_STREAM`，重置 MFT 内部流水线
439    ///   3. **关键**：drain 掉旧尺寸的 NV12 纹理池和 pending_textures，按新尺寸重建
440    ///   4. 软编路径同时重新分配 NV12 纹理 + CPU buffer
441    ///   5. 更新 `self.width` / `self.height`
442    pub fn renegotiate(&mut self, new_width: u32, new_height: u32) -> Result<()> {
443        if new_width == self.width && new_height == self.height {
444            return Ok(());
445        }
446        unsafe {
447            // [V4-C2] 微软文档规定：SetInputType/SetOutputType 之前必须先结束当前流 + flush。
448            //   否则 MFT 内部仍持有旧尺寸的 in-flight 帧引用：
449            //     - NVENC: SetInputType 后立即失效 → 后面帧全 garbage H.264 → 客户端解码崩
450            //     - AMF:   直接拒绝 SetOutputType，返回 MF_E_NOTACCEPTING
451            //     - QuickSync: 类似 NVENC
452            //   用 `let _ =` 容忍 MF_E_INVALID_STATE_TRANSITION（已不在 streaming）软错误。
453            let _ = self.encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_END_STREAMING, 0);
454            let _ = self.encoder.ProcessMessage(MFT_MESSAGE_COMMAND_FLUSH, 0);
455
456            // [V4-C2] flush 之后 MFT 不再引用 pending 里的纹理，把它们还回池子。
457            //   P1-1.1 修复时已经验证过这种 drain 模式是正确的（MFT 不再持有 → 安全还回）。
458            //   软编路径下 pending_textures 始终为空，drain 是 no-op。
459            for entry in self.pending_textures.drain(..) {
460                self.nv12_texture_pool.push_back(entry);
461            }
462
463            let frame_size = ((new_width as u64) << 32) | (new_height as u64);
464            let frame_rate = ((self.fps as u64) << 32) | 1;
465
466            // [SLICE] 重新下发 Slice 配置 —— 必须在新的 SetOutputType 之前调！
467            //   之前在 renegotiate 末尾 apply 的版本遇到软编时, ControlMode SetValue 会报
468            //   "User tried to set the slice mode after SetOutputType was called"。
469            //   移到 flush 之后、SetOutputType 之前, 三个属性 (SliceMode / ControlMode /
470            //   MaxNumber) 都能在合法时序下下发; 之前 4.1 节 "GUID 可能不对" 的警告也会消失。
471            Self::apply_slice_config_to(&self.encoder, &self.slice_config);
472
473            // 1. 重新设置输出 MediaType
474            let output_type = MFCreateMediaType()?;
475            output_type.SetGUID(&MF_MT_MAJOR_TYPE, &MFMediaType_Video)?;
476            output_type.SetGUID(&MF_MT_SUBTYPE, &MFVideoFormat_H264)?;
477            output_type.SetUINT64(&MF_MT_FRAME_SIZE, frame_size)?;
478            output_type.SetUINT64(&MF_MT_FRAME_RATE, frame_rate)?;
479            output_type.SetUINT32(&MF_MT_AVG_BITRATE, self.bitrate_bps)?;
480            output_type.SetUINT32(&MF_MT_INTERLACE_MODE, 2)?;
481            // [B3] 用 self.profile 替代硬编码 100
482            output_type.SetUINT32(&MF_MT_MPEG2_PROFILE, self.profile.to_mf_value())?;
483            // [B6 Phase 1] 显式 GOP：每 2 秒一个 IDR（fps * 2 帧）
484            // 让客户端断线重连时最多等 2 秒就能看到画面，而不是默认的 ~8 秒
485            output_type.SetUINT32(&MF_MT_MAX_KEYFRAME_SPACING, self.fps * 2)?;
486            self.encoder.SetOutputType(0, &output_type, 0)?;
487
488            // [V6-R1] 重新提取 SPS+PPS（尺寸变了 SPS 必然变）。
489            //   必须在 SetOutputType 之后调，output_type 才有 SPS/PPS 配置记录。
490            //   MFT 输出 MediaType 理论上一定包含 SPS/PPS；提取失败是编码器内部状态异常，
491            //   直接 ? 传播，让 Session::recover() 接走重建。
492            self.sps_pps_annex_b = extract_sps_pps_annex_b(&output_type)?;
493
494            // 2. 重新设置输入 MediaType
495            let input_type = MFCreateMediaType()?;
496            input_type.SetGUID(&MF_MT_MAJOR_TYPE, &MFMediaType_Video)?;
497            input_type.SetGUID(&MF_MT_SUBTYPE, &MFVideoFormat_NV12)?;
498            input_type.SetUINT64(&MF_MT_FRAME_SIZE, frame_size)?;
499            input_type.SetUINT64(&MF_MT_FRAME_RATE, frame_rate)?;
500            input_type.SetUINT32(&MF_MT_INTERLACE_MODE, 2)?;
501            input_type.SetUINT32(&MF_MT_DEFAULT_STRIDE, new_width)?;
502            // 软编路径需要 MF_MT_SAMPLE_SIZE（NV12 一帧 = w*h*3/2 字节）
503            if !self.is_hardware {
504                input_type.SetUINT32(&MF_MT_SAMPLE_SIZE, new_width * new_height * 3 / 2)?;
505            }
506            self.encoder.SetInputType(0, &input_type, 0)?;
507
508            // 3. 通知 MFT 重启流 —— 关键，没有这两行 MFT 内部状态不会刷新
509            self.encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_BEGIN_STREAMING, 0)?;
510            self.encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_START_OF_STREAM, 0)?;
511
512            // 4. 重建 NV12 纹理池 —— 旧纹理全是旧尺寸，VideoProcessorBlt 会失败
513            //    直接 clear()，VecDeque 内 (Texture, Buffer) 离开作用域自动 Release
514            if self.is_hardware {
515                self.nv12_texture_pool.clear();
516                self.pending_textures.clear();
517                // current_hw_* 由 convert_bgra_to_nv12_hw 在下一帧重新设置
518                self.current_hw_texture = None;
519                self.current_hw_buffer = None;
520
521                // 池大小 5 与 create_hw 保持一致（见 [MEMORY_OPTIMIZATION_V2.md §4]）
522                const HW_TEXTURE_POOL_SIZE: usize = 5;
523                for _ in 0..HW_TEXTURE_POOL_SIZE {
524                    let tex = Self::create_nv12_texture(&self.device, new_width, new_height)?;
525                    let buf = MFCreateDXGISurfaceBuffer(&ID3D11Texture2D::IID, &tex, 0, false)?;
526                    self.nv12_texture_pool.push_back((tex, buf));
527                }
528            } else {
529                // 软编路径：单个 NV12 纹理 + CPU buffer 都按新尺寸重分配
530                self.nv12_texture = Some(Self::create_nv12_texture(&self.device, new_width, new_height)?);
531                let buffer_size = new_width * new_height * 3 / 2;
532                self.cpu_buffer = Some(MFCreateMemoryBuffer(buffer_size)?);
533            }
534
535            // 5. 旧尺寸下可能还缓存了输出 sample，引用的是 MFT 旧内部状态，直接丢掉
536            self.recycled_output_sample = None;
537
538            // 6. 更新记录的尺寸
539            self.width = new_width;
540            self.height = new_height;
541
542            // [SLICE] Slice 配置已经在 flush 之后、新的 SetOutputType 之前下发过了
543            //   （本函数开头的 Self::apply_slice_config_to 那行）。
544            //   之前在末尾调 self.apply_slice_config() 的版本遇到软编 / 时序严格的
545            //   硬件 MFT 时会报 "User tried to set the slice mode after SetOutputType was called"。
546
547            Ok(())
548        }
549    }

551    /// 创建一个 NV12 格式的 D3D11 纹理。
552    /// 用作 VideoProcessor 的输出目标。
553    fn create_nv12_texture(_device: &ID3D11Device, width: u32, height: u32) -> Result<ID3D11Texture2D> {
554        unsafe {
555            let tex_desc = D3D11_TEXTURE2D_DESC {
556                Width: width,
557                Height: height,
558                MipLevels: 1,
559                ArraySize: 1,
560                // NV12 是 YUV 4:2:0 子采样，主流硬件编码器都吃这个格式。
561                Format: DXGI_FORMAT_NV12,
562                SampleDesc: DXGI_SAMPLE_DESC { Count: 1, Quality: 0 },
563                // DEFAULT = GPU 端最快访问，不要 STAGING（STAGING 才能 CPU 读）。
564                Usage: D3D11_USAGE_DEFAULT,
565                // 这个纹理要被 VideoProcessor 当 RT 写，又要被 DXGI Surface Buffer 当编码输入，
566                // 所以三种 bind flag 都要打。
567                BindFlags: (D3D11_BIND_RENDER_TARGET.0 | D3D11_BIND_VIDEO_ENCODER.0 | D3D11_BIND_SHADER_RESOURCE.0) as u32,
568                CPUAccessFlags: 0,
569                MiscFlags: 0,
570            };
571            let mut tex_opt: Option<ID3D11Texture2D> = None;
572            _device.CreateTexture2D(&tex_desc, None, Some(&mut tex_opt as *mut _))?;
573            tex_opt.ok_or_else(|| windows::core::Error::from(windows::Win32::Foundation::E_FAIL))
574        }
575    }

577    /// 查询 MFT 输出流的属性。
578    ///
579    /// 返回：
580    /// - `provides`: MFT 是否自己提供输出 sample
581    /// - `buffer_size`: 推荐输出 buffer 字节数
582    fn query_output_stream_info(encoder: &IMFTransform, width: u32, height: u32) -> (bool, u32) {
583        unsafe {
584            match encoder.GetOutputStreamInfo(0) {
585                Ok(stream_info) => {
586                    // dwFlags 是位掩码，看 MFT_OUTPUT_STREAM_PROVIDES_SAMPLES 这一位是否置 1
587                    let provides = (stream_info.dwFlags & MFT_OUTPUT_STREAM_PROVIDES_SAMPLES.0 as u32) != 0;
588                    // cbSize 是 MFT 推荐的输出 buffer 字节数
589                    // 如果它返回 0，就按 NV12 一帧的大小兜底（width * height * 3 / 2）
590                    let size = if stream_info.cbSize > 0 {
591                        stream_info.cbSize
592                    } else {
593                        width * height * 3 / 2
594                    };
595                    (provides, size)
596                }
597                Err(_) => (false, width * height * 3 / 2)
598            }
599        }

602    /// 拿一个输出 sample，准备喂给 `ProcessOutput`。
603    ///
604    /// - 如果 MFT 自己提供（`output_provides_samples=true`），返回 `None`，MFT 会自己 new。
605    /// - 否则优先从回收池里拿，复用前会清空 buffer 长度。
606    /// - 池为空时才 `MFCreateSample` 新建（避免每帧都走 COM 堆分配）。
607    pub fn create_output_sample(&mut self) -> Result<Option<IMFSample>> {
608        if self.output_provides_samples {
609            return Ok(None);
610        }
611        unsafe {
612            // 优先从池中复用
613            if let Some(sample) = self.recycled_output_sample.take() {
614                // 复用前清空 buffer 的当前长度
615                // （否则下次 Lock 出来数据会接着上次的尾巴）
616                // [H10 修复] SetCurrentLength(0) 失败时**不要**静默丢弃：
617                //   buffer 还残留上一帧的 cur_len → 编码器收到脏数据 → 输出混合了两帧的 H.264 流。
618                //   失败时走"新建 sample"路径：
619                //     1. 显式 drop 旧 sample（buffer + sample 一起 Release，避免引用泄漏）
620                //     2. 重新走"创建新 sample"逻辑
621                //   borrow-checker 友好做法：把"复用失败 → 新建"的逻辑抽到一个本地 closure
622                //   或者用 match 分支。
623                let reusable = if let Ok(buffer) = sample.GetBufferByIndex(0) {
624                    buffer.SetCurrentLength(0).is_ok()
625                } else {
626                    // 拿不到 buffer = sample 结构损坏，不能复用
627                    false
628                };
629                if reusable {
630                    return Ok(Some(sample));
631                }
632                // 复用失败：drop 旧 sample（buffer + sample 一起 Release）
633                drop(sample);
634                // fall through 到下面的"池为空时新建"分支
635            }
636
637            // 池为空 OR 复用失败 → 创建新 sample
638            let sample = MFCreateSample()?;
639            let buffer = MFCreateMemoryBuffer(self.output_buffer_size)?;
640            sample.AddBuffer(&buffer)?;
641            Ok(Some(sample))
642        }
643    }

652    /// 创建**硬件** H.264 编码器实例。
653    ///
654    /// 这是个非常典型的 Media Foundation 初始化流程：
655    ///   1. 创建并 Reset DXGIDeviceManager
656    ///   2. 用 `MFTEnumEx` 枚举出所有硬件 H.264 编码器
657    ///   3. 选第一个，`ActivateObject` 出真正的 `IMFTransform`
658    ///   4. 解锁异步（硬件 MFT 强制要求）
659    ///   5. 配置输入输出 MediaType 并 SetInputType/SetOutputType
660    ///   6. 发 ProcessMessage 开始流
661    fn create_hw(_device: &ID3D11Device, width: u32, height: u32, fps: u32, bitrate_bps: u32, profile: H264Profile, slice_config: H264SliceConfig) -> Result<Self> {
662        unsafe {
663            // 1. 创建 DXGIDeviceManager，让 MFT 可以直接读写我们的 GPU 显存
664            let mut dxgi_manager_opt: Option<IMFDXGIDeviceManager> = None;
665            let mut reset_token: u32 = 0;
666            MFCreateDXGIDeviceManager(&mut reset_token, &mut dxgi_manager_opt)?;
667            let dxgi_manager = dxgi_manager_opt.ok_or_else(|| windows::core::Error::from(windows::Win32::Foundation::E_FAIL))?;
668
669            // 把我们的 D3D11 设备塞给 manager。`IUnknown::cast` 是 `windows` crate 提供的
670            // 安全转型，内部走 QueryInterface。
671            let unk: IUnknown = _device.cast()?;
672            dxgi_manager.ResetDevice(&unk, reset_token)?;
673
674            // 2. 告诉系统我们要找：输入是 NV12、输出是 H.264 的视频编码器
675            let input_info = MFT_REGISTER_TYPE_INFO { guidMajorType: MFMediaType_Video, guidSubtype: MFVideoFormat_NV12 };
676            let output_info = MFT_REGISTER_TYPE_INFO { guidMajorType: MFMediaType_Video, guidSubtype: MFVideoFormat_H264 };
677
678            // 3. MFTEnumEx 枚举所有匹配的编码器（激活器形式）
679            //    返回的是一个 IMFActivate* 数组，由系统分配在 COM 任务堆上。
680            // [1.11 修复] 顺序：
681            //   1. MFTEnumEx 拿原始数组
682            //   2. 先 clone 所有到 Vec<IMFActivate>（每元素 AddRef，独立的 Rust 所有权）
683            //   3. 立刻 CoTaskMemFree 原始数组（不再 drop_in_place —— 数组内存是 COM 任务堆，
684            //      我们已经把所有权转到 Vec 里了，原数组不需要逐个 Release）
685            //   4. Vec 走正常 Rust Drop，未选中的激活器自动 Release
686            //   原实现的 drop_in_place 顺序和 clone 顺序耦合：
687            //   一旦 clone 之后、drop_in_place 之前发生 panic / 早返回，原数组内存就漏了。
688            let mut pp_activators: *mut Option<IMFActivate> = std::ptr::null_mut();
689            let mut count: u32 = 0;
690            MFTEnumEx(
691                MFT_CATEGORY_VIDEO_ENCODER,
692                MFT_ENUM_FLAG_HARDWARE | MFT_ENUM_FLAG_SORTANDFILTER,
693                Some(&input_info),
694                Some(&output_info),
695                &mut pp_activators,
696                &mut count,
697            )?;
698
699            if count == 0 {
700                if !pp_activators.is_null() {
701                    CoTaskMemFree(Some(pp_activators as *const core::ffi::c_void));
702                }
703                return Err(windows::core::Error::from(windows::Win32::Foundation::E_FAIL));
704            }
705
706            // 把所有激活器 clone 到 Vec（每个都 AddRef，独立的 Rust 所有权）
707            let raw_slice = std::slice::from_raw_parts(pp_activators, count as usize);
708            let activators: Vec<IMFActivate> = raw_slice
709                .iter()
710                .filter_map(|opt| opt.as_ref().cloned())
711                .collect();
712
713            // 立刻释放 CoTaskMemAlloc 的原始数组（每个元素已经被 clone 走了，不需要 drop_in_place）
714            CoTaskMemFree(Some(pp_activators as *const core::ffi::c_void));
715
716            // 4. 尝试按顺序激活，挑第一个成功的
717            //   软硬件混合环境里 MFTEnumEx 可能返回多个候选（NVENC / AMF / 微软软编等），
718            //   第一个未必能 ActivateObject（驱动问题），要 fall back。
719            let mut encoder_opt: Option<IMFTransform> = None;
720            #[cfg(any(debug_assertions, feature = "release-log"))]
721            let mut attempt_count: u32 = 0;
722            for act in &activators {
723                #[cfg(any(debug_assertions, feature = "release-log"))]
724                {
725                    attempt_count += 1;
726                }
727                match act.ActivateObject::<IMFTransform>() {
728                    Ok(t) => {
729                        encoder_opt = Some(t);
730                        #[cfg(any(debug_assertions, feature = "release-log"))]
731                        if attempt_count > 1 {
732                            println!("⚠️ [MFT] 第 {} 个候选硬件编码器 fallback (前 {} 个 ActivateObject 失败)",
733                                attempt_count, attempt_count - 1);
734                        }
735                        break;
736                    }
737                    Err(_e) => {
738                        #[cfg(any(debug_assertions, feature = "release-log"))]
739                        eprintln!("⚠️ [MFT] ActivateObject 失败 (attempt={}, code=0x{:08X}), 试下一个",
740                            attempt_count, _e.code().0 as u32);
741                        // 继续试下一个
742                    }
743                }
744            }
745            let encoder: IMFTransform = encoder_opt
746                .ok_or_else(|| windows::core::Error::from(windows::Win32::Foundation::E_FAIL))?;
747            // activators Vec 离开作用域，未选中的全部自动 Release
748
749            // 5. 硬件 MFT 默认是异步 + 锁定的，必须显式解锁才能用
750            //   MF_TRANSFORM_ASYNC_UNLOCK = 1 表示"我懂异步，我会自己驱动状态机"
751            let attributes = encoder.GetAttributes()?;
752            attributes.SetUINT32(&MF_TRANSFORM_ASYNC_UNLOCK, 1)?;
753
754            // 拿到事件生成器，后面 process_input/output 用它等 MFT 事件
755            let event_gen: IMFMediaEventGenerator = encoder.cast()?;
756
757            // 把 D3D Manager 告诉 MFT（之后 MFT 就能直接读写我们的 GPU 纹理了）

787            //   Encoder / 某些驱动版本 NVENC）会 SetOutputType 报 ERROR_NOT_FOUND
788            //   0x80070490 然后 create_hw 整体失败、误判为"硬件编码器不可用"——

791            //   - VIDEO_LEVEL: H.264 level。2048x1280@59fps 至少要 5.0, 5.1 更安全

885            eprintln!("[MFT] 11e. 准备 cast context -> ID3D11VideoContext");
886            let video_context: ID3D11VideoContext = context.cast()?;
887            #[cfg(any(debug_assertions, feature = "release-log"))]
888            eprintln!("[MFT] 11f. cast ID3D11VideoContext 成功");

898            // [SLICE] Slice 配置已经在 SET_D3D_MANAGER 之后、SetOutputType 之前下发过了
899            //   （本函数上面 Self::apply_slice_config_to(&encoder, &slice_config) 那行）。
900            //   之前在末尾调 `s.apply_slice_config()` 的版本有时序 bug：ControlMode 等
901            //   ICodecAPI 属性必须在 SetInputType 之前调, 否则会报 MF_E_INVALID_STATE_TRANSITION
902            //   "User tried to set the slice mode after SetOutputType was called"。
903            let s = Self {

905                event_gen: Some(event_gen),
906                pending_need_input: false,

910                nv12_texture: None,

916                //   MFT 内部会 AddRef，本函数返回时让它自动 Drop 即可。
917                staging_texture: None, cpu_buffer: None,
918                bgra_staging_texture: None,
919                use_cpu_conversion: false,
920                is_hardware: true, width, height,
921                fps, bitrate_bps,
922                recycled_output_sample: None,
923                recycled_input_sample: None,    // 硬编不用输入池
924                #[cfg(any(debug_assertions, feature = "release-log"))]
925                logged_first_input: false,
926                #[cfg(any(debug_assertions, feature = "release-log"))]
927                logged_first_output: false,
928                output_provides_samples,
929                output_buffer_size,
930                pending_force_idr: false,  // [B6 Phase 2]
931                profile,  // [B3]

1063    // ========== 色彩转换：统一入口 ==========
1064    /// BGRA → NV12 的总入口。会根据当前模式选硬件或 CPU 路径。
1065    pub fn convert_bgra_to_nv12(&mut self, bgra_texture: &ID3D11Texture2D) -> windows::core::Result<()> {
1066        if self.use_cpu_conversion {
1067            return self.convert_bgra_to_nv12_cpu(bgra_texture);

1070        match self.convert_bgra_to_nv12_hw(bgra_texture) {
1071            Ok(()) => Ok(()),
1072            Err(e) => {
1073                // 硬件色彩转换失败，回退到 CPU（仅软件编码器模式）
1074                if !self.is_hardware {
1075                    // [H2 修复] 加 cfg gate（错误路径但保持 release 0 开销）
1076                    #[cfg(any(debug_assertions, feature = "release-log"))]
1077                    println!("⚠️ 硬件色彩转换失败: {}，回退到 CPU 色彩转换...", e.message());
1078                    self.use_cpu_conversion = true;
1079                    return self.convert_bgra_to_nv12_cpu(bgra_texture);
1080                }
1081                Err(e)
1082            }
1083        }

1086    // ========== 硬件色彩转换（GPU Video Processor）==========
1087    /// 用 D3D11 VideoProcessor 把 BGRA 纹理转成 NV12 纹理，全程 GPU 操作，零 CPU 拷贝。
1088    ///
1089    /// VideoProcessor 本来是用来做颜色空间转换 / 缩放 / deinterlace 的，
1090    /// 这里我们只用到它的"BGRA → NV12 颜色空间转换"能力。
1091    fn convert_bgra_to_nv12_hw(&mut self, bgra_texture: &ID3D11Texture2D) -> windows::core::Result<()> {
1092        unsafe {
1093            // 根据模式确定目标 NV12 纹理
1094            // 硬件模式：从池里取一张可用的（轮换使用，避免被编码器 in-flight 引用时被覆盖）
1095            //
1096            // [P0-1 修复] pop 出来后立刻存进 current_hw_*；中间任何 `?` 早返回都必须手动
1097            //   回收回池子。否则 GPU 抖动 → CreateVideoProcessor*View 失败 → 池子单调缩水 → 池空崩。
1098            let nv12_texture = if self.is_hardware {
1099                let (tex, buf) = match self.nv12_texture_pool.pop_front() {
1100                    Some(x) => x,

1360    /// 当 GPU 不支持 VideoProcessor 时，CPU 兜底：把 BGRA 读到内存里手算 NV12。

1362    fn convert_bgra_to_nv12_cpu(&mut self, bgra_texture: &ID3D11Texture2D) -> windows::core::Result<()> {
1363        unsafe {
1364            // 首次调用时创建 BGRA staging 纹理（GPU 复制过来后 CPU 就能读了）
1365            if self.bgra_staging_texture.is_none() {
1366                let mut src_desc = D3D11_TEXTURE2D_DESC::default();
1367                bgra_texture.GetDesc(&mut src_desc);
1368                let staging_desc = D3D11_TEXTURE2D_DESC {
1369                    Width: src_desc.Width,
1370                    Height: src_desc.Height,
1371                    MipLevels: 1,
1372                    ArraySize: 1,
1373                    Format: src_desc.Format, // 通常是 DXGI_FORMAT_B8G8R8A8_UNORM
1374                    SampleDesc: DXGI_SAMPLE_DESC { Count: 1, Quality: 0 },
1375                    Usage: D3D11_USAGE_STAGING,
1376                    BindFlags: 0,
1377                    CPUAccessFlags: D3D11_CPU_ACCESS_READ.0 as u32,
1378                    MiscFlags: 0,
1379                };
1380                let mut staging_opt: Option<ID3D11Texture2D> = None;
1381                self.device.CreateTexture2D(&staging_desc, None, Some(&mut staging_opt as *mut _))?;
1382                self.bgra_staging_texture = staging_opt;
1383            }
1384
1385            // 复制 BGRA 纹理到 staging（GPU -> CPU 可读）
1386            // CopyResource 是异步的，但接下来 Map 会强制等它完成（CPU/GPU 同步点）
1387            let dst_resource: ID3D11Resource = self.bgra_staging_texture.as_ref().unwrap().cast()?;
1388            let src_resource: ID3D11Resource = bgra_texture.cast()?;
1389            self.context.CopyResource(&dst_resource, &src_resource);
1390
1391            // 映射 staging 纹理以读取 BGRA 数据
1392            // 这里产生一个 CPU/GPU 同步：CPU 阻塞直到 GPU 把 CopyResource 做完
1393            let mut mapped = D3D11_MAPPED_SUBRESOURCE::default();
1394            self.context.Map(&dst_resource, 0, D3D11_MAP_READ, 0, Some(&mut mapped as *mut _))?;
1395
1396            let bgra_stride = mapped.RowPitch as usize;
1397            let bgra_data = mapped.pData as *const u8;
1398            let width = self.width as usize;
1399            let height = self.height as usize;
1400            let y_size = width * height;
1401
1402            // 获取 CPU 缓冲区并锁定（拿到裸指针）
1403            let buffer = self.cpu_buffer.as_ref().ok_or_else(|| {
1404                windows::core::Error::from(windows::Win32::Foundation::E_FAIL)
1405            })?;
1406
1407            let mut nv12_data: *mut u8 = std::ptr::null_mut();
1408            let mut max_len = 0u32;
1409            let mut cur_len = 0u32;
1410            // [1.7 修复] Lock 失败时不能直接 ? 早返回：
1411            //   staging texture 此时已被 Map 但还没 Unmap，错误路径会泄露 GPU 资源。
1412            //   这里记日志 + 强制 Unmap + 返回 Err（让 capture 主循环走 recover 路径）
1413            if let Err(e) = buffer.Lock(
1414                &mut nv12_data,
1415                Some(&mut max_len as *mut u32),
1416                Some(&mut cur_len as *mut u32),
1417            ) {
1418                #[cfg(any(debug_assertions, feature = "release-log"))]
1419                eprintln!("❌ [cpu] cpu_buffer.Lock 失败: {:?} (已 Unmap staging)", e);
1420                self.context.Unmap(&dst_resource, 0);
1421                return Err(e);
1422            }
1423            // [1.5 修复] Lock 返回 Ok 也要验证指针 / 长度：
1424            //   理论上 Lock 成功时 MF 一定会返回非空指针 + max_len >= NV12 帧大小 (width*height*3/2)。
1425            //   实际遇到过驱动 bug 返回 Ok 但 max_len=0 / ptr=NULL 的情况，
1426            //   后续 ptr.add()/copy_nonoverlapping 会踩到 NULL deref 或者 size 0 的 malloc 区，
1427            //   都是未定义行为（UB）。这里防御性检查并显式 fail fast。
1428            let expected_size = (width * height * 3 / 2) as u32;
1429            if nv12_data.is_null() || max_len < expected_size {
1430                #[cfg(any(debug_assertions, feature = "release-log"))]
1431                eprintln!(
1432                    "❌ [cpu] cpu_buffer.Lock 返回 Ok 但状态异常: ptr={:?}, max_len={}, 期望 >= {}",
1433                    nv12_data, max_len, expected_size
1434                );

1437                return Err(windows::core::Error::from(windows::Win32::Foundation::E_FAIL));
1438            }

1442            let uv_plane = nv12_data.add(y_size);
1443            let nv12_stride = width; // MF_MT_DEFAULT_STRIDE = width
1444
1445            // ---- Y 平面：BGRA -> Y (BT.601 limited range) ----
1446            // 公式：Y = ((66*R + 129*G + 25*B + 128) >> 8) + 16
1447            // limited range 表示 Y 取值 [16, 235]，对应实际亮度
1448            for row in 0..height {
1449                let bgra_row = bgra_data.add(row * bgra_stride);
1450                let y_row = y_plane.add(row * nv12_stride);
1451                for col in 0..width {
1452                    let b = *bgra_row.add(col * 4) as i32;
1453                    let g = *bgra_row.add(col * 4 + 1) as i32;
1454                    let r = *bgra_row.add(col * 4 + 2) as i32;
1455                    let y = ((66 * r + 129 * g + 25 * b + 128) >> 8) + 16;

1458            }

1462            //       V = ((112*R - 94*G - 18*B + 128) >> 8) + 128
1463            // UV 平面分辨率是 Y 的一半（4:2:0），每 2x2 像素共用一组 UV
1464            for row in 0..(height / 2) {
1465                let bgra_row0 = bgra_data.add(row * 2 * bgra_stride);
1466                let bgra_row1 = bgra_data.add((row * 2 + 1) * bgra_stride);
1467                let uv_row = uv_plane.add(row * nv12_stride);
1468                for col in 0..(width / 2) {
1469                    // 取 2x2 像素块的平均值
1470                    let i00 = col * 2 * 4;
1471                    let i01 = (col * 2 + 1) * 4;
1472
1473                    let b0 = *bgra_row0.add(i00) as i32;
1474                    let g0 = *bgra_row0.add(i00 + 1) as i32;
1475                    let r0 = *bgra_row0.add(i00 + 2) as i32;
1476
1477                    let b1 = *bgra_row0.add(i01) as i32;
1478                    let g1 = *bgra_row0.add(i01 + 1) as i32;
1479                    let r1 = *bgra_row0.add(i01 + 2) as i32;
1480
1481                    let b2 = *bgra_row1.add(i00) as i32;
1482                    let g2 = *bgra_row1.add(i00 + 1) as i32;
1483                    let r2 = *bgra_row1.add(i00 + 2) as i32;
1484
1485                    let b3 = *bgra_row1.add(i01) as i32;
1486                    let g3 = *bgra_row1.add(i01 + 1) as i32;
1487                    let r3 = *bgra_row1.add(i01 + 2) as i32;
1488
1489                    let b = (b0 + b1 + b2 + b3) / 4;
1490                    let g = (g0 + g1 + g2 + g3) / 4;
1491                    let r = (r0 + r1 + r2 + r3) / 4;
1492
1493                    let u = ((-38 * r - 74 * g + 112 * b + 128) >> 8) + 128;
1494                    let v = ((112 * r - 94 * g - 18 * b + 128) >> 8) + 128;
1495
1496                    *uv_row.add(col * 2) = u.clamp(0, 255) as u8;     // U (Cb)
1497                    *uv_row.add(col * 2 + 1) = v.clamp(0, 255) as u8; // V (Cr)
1498                }
1499            }
1500
1501            // 设置 buffer 当前长度（重要，否则编码器不知道这一帧实际有多少字节）
1502            // [1.7 修复] SetCurrentLength/Unlock 失败时也不能直接 ?：
1503            //   buffer 处于 Locked 状态 + staging 处于 Mapped 状态，
1504            //   错误路径必须先 Unlock + Unmap 再返回，避免锁死。
1505            let total_bytes = (y_size + width * height / 2) as u32;
1506            if let Err(e) = buffer.SetCurrentLength(total_bytes) {
1507                #[cfg(any(debug_assertions, feature = "release-log"))]
1508                eprintln!("❌ [cpu] cpu_buffer.SetCurrentLength 失败: {:?}", e);
1509                let _ = buffer.Unlock();
1510                self.context.Unmap(&dst_resource, 0);
1511                return Err(e);
1512            }
1513            if let Err(e) = buffer.Unlock() {
1514                #[cfg(any(debug_assertions, feature = "release-log"))]
1515                eprintln!("❌ [cpu] cpu_buffer.Unlock 失败: {:?}", e);
1516                self.context.Unmap(&dst_resource, 0);
1517                return Err(e);
1518            }
1519            self.context.Unmap(&dst_resource, 0);
1520
1521            Ok(())
1522        }
1523    }

1525    // ========== 构建 IMFSample ==========
1526    /// 把当前帧打包成一个 IMFSample，准备喂给 `ProcessInput`。
1527    ///
1528    /// 三种路径：
1529    /// - 硬件 MFT：直接用 DXGI Surface Buffer（指向刚 VideoProcessor 写过的 NV12 纹理）
1530    /// - 软件 MFT + CPU 转换：直接用 cpu_buffer（数据已经在里面了）
1531    /// - 软件 MFT + 硬件转换：把 NV12 纹理拷到 staging，CPU 拷到 cpu_buffer
1532    pub fn build_sample(&mut self, timestamp: i64, duration: i64) -> Result<IMFSample> {
1533        unsafe {
1534            // MFCreateSample: 创建一个空的"样本容器"
1535            // [修复 §3.3] 软编路径复用输入 sample：
1536            //   - 池空时走 MFCreateSample；命中池则 RemoveAllBuffers 后复用
1537            //   - 硬编异步消费，不能复用（MFT 可能还在 GPU 队列里没读）
1538            //   - 软编同步消费，MFT 不会自己清 sample 的 buffers，必须手动 RemoveAllBuffers
1539            // [P0-2 修复] 硬件路径 MFCreateSample 失败时不需要回收池子（还没拿 tex/buf），
1540            //   但 sample 本身已经分配要让它正常 Drop。改成 match。
1541            let sample = if self.is_hardware {
1542                match MFCreateSample() {
1543                    Ok(s) => s,
1544                    Err(e) => {
1545                        // MFCreateSample 失败但 current_hw_* 还占着 → 还回池子
1546                        if let (Some(tex), Some(buf)) = (
1547                            self.current_hw_texture.take(),
1548                            self.current_hw_buffer.take(),
1549                        ) {
1550                            self.nv12_texture_pool.push_back((tex, buf));
1551                        }
1552                        return Err(e);
1553                    }
1554                }
1555            } else if let Some(s) = self.recycled_input_sample.take() {
1556                let _ = s.RemoveAllBuffers();
1557                s
1558            } else {
1559                MFCreateSample()?
1560            };
1561            // 设置时间戳（PTS）和时长，用于解码端按帧间隔播放
1562            // [P0-2 修复] 失败时 sample 已经构造 + current_hw_* 还占着 → 都得手动回收
1563            if let Err(e) = sample.SetSampleTime(timestamp) {
1564                // sample 直接 Drop（自动 Release）
1565                if self.is_hardware {
1566                    if let (Some(tex), Some(buf)) = (
1567                        self.current_hw_texture.take(),
1568                        self.current_hw_buffer.take(),
1569                    ) {
1570                        self.nv12_texture_pool.push_back((tex, buf));
1571                    }
1572                }
1573                return Err(e);
1574            }
1575            if let Err(e) = sample.SetSampleDuration(duration) {
1576                if self.is_hardware {
1577                    if let (Some(tex), Some(buf)) = (
1578                        self.current_hw_texture.take(),
1579                        self.current_hw_buffer.take(),
1580                    ) {
1581                        self.nv12_texture_pool.push_back((tex, buf));
1582                    }
1583                }
1584                return Err(e);
1585            }
1586
1587            if self.is_hardware {
1588                // 硬件路径：把当前帧的 DXGI Surface Buffer 挂到 sample 上
1589                // 当前帧的纹理和 buffer 是上一帧 `convert_bgra_to_nv12_hw` 设置的
1590                let tex = self.current_hw_texture.take().ok_or_else(|| windows::core::Error::from(windows::Win32::Foundation::E_FAIL))?;
1591                let buf = self.current_hw_buffer.take().ok_or_else(|| windows::core::Error::from(windows::Win32::Foundation::E_FAIL))?;
1592
1593                // [P0-2 修复] AddBuffer 失败时 (tex, buf) 还没进 pending_textures → 还回池子
1594                if let Err(e) = sample.AddBuffer(&buf) {
1595                    self.nv12_texture_pool.push_back((tex, buf));
1596                    return Err(e);
1597                }
1598                // 这张纹理已经交给 MFT，等它消费完才能复用，所以压进 pending_textures
1599                self.pending_textures.push_back((tex, buf));
1600            } else if self.use_cpu_conversion {
1601                // CPU 转换：数据已经直接写入 cpu_buffer 了
1602                sample.AddBuffer(self.cpu_buffer.as_ref().unwrap())?;
1603            } else {
1604                // 软件编码器 + 硬件色彩转换的"半 GPU"路径
1605                // 需要把 NV12 纹理读回 CPU，再写到 cpu_buffer
1606                let dst_resource: ID3D11Resource = self.staging_texture.as_ref().unwrap().cast()?;
1607                let src_resource: ID3D11Resource = self.nv12_texture.as_ref().unwrap().cast()?;
1608                self.context.CopyResource(&dst_resource, &src_resource);
1609
1610                let mut mapped = D3D11_MAPPED_SUBRESOURCE::default();
1611                self.context.Map(&dst_resource, 0, D3D11_MAP_READ, 0, Some(&mut mapped as *mut _))?;
1612
1613                let stride = mapped.RowPitch as usize;
1614                let width = self.width as usize;
1615                let height = self.height as usize;
1616                let y_size = width * height;
1617
1618                let buffer = self.cpu_buffer.as_ref().unwrap();
1619
1620                let mut data: *mut u8 = std::ptr::null_mut();
1621                let mut max_len = 0u32;
1622                let mut cur_len = 0u32;
1623                // [1.7 修复] Lock 失败时不能直接 ?：staging 此时 Mapped，错误路径会泄露 GPU 资源。
1624                //   记录日志 + 强制 Unmap staging + 把 sample 丢掉（不喂给编码器）+ 返回 Err
1625                if let Err(e) = buffer.Lock(
1626                    &mut data,
1627                    Some(&mut max_len as *mut u32),
1628                    Some(&mut cur_len as *mut u32),
1629                ) {
1630                    #[cfg(any(debug_assertions, feature = "release-log"))]
1631                    eprintln!("❌ [build_sample] cpu_buffer.Lock 失败: {:?} (丢 sample + Unmap staging)", e);
1632                    self.context.Unmap(&dst_resource, 0);
1633                    return Err(e);
1634                }
1635                // [1.5 修复] 同 cpu 路径：Lock 返回 Ok 也要验证指针 / 长度。
1636                //   见 convert_bgra_to_nv12_cpu 里的同款注释。
1637                let expected_size2 = (width * height * 3 / 2) as u32;
1638                if data.is_null() || max_len < expected_size2 {
1639                    #[cfg(any(debug_assertions, feature = "release-log"))]
1640                    eprintln!(
1641                        "❌ [build_sample] cpu_buffer.Lock 返回 Ok 但状态异常: ptr={:?}, max_len={}, 期望 >= {}",
1642                        data, max_len, expected_size2
1643                    );
1644                    let _ = buffer.Unlock();
1645                    self.context.Unmap(&dst_resource, 0);
1646                    return Err(windows::core::Error::from(windows::Win32::Foundation::E_FAIL));
1647                }
1648
1649                // 把 staging 里的 NV12 数据（Y 平面 + UV 平面）一行行拷到 cpu_buffer
1650                let y_src = mapped.pData as *const u8;
1651                let uv_src = y_src.add(stride * height);
1652
1653                for y in 0..height {
1654                    std::ptr::copy_nonoverlapping(y_src.add(y * stride), data.add(y * width), width);
1655                }
1656                for y in 0..(height / 2) {
1657                    std::ptr::copy_nonoverlapping(uv_src.add(y * stride), data.add(y_size + y * width), width);
1658                }
1659
1660                // [1.7 修复] SetCurrentLength/Unlock 失败时也不能直接 ?：
1661                //   同样的 staging Mapped + cpu_buffer Locked 资源泄露问题。
1662                //   这里要把 sample 丢掉（已经构造了）不喂给编码器，否则编码器收到残缺数据
1663                //   会输出脏 H.264 流。
1664                let total_bytes2 = (y_size + width * height / 2) as u32;
1665                if let Err(e) = buffer.SetCurrentLength(total_bytes2) {
1666                    #[cfg(any(debug_assertions, feature = "release-log"))]
1667                    eprintln!("❌ [build_sample] cpu_buffer.SetCurrentLength 失败: {:?} (丢 sample)", e);
1668                    let _ = buffer.Unlock();
1669                    self.context.Unmap(&dst_resource, 0);
1670                    return Err(e);
1671                }
1672                if let Err(e) = buffer.Unlock() {
1673                    #[cfg(any(debug_assertions, feature = "release-log"))]
1674                    eprintln!("❌ [build_sample] cpu_buffer.Unlock 失败: {:?} (丢 sample)", e);
1675                    self.context.Unmap(&dst_resource, 0);
1676                    return Err(e);
1677                }
1678                self.context.Unmap(&dst_resource, 0);
1679
1680                sample.AddBuffer(buffer)?;
1681            }
1682            Ok(sample)
1683        }
1684    }
1685    // 当 ProcessOutput 成功输出一帧时，回收最早输入的纹理
1686    /// 把 MFT 已经"消费完"的最早一张 NV12 纹理还回纹理池。
1687    /// 因为 MFT 内部有流水线，输入和输出不是 1:1 的，
1688    /// 所以用 `pending_textures` 队列来追踪 in-flight 的纹理。
1689    pub fn recycle_texture(&mut self) {
1690        if self.is_hardware {
1691            if let Some((tex, buf)) = self.pending_textures.pop_front() {
1692                self.nv12_texture_pool.push_back((tex, buf));
1693            }
1694        }
1695    }

1700    pub fn sps_pps_annex_b(&self) -> &[u8] {

1704    /// [FIX v5] 硬件 MFT（D3D11 Video Encoder / NVENC）不把 SPS/PPS 写进 bitstream，
1705    ///   而是写到 output MediaType 的 `MF_MT_MPEG_SEQUENCE_HEADER` 属性里——
1706    ///   但**首帧之后**才填（v5 之前的版本以为 bitstream 里能扫到，实测不行）。
1707    ///
1708    ///   本函数是补救路径：调 `encoder.GetOutputCurrentType(0)` 拿到**当前实际**的
1709    ///   output MediaType（首帧之后 `MF_MT_MPEG_SEQUENCE_HEADER` 已经填好），
1710    ///   再走 `extract_sps_pps_annex_b` 路径解析 AVCC blob 成 Annex-B。
1711    ///
1712    ///   调用方约定（capture.rs 主循环）：
1713    ///   ```text
1714    ///   if encoder.sps_pps_annex_b().is_empty() {
1715    ///       encoder.lazy_extract_sps_pps_from_output_type();
1716    ///   }
1717    ///   ```
1718    ///   只在 sps_pps_annex_b 为空时调，避免每帧重复 IO 浪费 CPU。
1719    ///
1720    ///   软编（`CLSID_MSH264EncoderMFT`）**不需要**走这条路径 —— 它在
1721    ///   `SetOutputType` 之后就立刻把 SPS/PPS 写到 output_type，
1722    ///   `extract_sps_pps_annex_b` 在 create_sw 末尾能拿到 blob。
1723    pub fn lazy_extract_sps_pps_from_output_type(&mut self) {
1724        // [FIX-2026-06-24] 节流：本函数 capture.rs 每帧都会调一次（sps_pps_annex_b 空时），
1725        //   blob 为空是硬件 MFT 的正常过渡状态。但若一直为空（例如上层修了 capture.rs 的
1726        //   ConvertToContiguousBuffer 路径之后 MFT 仍未填 blob），每帧打一次会刷屏。
1727        //   改为：首次失败 + 之后每 SPARSE_LOG_EVERY 帧打一次。
1728        //   计数器放 self 上而不是函数局部，避免丢帧计数失真。
1729        self.sps_pps_retry_count += 1;
1730        let should_log = self.sps_pps_retry_count == 1
1731            || self.sps_pps_retry_count % Self::SPARSE_LOG_EVERY == 0;
1732
1733        // GetOutputCurrentType 在 ProcessOutput 至少成功一次之后，MediaType 属性才完整
1734        let out_type = unsafe {
1735            match self.encoder.GetOutputCurrentType(0) {
1736                Ok(t) => t,
1737                Err(e) => {
1738                    #[cfg(any(debug_assertions, feature = "release-log"))]
1739                    if should_log {
1740                        eprintln!("[SPS] GetOutputCurrentType 失败: {} (0x{:08X}), 暂时拿不到 SPS/PPS",
1741                            e.message(), e.code().0 as u32);
1742                    }
1743                    return;
1744                }
1745            }
1746        };
1747        match extract_sps_pps_annex_b(&out_type) {
1748            Ok(blob) if !blob.is_empty() => {
1749                #[cfg(any(debug_assertions, feature = "release-log"))]
1750                eprintln!("[SPS] 从 output MediaType 懒提取 SPS/PPS: {} bytes (retry={})",
1751                    blob.len(), self.sps_pps_retry_count);
1752                self.sps_pps_annex_b = blob;
1753            }
1754            Ok(_) => {
1755                #[cfg(any(debug_assertions, feature = "release-log"))]
1756                if should_log {
1757                    eprintln!("[SPS] GetOutputCurrentType 成功但 blob 为空, MFT 还没写 SPS/PPS, retry={}",
1758                        self.sps_pps_retry_count);
1759                }
1760            }
1761            Err(e) => {
1762                #[cfg(any(debug_assertions, feature = "release-log"))]
1763                if should_log {
1764                    eprintln!("[SPS] extract_sps_pps_annex_b(GetOutputCurrentType) 失败: {} (0x{:08X}), retry={}",
1765                        e.message(), e.code().0 as u32, self.sps_pps_retry_count);
1766                }
1767            }
1768        }
1769    }

1771    // ========== [B6 Phase 2] 强制 IDR ==========
1772
1773    /// 用户请求"下一帧必须是 IDR"。
1774    ///
1775    /// 典型场景：客户端刚连入 / 客户端断线重连 / 编码器自愈（C2 recover）后。
1776    /// 在下一次 `process_input` 调用时被消费。
1777    ///
1778    /// 实现策略：直接走 flush + restart 路径。`ICodecAPI` 路径在 windows crate
1779    /// 里要手搓 VARIANT（`ICodecAPI::SetValue` 接的是 `*const VARIANT`），
1780    /// 太复杂，性价比不高。flush 会丢 1-2 帧但保证 IDR，远程控制场景可接受。
1781    /// 行为：每次请求 IDR = 丢 1-2 帧 + 下一帧 IDR。
1782    ///
1783    /// 调用方：capture loop 的 proactive 分辨率变化 + MFT STREAM_CHANGE 分支，
1784    /// 以及 `Session::recover` / `Session::recreate` 重建后立即触发。
1785    pub fn request_keyframe(&mut self) {
1786        self.pending_force_idr = true;
1787    }
1788
1789    /// flush MFT + 重启流，强制下一帧为 IDR。
1790    /// 会丢弃所有 pending 帧（可能丢 1-2 帧画面），但保证下一帧是 IDR。
1791    unsafe fn force_idr_via_flush(&mut self) -> windows::core::Result<()> {
1792        // 1. flush —— 清空 MFT 内部缓冲的输入帧
1793        unsafe {
1794            self.encoder.ProcessMessage(MFT_MESSAGE_COMMAND_FLUSH, 0)?;
1795            // 2. 重启流（这些消息已经在 create_hw / create_sw 里发过一次，再发一次 idempotent）
1796            self.encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_BEGIN_STREAMING, 0)?;
1797            self.encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_START_OF_STREAM, 0)?;
1798        }
1799
1800        // 3. [P1-1.1 修复] flush 后 MFT 已经不引用 pending 里的纹理，但 clear() 只是 Drop
1801        //   元素——纹理没回到 nv12_texture_pool。request_keyframe 一旦启用，池子会单调缩水。
1802        //   改成 drain 回池子。
1803        for entry in self.pending_textures.drain(..) {
1804            self.nv12_texture_pool.push_back(entry);
1805        }
1806
1807        // 4. [P1-1.1 修复] 清掉 recycled_output_sample：flush 后它可能引用 MFT 旧内部状态
1808        //   （MFT 内部 sample 缓冲被 flush 清了，再还回去 MFT 行为未定义）。
1809        self.recycled_output_sample = None;
1810
1811        Ok(())
1812    }

1816    /// 把 `self.slice_config` 里所有 `Some(...)` 字段下发给 MFT（走 ICodecAPI）。
1817    ///
1818    /// # 调用时机
1819    /// **必须在 `SetOutputType` 之前调用**——Media Foundation 规定
1820    /// `AVEncSliceMode` / `AVEncSliceControlMode` / `AVEncSliceMaxNumber`
1821    /// 这些属性只能在 `SetInputType` 之前下发；`SetOutputType` 之后改会报
1822    /// `MF_E_INVALID_STATE_TRANSITION` "User tried to set the slice mode
1823    /// after SetOutputType was called"。
1824    ///
1825    /// 三个调用点：
1826    /// - `create_hw` / `create_sw`：在 `SetOutputType` 之前（SET_D3D_MANAGER 之后）
1827    /// - `renegotiate`：在新的 `SetOutputType` 之前（end_of_stream + flush 之后）
1828    ///
1829    /// # 失败处理
1830    /// 任何子项失败都只 log 不传播：
1831    /// - MFT 不支持 ICodecAPI 整体 → log + 直接 return
1832    /// - MFT 不支持某具体 GUID → log + 跳过该项
1833    /// - SetValue 返回 E_INVALIDARG（参数非法组合）→ log + 继续下一项
1834    /// 这样保证"用户给的某个值 MFT 不支持"不会让整个 encoder 挂掉。
1835    /// 全部 cfg 守门，release 模式 0 开销。
1836    ///
1837    /// # 安全
1838    /// - `cast::<ICodecAPI>()` 是 `QueryInterface` → 返回 owned 智能指针，函数结尾自动 Release。
1839    /// - `variant_ui4` 走 `zeroed()` + 手工设 `vt=VT_UI4` / `ulVal` 标准模式，ManuallyDrop 字段不会被读。
1840    #[allow(dead_code)] // 当前 `create_hw` / `create_sw` / `renegotiate` 都直接调 `apply_slice_config_to`。
1841        //  留这个公开方法是为了未来 "运行时改 Slice 配置"（POTENTIAL_ISSUES TODO：需要先 flush
1842        //  stream, 然后调本方法, 然后再 force_idr_via_flush 重启）。

1845    }

1847    /// 内部 static helper：把 `slice_config` 下发到任意 `IMFTransform`。
1848    /// 设计成接受 `&IMFTransform` + `&H264SliceConfig`，这样在 `Self` 构造前
1849    /// 也能调（`create_hw` / `create_sw` / `renegotiate` 在 `let mut s = Self { ... }`
1850    /// 之前就要触发；公开方法 `apply_slice_config` 是给"已构造好的实例"用的薄封装）。
1851    fn apply_slice_config_to(encoder: &IMFTransform, slice_config: &H264SliceConfig) {
1852        // 1) 拿 ICodecAPI
1853        let codec_api: ICodecAPI = match encoder.cast::<ICodecAPI>() {
1854            Ok(c) => c,
1855            Err(e) => {
1856                #[cfg(any(debug_assertions, feature = "release-log"))]
1857                eprintln!("⚠️ [slice] MFT 不支持 ICodecAPI, slice 参数不应用: {}", e.message());
1858                return;
1859            }
1860        };
1861
1862        // 2) SetValue(SliceMode)
1863        if let Some(mode) = slice_config.mode {
1864            // 先 IsSupported 检查：硬件 MFT 经常静默接受 SetValue 但忽略值，
1865            // 提前 IsSupported 让"不支持"的 case 走 cfg log 而不是悄悄吞。
1866            match unsafe { codec_api.IsSupported(&CODECAPI_AVEncSliceMode) } {
1867                Ok(()) => {
1868                    let v = variant_ui4(mode as u32);

2128                }
2129                Err(e) => {
2130                    // [1.2 修复] 任何非 NEED_MORE_INPUT 的错误也走 recycle 路径：
2131                    //   原实现 Err(e) 直接返回，out_sample 走出作用域被 Release，

2139            }
2140        }

2145// ========== [V6-R1] H.264 Annex-B 字节流转换（模块级自由函数）==========
2146
2147/// [V6-R1] 从输出 MediaType 提取 SPS+PPS AVCC 配置记录，转成 Annex-B 字节流。
2148///
2149/// ## 为什么需要这个函数
2150///
2151/// MFT 用 `MFVideoFormat_H264` 输出时，码流是 **AVCC 格式**（H.264 Annex E）：
2152/// - 每个 NALU 前有 4 字节大端长度前缀
2153/// - SPS/PPS **不在码流里**，单独存在 MediaType 的 `MF_MT_MPEG_SEQUENCE_HEADER` blob
2154///
2155/// 但 ffmpeg / 浏览器 MSE / libavcodec / VideoToolbox 期望 **Annex-B 格式**（H.264 Annex B）：
2156/// - 每个 NALU 前有 `00 00 00 01` 起始码
2157/// - SPS/PPS 必须独立出现在 IDR 帧**之前**
2158///
2159/// 所以本函数：
2160/// 1. 调 `output_type.GetBlob(MF_MT_MPEG_SEQUENCE_HEADER, ...)` 拿 AVCC 配置 blob
2161/// 2. 按 ISO/IEC 14496-15 §5.2.4.1 解析（5 字节头 + num_sps + 每个 sps 长度 + NALU + num_pps + 同理）
2162/// 3. 把每个 SPS/PPS NALU 用 `00 00 00 01` 起始码包起来
2163///
2164/// ## 失败处理
2165///
2166/// 返回 `Err` 而不是 `panic!`：MFT 输出 MediaType **理论上一定** 包含 SPS/PPS，
2167/// 但如果 MFT 实现 bug 或者某种 degenerate 情况 blob 不存在，应该 graceful 失败
2168/// 让调用方 fallback 到空 Vec（不阻塞编码流程）。
2169///
2170/// ## 安全
2171///
2172/// `unsafe` 是必须的（GetBlob 接裸指针）。
2173pub fn extract_sps_pps_annex_b(output_type: &IMFMediaType) -> windows::core::Result<Vec<u8>> {
2174    unsafe {
2175        // 1. 先问 blob 长度（windows 0.62.2 的 GetBlobSize 返回 Result<u32>）
2176        //
2177        // [FIX v5] 硬件 MFT（D3D11 Video Encoder / NVENC 等）**不会**在 SetOutputType 之后
2178        //   立即把 MF_MT_MPEG_SEQUENCE_HEADER 写到 output MediaType —— 要等编出第一帧后
2179        //   才填充。v4 之前的版本直接 `?` 早返回 0xC00D36E6 / 0xC00D36C4
2180        //   （MF_E_ATTRIBUTENOTFOUND 或 MF_E_TRANSFORM_NOT_POSSIBLE_FOR_*），
2181        //   create_hw 整体失败、误判为"硬件编码器不可用"。
2182        //
2183        //   修复：属性不存在时返回 Ok(Vec::new())，让上层 capture.rs 在收到首帧输出
2184        //   sample 后再从 H.264 bitstream 里扫 0x00000001 起始码提取 SPS/PPS 补回
2185        //   sps_pps_annex_b 字段。
2186        //
2187        //   软编（CLSID_MSH264EncoderMFT）走另一条路径，它在 SetOutputType 之后立刻就
2188        //   填充 SPS/PPS，所以这里直接拿到 blob 是正常路径。
2189        let blob_len = match output_type.GetBlobSize(&MF_MT_MPEG_SEQUENCE_HEADER) {
2190            Ok(n) => n,
2191            Err(e) => {
2192                #[cfg(any(debug_assertions, feature = "release-log"))]
2193                eprintln!("[SPS] output_type.GetBlobSize(MF_MT_MPEG_SEQUENCE_HEADER) 失败: {} (0x{:08X}), 硬件 MFT 不会立刻写 SPS/PPS, 改从首帧提取",
2194                    e.message(), e.code().0 as u32);
2195                return Ok(Vec::new());
2196            }
2197        };
2198
2199        if blob_len == 0 {
2200            return Ok(Vec::new());
2201        }
2202
2203        // 2. 分配 + 实际取 blob（windows 0.62.2 GetBlob 签名：*const GUID, &mut [u8], Option<*mut u32>）
2204        let mut blob = vec![0u8; blob_len as usize];
2205        if let Err(e) = output_type.GetBlob(&MF_MT_MPEG_SEQUENCE_HEADER, &mut blob, None) {
2206            #[cfg(any(debug_assertions, feature = "release-log"))]
2207            eprintln!("[SPS] output_type.GetBlob(MF_MT_MPEG_SEQUENCE_HEADER) 失败: {} (0x{:08X}), 改从首帧提取",
2208                e.message(), e.code().0 as u32);
2209            return Ok(Vec::new());
2210        }

2213        //    layout:
2214        //      [1] configurationVersion (固定 1)
2215        //      [1] AVCProfileIndication
2216        //      [1] profile_compatibility
2217        //      [1] AVCLevelIndication
2218        //      [1] reserved(6 bits=111111) + lengthSizeMinusOne(2 bits)
2219        //      [1] numOfSequenceParameterSets (高 3 bit 是 reserved=111，低 5 bit 是 count)
2220        //      [2] sps_length (BE)
2221        //      [sps_length] sps_data
2222        //      ...重复每个 SPS
2223        //      [1] numOfPictureParameterSets
2224        //      [2] pps_length (BE)
2225        //      [pps_length] pps_data
2226        //      ...重复每个 PPS

2228        let mut c = 5usize; // 跳过 configurationVersion/AVCProfile/profile_compat/AVCLevel/lengthSize
2229
2230        if blob.len() < 6 {
2231            return Err(windows::core::Error::from(windows::Win32::Foundation::E_FAIL));
2232        }

2235        c += 1;

2237        for _ in 0..num_sps {
2238            if c + 2 > blob.len() {
2239                break; // 防御性：blob 截断就跳出

2244                break;
2245            }

2249        }
2250

2254

2256        c += 1;
2257

2278/// 历史注释：当时为定位 "5 字节 is_keyframe=false" 输出打的诊断，
2279/// 确认 NVENC 输出是 Annex-B（含 AUD + SPS + PPS + IDR slice），不是 AVCC。
2280
2281/// [V6-R1] AVCC 字节流转 Annex-B。
2282///
2283/// ## 算法
2284///
2285/// AVCC 格式：
2286/// ```text
2287/// [u32 BE length1][NALU1 bytes][u32 BE length2][NALU2 bytes]...
2288/// ```
2289///
2290/// Annex-B 格式：
2291/// ```text
2292/// [00 00 00 01][NALU1 bytes][00 00 00 01][NALU2 bytes]...
2293/// ```
2294///
2295/// 所以转换就是把每个 4 字节长度前缀换成 `00 00 00 01` 起始码。
2296///
2297/// 第一个 IDR NALU（header byte & 0x1f == 5）**之前**插入 `prepend_sps_pps`（已
2298/// 是 Annex-B 格式的 SPS+PPS 串）。这保证客户端解码器拿到 IDR 时已经有序列头。
2299///
2300/// 非 IDR 帧（P/B slice）不重复插 SPS/PPS（节省带宽，但有些客户端为简化处理每帧都带——可接受，
2301/// 因为 `sps_pps_annex_b` 缓存里只有一次 IDR 触发插）。
2302///
2303/// ## 边界
2304///
2305/// - 返回 `(Vec<u8>, bool)`：第二个元素是 `is_keyframe`，告诉调用方这帧是不是 IDR。
2306/// - `prepend_sps_pps` 为空时不插入（graceful fallback）。
2307/// - 输入字节被 defensive 解析：如果某个 NALU 长度字段错乱导致越界，跳出循环。
2308///
2309/// ## 安全
2310///
2311/// 纯函数，无 unsafe。
2312/// 旧版 `avcc_to_annex_b`，返回 `(Vec<u8>, bool)`。**热路径不要用**——
2313/// 热路径走 [`avcc_to_annex_b_into`] 写到调用方传入的 `BytesMut`，省一次 alloc。
2314///
2315/// 本函数只保留给单元测试（验证 AVCC→Annex-B 转换的正确性），用 `#[cfg(test)]` 框住，
2316/// 避免在 release / release-log 构建里产生 dead code warning。
2317#[cfg(test)]
2318pub fn avcc_to_annex_b(input: &[u8], prepend_sps_pps: &[u8]) -> (Vec<u8>, bool) {
2319    // [FIX-2026-06-24-v3] 自动检测输入格式：
2320    // - **Annex-B**（以 `00 00 00 01` 或 `00 00 01` 起始码开头）：NVENC / QuickSync / AMF 等
2321    //   硬件 MFT 直接给的就是 Annex-B，字节流已含 AUD + SPS + PPS + slice。
2322    //   这种情况下透传即可，**不要**按 AVCC 的 4 字节长度字段解析（会把起始码误读为 length=1，
2323    //   只读出 1 字节 NAL 就 break，实测产生 5 字节垃圾输出）。
2324    // - **AVCC**（4 字节 BE 长度前缀）：软编码器（`CLSID_MSH264EncoderMFT`）可能给这个。
2325    //   走老逻辑：长度 → `00 00 00 01` 起始码。
2326    //
2327    // H.264 emulation prevention 保证 `00 00 00 01` 在 NALU content 里不会出现（会被
2328    // 转义成 `00 00 03 00 00 01`），所以"开头 4 字节 = 00 00 00 01"是 Annex-B 的可靠判据。
2329    // 唯一歧义：AVCC 流里首个 NALU length=1（极少见），prefix 也是 `00 00 00 01`。
2330    // 这种情况按 Annex-B 透传的结果 = AVCC 转换的结果（都是 `00 00 00 01 XX`），无副作用。
2331    if input.len() >= 4 && &input[0..4] == &[0, 0, 0, 1] {
2332        return passthrough_annex_b(input, prepend_sps_pps);
2333    }
2334    if input.len() >= 3 && &input[0..3] == &[0, 0, 1] {
2335        return passthrough_annex_b(input, prepend_sps_pps);
2336    }
2337    avcc_bytes_to_annex_b(input, prepend_sps_pps)
2338}

2340/// Annex-B 字节流透传：按 NALU 起始码分段，仅在第一个 IDR slice 前注入 `prepend_sps_pps`
2341/// （且仅当流里没 SPS 时）。硬件 MFT 输出 SPS/PPS 通常在流里，prepend 是冗余兜底。
2342/// Annex-B 字节流透传：按 NALU 起始码分段，仅在第一个 IDR slice 前注入 `prepend_sps_pps`
2343/// （且仅当流里没 SPS 时）。硬件 MFT 输出 SPS/PPS 通常在流里，prepend 是冗余兜底。
2344///
2345/// `#[cfg(test)]`：只被 `avcc_to_annex_b` 调，本函数本身不直接进热路径。
2346#[cfg(test)]
2347fn passthrough_annex_b(annex_b: &[u8], prepend_sps_pps: &[u8]) -> (Vec<u8>, bool) {
2348    let starts = find_annex_b_nalu_starts(annex_b);
2349    if starts.is_empty() {
2350        // 没有起始码 = 不是合法 H.264 字节流，原样返回避免丢数据
2351        return (annex_b.to_vec(), false);
2352    }
2353
2354    let mut out = Vec::with_capacity(annex_b.len() + prepend_sps_pps.len());
2355    let mut saw_idr = false;
2356    let mut injected = false;
2357
2358    for (i, &pos) in starts.iter().enumerate() {
2359        let end = starts.get(i + 1).copied().unwrap_or(annex_b.len());
2360        // NALU header 在起始码之后
2361        let sc_len = if pos + 4 <= annex_b.len() && &annex_b[pos..pos + 4] == &[0, 0, 0, 1] {
2362            4
2363        } else {
2364            3
2365        };
2366        let nalu_start = pos + sc_len;
2367        let nal_type = if nalu_start < annex_b.len() {
2368            annex_b[nalu_start] & 0x1f
2369        } else {
2370            0xff
2371        };
2372
2373        // IDR (type 5) 前注入 SPS+PPS（仅一次；仅当流里没 SPS + prepend 非空）
2374        if nal_type == 5 && !injected {
2375            let has_sps_before = (0..i).any(|j| {
2376                let p = starts[j];
2377                let sl = if p + 4 <= annex_b.len() && &annex_b[p..p + 4] == &[0, 0, 0, 1] {
2378                    4
2379                } else {
2380                    3
2381                };
2382                let ns = p + sl;
2383                ns < annex_b.len() && annex_b[ns] & 0x1f == 7
2384            });
2385            if !has_sps_before && !prepend_sps_pps.is_empty() {
2386                out.extend_from_slice(prepend_sps_pps);
2387            }
2388            injected = true;
2389            saw_idr = true;
2390        }
2391
2392        out.extend_from_slice(&annex_b[pos..end]);
2393    }
2394
2395    (out, saw_idr)
2396}

2398/// 扫描 Annex-B 字节流里所有 NALU 起始码位置（`00 00 00 01` 4 字节 / `00 00 01` 3 字节）。
2399///
2400/// H.264 emulation prevention 保证 NALU content 内部不会出现裸 `00 00 01`（会被转义成
2401/// `00 00 03 01`），所以这里每个起始码都对应一个真实的 NALU 分界。
2402fn find_annex_b_nalu_starts(data: &[u8]) -> Vec<usize> {
2403    let mut out = Vec::new();
2404    let mut i = 0;
2405    while i < data.len() {
2406        if i + 4 <= data.len() && &data[i..i + 4] == &[0, 0, 0, 1] {
2407            out.push(i);
2408            i += 4;
2409        } else if i + 3 <= data.len() && &data[i..i + 3] == &[0, 0, 1] {
2410            out.push(i);
2411            i += 3;
2412        } else {
2413            i += 1;
2414        }
2415    }
2416    out
2417}

2462// [V6-R3] 零拷贝锚点：avcc_to_annex_b_into
2463// ====================================================================
2464//
2465// 热路径（capture 主循环每帧一次）需要的不是 `avcc_to_annex_b` 返回的 `Vec<u8>`
2466// —— 那个每次都做一次 heap allocation + N 次 NALU memcpy。

2470//   2. 调本函数追加 Annex-B 字节（只 memcpy，不 alloc）
2471//   3. 调 `buf.freeze()` 得到 `Bytes`（zero-cost 转 Arc 共享）

2475// - AVCC → Annex-B 转换要写新字节（4 字节起始码是新字节）→ 不可避免

2482/// - **不**在函数内部分配
2483/// - **不**清空 `buf`（调用方负责，如果要复用的话）
2484/// - 内部用 `extend_from_slice` 追加，一次 memcpy per slice 段
2485///
2486/// ## 调用约定

2488/// ```ignore

2494/// ```

2764// 内存显存释放
2765//
2766// Drop 是 Rust 的 RAII 析构。结构体里的每个 COM 智能指针字段在 Drop 时会自动 Release
2767//（windows crate 的 `IUnknown::drop` 会调 vtable 的 `Release`），所以本函数只需要做
2768// "业务层"的清理：MFT 流结束 / 解绑 D3D 设备 / 关闭 MF runtime。
2769//
2770// 字段会按声明顺序的逆序自动 drop（→ Release COM）。`MFShutdown` 是 body 的最后一
2771// 个语句，body 返回后才轮到 self 字段 drop —— 满足微软文档"MFShutdown 必须在所有
2772// MF 对象 Release 之后"的要求。
2773impl Drop for H264Encoder {
2774    fn drop(&mut self) {
2775        unsafe {
2776            // [release-log] drop trace
2777            #[cfg(any(debug_assertions, feature = "release-log"))]
2778            println!("[Drop] 正在清理 H264Encoder 资源...");
2779
2780            // 通知 MFT 结束流，并排空内部缓冲的剩余帧
2781            let _ = self.encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_END_STREAMING, 0);
2782            let _ = self.encoder.ProcessMessage(MFT_MESSAGE_COMMAND_DRAIN, 0);
2783
2784            // [V4-C3] 微软文档规定顺序：先 NOTIFY_RELEASE_RESOURCES 再 SET_D3D_MANAGER(0)
2785            // 否则严格 MFT（QuickSync）在解绑时仍持有 GPU 内部资源 → AV / D3D11 内部错。
2786            let _ = self.encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_RELEASE_RESOURCES, 0);
2787            let _ = self.encoder.ProcessMessage(MFT_MESSAGE_SET_D3D_MANAGER, 0);
2788
2789            // MFShutdown：必须在所有 MF/COM 句柄 Release 之后调用（微软文档要求）。
2790            // body 返回后，self 字段会按声明顺序的逆序自动 drop（COM Release）——满足要求。
2791            // MFStartup/MFShutdown 是 refcount 的，单实例下 1:1 配对最干净。
2792            let _ = MFShutdown();
2793        }
2794    }
