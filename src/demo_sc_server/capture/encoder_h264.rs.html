1//! H.264 视频编码器（基于 Windows Media Foundation）
2//!
3//! # 这个文件做什么
4//! 把 GPU 上的 BGRA 纹理（来自 DXGI 桌面复制）转成 H.264 码流。
5//! 优先用硬件编码器（NVENC / QuickSync / AMF 等），失败回退到微软自带的软编。
6//!
7//! # COM 概念速览（看这里能少踩 80% 的坑）
8//!
9//! Windows 的多媒体栈几乎全部用 COM（Component Object Model）写成。COM 对象的核心规则：
10//! - 每个接口都引用计数。`AddRef` +1，`Release` -1，归零时对象自我销毁。
11//! - Rust 这边 `windows` crate 用智能指针（如 `IMFTransform`、`IMFSample`）包装，
12//!   离开作用域时会自动调 `Release`，所以**正常情况下你不需要手动 Release**。
13//! - 所有跨线程/跨方法的 COM 对象都靠引用计数传递所有权。
14//!
15//! # Media Foundation 核心接口（这个文件会反复用到）
16//!
17//! | 接口                          | 作用                                                  |
18//! |-------------------------------|-------------------------------------------------------|
19//! | `IMFTransform` (MFT)          | 编/解码器本体，`ProcessInput` / `ProcessOutput` 是出入口 |
20//! | `IMFActivate`                 | MFT 的"工厂对象"，要先 `ActivateObject` 才能拿到 MFT    |
21//! | `IMFMediaType`                | 描述一种视频格式（分辨率、帧率、子类型 NV12/H264 等）  |
22//! | `IMFSample`                   | 一帧数据的容器，内部有 1 个或多个 `IMFMediaBuffer`       |
23//! | `IMFMediaBuffer`              | 一块连续的字节/显存                                     |
24//! | `IMFDXGIDeviceManager`        | 把 D3D11 设备"借"给 MFT，让它直接读写 GPU 显存          |
25//! | `IMFMediaEventGenerator`      | MFT 的事件来源，硬件 MFT 是异步的，要靠它推状态机        |
26//!
27//! # 数据流（一帧 H.264 输出的生命周期）
28//!
29//! ```text
30//! DXGI BGRA 纹理
31//!   └─ ID3D11Texture2D (bgra_texture)
32//!       └─ convert_bgra_to_nv12:   GPU VideoProcessor → NV12 纹理（零拷贝）
33//!       └─ build_sample:           把 NV12 纹理包成 IMFSample（硬件路径：包成 DXGI Surface Buffer）
34//!       └─ process_input:          IMFSample → MFT 输入
35//!       └─ process_output:         MFT 输出 → IMFSample（含一段 H.264 码流）
36//!       └─ 业务层读出 H.264 字节（这里只是 println，没真的发出去）
37//! ```

51/// [B3] H.264 编码 Profile。
52///
53/// 影响压缩率 / 兼容性 / 延迟：
54/// - Baseline: 无 B 帧，最广兼容（移动端 / 老硬件）
55/// - Main:     有 B 帧 + CABAC，中等兼容
56/// - High:     Main + 8x8 变换，最高压缩率，新硬件都支持
57///
58/// 注意：值对应 `eAVEncH264VProfile_*` 常量，直接传给 `MF_MT_MPEG2_PROFILE`。
59#[derive(Debug, Clone, Copy, PartialEq, Eq)]
60pub enum H264Profile {
61    Baseline,
62    Main,
63    High,
64}

71            H264Profile::Main => 77,       // eAVEncH264VProfile_Main

230impl H264Encoder {
231    /// 构造一个编码器：先试硬件，失败回退到软件。
232    ///
233    /// `bitrate_bps` 用 bps 单位（1 Mbps = 1_000_000），会被写到
234    /// 输出 MediaType 的 `MF_MT_AVG_BITRATE` 字段，并在 `renegotiate` 中复用。
235    pub fn new(_device: &ID3D11Device, width: u32, height: u32, fps: u32, bitrate_bps: u32, profile: H264Profile) -> Result<Self> {
236        unsafe {
237            // MFStartup: 初始化进程内的 Media Foundation 运行时。
238            // 整个进程只需要调一次（实际是 refcount 累加），但这里每次构造都调一下也行。
239            // MFSTARTUP_NOSOCKET 表示不启用 Media Foundation 网络功能，更轻。
240            // [release-log] MFStartup 启动 trace
241            #[cfg(any(debug_assertions, feature = "release-log"))]
242            println!("[MFT] 1. MFStartup...");
243            MFStartup(MF_VERSION, MFSTARTUP_NOSOCKET)?;
244
245            // 优先尝试硬件编码器
246            match Self::create_hw(_device, width, height, fps, bitrate_bps, profile) {
247                Ok(enc) => {
248                    // [release-log] 硬件编码器就绪摘要
249                    #[cfg(any(debug_assertions, feature = "release-log"))]
250                    {
251                        println!("✅ 硬件 H.264 编码器已就绪 (NVENC / QuickSync / AMF)");
252                        println!(
253                            "   └─ 配置: {width}x{height}@{fps}fps, 码率 {} bps ({} Mbps), GOP={} ({} 秒), 输出由 MFT 自提供={}, 推荐 buffer={} 字节",
254                            bitrate_bps, bitrate_bps / 1_000_000, fps * 2, 2, enc.output_provides_samples, enc.output_buffer_size
255                        );
256                        println!("   └─ NV12 纹理池: 16 张, 异步事件驱动 (event_gen 已挂上)");
257                    }
258                    return Ok(enc);
259                }
260                Err(e) => {
261                    // ⚠️ 这是 fallback warning，保留——生产要知道自己在用软编
262                    println!("⚠️ 硬件编码器不可用: {}，尝试回退到软件编码器...", e.message());
263                }
264            }
265
266            // 回退到软件编码器（微软自带，CPU 跑）
267            match Self::create_sw(_device, width, height, fps, bitrate_bps, profile) {
268                Ok(enc) => {
269                    // [release-log] 软件编码器就绪摘要
270                    #[cfg(any(debug_assertions, feature = "release-log"))]
271                    {
272                        println!("✅ 软件 H.264 编码器已就绪 (CLSID_MSH264EncoderMFT)");
273                        println!(
274                            "   └─ 配置: {width}x{height}@{fps}fps, 码率 {} bps ({} Mbps), GOP={} ({} 秒), 输出由 MFT 自提供={}, 推荐 buffer={} 字节",
275                            bitrate_bps, bitrate_bps / 1_000_000, fps * 2, 2, enc.output_provides_samples, enc.output_buffer_size
276                        );
277                        println!("   └─ CPU 色彩转换={}, 同步事件模式", enc.use_cpu_conversion);
278                    }
279                    return Ok(enc);
280                }
281                Err(e) => {
282                    // ❌ 这是 fatal error，保留
283                    println!("❌ 软件编码器也失败了: {}", e.message());
284                    return Err(e);
285                }
286            }
287        }
288    }

290    /// 分辨率变化时重协商 MediaType + 重建 NV12 纹理池。
291    ///
292    /// 触发场景：屏幕分辨率变了（用户改显示设置 / 切换屏幕），
293    /// 当前 MediaType 的 `MF_MT_FRAME_SIZE` 和新的帧尺寸对不上，
294    /// MFT 接下来会报 `MF_E_TRANSFORM_STREAM_CHANGE`。
295    ///
296    /// 调用方传入新尺寸（一般是 `frame.texture_desc()` 的 `Width`/`Height`），
297    /// 本方法会：
298    ///   1. 用新尺寸重新 SetInputType / SetOutputType（保留原 fps / bitrate / profile）
299    ///   2. 发 `NOTIFY_BEGIN_STREAMING` + `NOTIFY_START_OF_STREAM`，重置 MFT 内部流水线
300    ///   3. **关键**：drain 掉旧尺寸的 NV12 纹理池和 pending_textures，按新尺寸重建
301    ///   4. 软编路径同时重新分配 NV12 纹理 + CPU buffer
302    ///   5. 更新 `self.width` / `self.height`
303    pub fn renegotiate(&mut self, new_width: u32, new_height: u32) -> Result<()> {
304        if new_width == self.width && new_height == self.height {
305            return Ok(());
306        }
307        unsafe {
308            // [V4-C2] 微软文档规定：SetInputType/SetOutputType 之前必须先结束当前流 + flush。
309            //   否则 MFT 内部仍持有旧尺寸的 in-flight 帧引用：
310            //     - NVENC: SetInputType 后立即失效 → 后面帧全 garbage H.264 → 客户端解码崩
311            //     - AMF:   直接拒绝 SetOutputType，返回 MF_E_NOTACCEPTING
312            //     - QuickSync: 类似 NVENC
313            //   用 `let _ =` 容忍 MF_E_INVALID_STATE_TRANSITION（已不在 streaming）软错误。
314            let _ = self.encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_END_STREAMING, 0);
315            let _ = self.encoder.ProcessMessage(MFT_MESSAGE_COMMAND_FLUSH, 0);
316
317            // [V4-C2] flush 之后 MFT 不再引用 pending 里的纹理，把它们还回池子。
318            //   P1-1.1 修复时已经验证过这种 drain 模式是正确的（MFT 不再持有 → 安全还回）。
319            //   软编路径下 pending_textures 始终为空，drain 是 no-op。
320            for entry in self.pending_textures.drain(..) {
321                self.nv12_texture_pool.push_back(entry);
322            }
323
324            let frame_size = ((new_width as u64) << 32) | (new_height as u64);
325            let frame_rate = ((self.fps as u64) << 32) | 1;
326
327            // 1. 重新设置输出 MediaType
328            let output_type = MFCreateMediaType()?;
329            output_type.SetGUID(&MF_MT_MAJOR_TYPE, &MFMediaType_Video)?;
330            output_type.SetGUID(&MF_MT_SUBTYPE, &MFVideoFormat_H264)?;
331            output_type.SetUINT64(&MF_MT_FRAME_SIZE, frame_size)?;
332            output_type.SetUINT64(&MF_MT_FRAME_RATE, frame_rate)?;
333            output_type.SetUINT32(&MF_MT_AVG_BITRATE, self.bitrate_bps)?;
334            output_type.SetUINT32(&MF_MT_INTERLACE_MODE, 2)?;
335            // [B3] 用 self.profile 替代硬编码 100
336            output_type.SetUINT32(&MF_MT_MPEG2_PROFILE, self.profile.to_mf_value())?;
337            // [B6 Phase 1] 显式 GOP：每 2 秒一个 IDR（fps * 2 帧）
338            // 让客户端断线重连时最多等 2 秒就能看到画面，而不是默认的 ~8 秒
339            output_type.SetUINT32(&MF_MT_MAX_KEYFRAME_SPACING, self.fps * 2)?;
340            self.encoder.SetOutputType(0, &output_type, 0)?;
341
342            // [V6-R1] 重新提取 SPS+PPS（尺寸变了 SPS 必然变）。
343            //   必须在 SetOutputType 之后调，output_type 才有 SPS/PPS 配置记录。
344            //   MFT 输出 MediaType 理论上一定包含 SPS/PPS；提取失败是编码器内部状态异常，
345            //   直接 ? 传播，让 Session::recover() 接走重建。
346            self.sps_pps_annex_b = extract_sps_pps_annex_b(&output_type)?;
347
348            // 2. 重新设置输入 MediaType
349            let input_type = MFCreateMediaType()?;
350            input_type.SetGUID(&MF_MT_MAJOR_TYPE, &MFMediaType_Video)?;
351            input_type.SetGUID(&MF_MT_SUBTYPE, &MFVideoFormat_NV12)?;
352            input_type.SetUINT64(&MF_MT_FRAME_SIZE, frame_size)?;
353            input_type.SetUINT64(&MF_MT_FRAME_RATE, frame_rate)?;
354            input_type.SetUINT32(&MF_MT_INTERLACE_MODE, 2)?;
355            input_type.SetUINT32(&MF_MT_DEFAULT_STRIDE, new_width)?;
356            // 软编路径需要 MF_MT_SAMPLE_SIZE（NV12 一帧 = w*h*3/2 字节）
357            if !self.is_hardware {
358                input_type.SetUINT32(&MF_MT_SAMPLE_SIZE, new_width * new_height * 3 / 2)?;
359            }
360            self.encoder.SetInputType(0, &input_type, 0)?;
361
362            // 3. 通知 MFT 重启流 —— 关键，没有这两行 MFT 内部状态不会刷新
363            self.encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_BEGIN_STREAMING, 0)?;
364            self.encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_START_OF_STREAM, 0)?;
365
366            // 4. 重建 NV12 纹理池 —— 旧纹理全是旧尺寸，VideoProcessorBlt 会失败
367            //    直接 clear()，VecDeque 内 (Texture, Buffer) 离开作用域自动 Release
368            if self.is_hardware {
369                self.nv12_texture_pool.clear();
370                self.pending_textures.clear();
371                // current_hw_* 由 convert_bgra_to_nv12_hw 在下一帧重新设置
372                self.current_hw_texture = None;
373                self.current_hw_buffer = None;
374
375                // 池大小 5 与 create_hw 保持一致（见 [MEMORY_OPTIMIZATION_V2.md §4]）
376                const HW_TEXTURE_POOL_SIZE: usize = 5;
377                for _ in 0..HW_TEXTURE_POOL_SIZE {
378                    let tex = Self::create_nv12_texture(&self.device, new_width, new_height)?;
379                    let buf = MFCreateDXGISurfaceBuffer(&ID3D11Texture2D::IID, &tex, 0, false)?;
380                    self.nv12_texture_pool.push_back((tex, buf));
381                }
382            } else {
383                // 软编路径：单个 NV12 纹理 + CPU buffer 都按新尺寸重分配
384                self.nv12_texture = Some(Self::create_nv12_texture(&self.device, new_width, new_height)?);
385                let buffer_size = new_width * new_height * 3 / 2;
386                self.cpu_buffer = Some(MFCreateMemoryBuffer(buffer_size)?);
387            }
388
389            // 5. 旧尺寸下可能还缓存了输出 sample，引用的是 MFT 旧内部状态，直接丢掉
390            self.recycled_output_sample = None;
391
392            // 6. 更新记录的尺寸
393            self.width = new_width;
394            self.height = new_height;
395
396            Ok(())
397        }
398    }

400    /// 创建一个 NV12 格式的 D3D11 纹理。
401    /// 用作 VideoProcessor 的输出目标。
402    fn create_nv12_texture(_device: &ID3D11Device, width: u32, height: u32) -> Result<ID3D11Texture2D> {
403        unsafe {
404            let tex_desc = D3D11_TEXTURE2D_DESC {
405                Width: width,
406                Height: height,
407                MipLevels: 1,
408                ArraySize: 1,
409                // NV12 是 YUV 4:2:0 子采样，主流硬件编码器都吃这个格式。
410                Format: DXGI_FORMAT_NV12,
411                SampleDesc: DXGI_SAMPLE_DESC { Count: 1, Quality: 0 },
412                // DEFAULT = GPU 端最快访问，不要 STAGING（STAGING 才能 CPU 读）。
413                Usage: D3D11_USAGE_DEFAULT,
414                // 这个纹理要被 VideoProcessor 当 RT 写，又要被 DXGI Surface Buffer 当编码输入，
415                // 所以三种 bind flag 都要打。
416                BindFlags: (D3D11_BIND_RENDER_TARGET.0 | D3D11_BIND_VIDEO_ENCODER.0 | D3D11_BIND_SHADER_RESOURCE.0) as u32,
417                CPUAccessFlags: 0,
418                MiscFlags: 0,
419            };
420            let mut tex_opt: Option<ID3D11Texture2D> = None;
421            _device.CreateTexture2D(&tex_desc, None, Some(&mut tex_opt as *mut _))?;
422            tex_opt.ok_or_else(|| windows::core::Error::from(windows::Win32::Foundation::E_FAIL))
423        }
424    }

426    /// 查询 MFT 输出流的属性。
427    ///
428    /// 返回：
429    /// - `provides`: MFT 是否自己提供输出 sample
430    /// - `buffer_size`: 推荐输出 buffer 字节数
431    fn query_output_stream_info(encoder: &IMFTransform, width: u32, height: u32) -> (bool, u32) {
432        unsafe {
433            match encoder.GetOutputStreamInfo(0) {
434                Ok(stream_info) => {
435                    // dwFlags 是位掩码，看 MFT_OUTPUT_STREAM_PROVIDES_SAMPLES 这一位是否置 1
436                    let provides = (stream_info.dwFlags & MFT_OUTPUT_STREAM_PROVIDES_SAMPLES.0 as u32) != 0;
437                    // cbSize 是 MFT 推荐的输出 buffer 字节数
438                    // 如果它返回 0，就按 NV12 一帧的大小兜底（width * height * 3 / 2）
439                    let size = if stream_info.cbSize > 0 {
440                        stream_info.cbSize
441                    } else {
442                        width * height * 3 / 2
443                    };
444                    (provides, size)
445                }
446                Err(_) => (false, width * height * 3 / 2)
447            }
448        }
449    }

451    /// 拿一个输出 sample，准备喂给 `ProcessOutput`。
452    ///
453    /// - 如果 MFT 自己提供（`output_provides_samples=true`），返回 `None`，MFT 会自己 new。
454    /// - 否则优先从回收池里拿，复用前会清空 buffer 长度。
455    /// - 池为空时才 `MFCreateSample` 新建（避免每帧都走 COM 堆分配）。
456    pub fn create_output_sample(&mut self) -> Result<Option<IMFSample>> {
457        if self.output_provides_samples {
458            return Ok(None);
459        }
460        unsafe {
461            // 优先从池中复用
462            if let Some(sample) = self.recycled_output_sample.take() {
463                // 复用前清空 buffer 的当前长度
464                // （否则下次 Lock 出来数据会接着上次的尾巴）
465                // [H10 修复] SetCurrentLength(0) 失败时**不要**静默丢弃：
466                //   buffer 还残留上一帧的 cur_len → 编码器收到脏数据 → 输出混合了两帧的 H.264 流。
467                //   失败时走"新建 sample"路径：
468                //     1. 显式 drop 旧 sample（buffer + sample 一起 Release，避免引用泄漏）
469                //     2. 重新走"创建新 sample"逻辑
470                //   borrow-checker 友好做法：把"复用失败 → 新建"的逻辑抽到一个本地 closure
471                //   或者用 match 分支。
472                let reusable = if let Ok(buffer) = sample.GetBufferByIndex(0) {
473                    buffer.SetCurrentLength(0).is_ok()
474                } else {
475                    // 拿不到 buffer = sample 结构损坏，不能复用
476                    false
477                };
478                if reusable {
479                    return Ok(Some(sample));
480                }
481                // 复用失败：drop 旧 sample（buffer + sample 一起 Release）
482                drop(sample);
483                // fall through 到下面的"池为空时新建"分支
484            }
485
486            // 池为空 OR 复用失败 → 创建新 sample
487            let sample = MFCreateSample()?;
488            let buffer = MFCreateMemoryBuffer(self.output_buffer_size)?;
489            sample.AddBuffer(&buffer)?;
490            Ok(Some(sample))
491        }
492    }

495    pub fn recycle_output_sample(&mut self, sample: Option<IMFSample>) {
496        if let Some(s) = sample {
497            self.recycled_output_sample = Some(s);
498        }

501    /// 创建**硬件** H.264 编码器实例。
502    ///
503    /// 这是个非常典型的 Media Foundation 初始化流程：
504    ///   1. 创建并 Reset DXGIDeviceManager
505    ///   2. 用 `MFTEnumEx` 枚举出所有硬件 H.264 编码器
506    ///   3. 选第一个，`ActivateObject` 出真正的 `IMFTransform`
507    ///   4. 解锁异步（硬件 MFT 强制要求）
508    ///   5. 配置输入输出 MediaType 并 SetInputType/SetOutputType
509    ///   6. 发 ProcessMessage 开始流
510    fn create_hw(_device: &ID3D11Device, width: u32, height: u32, fps: u32, bitrate_bps: u32, profile: H264Profile) -> Result<Self> {
511        unsafe {
512            // 1. 创建 DXGIDeviceManager，让 MFT 可以直接读写我们的 GPU 显存
513            let mut dxgi_manager_opt: Option<IMFDXGIDeviceManager> = None;
514            let mut reset_token: u32 = 0;
515            MFCreateDXGIDeviceManager(&mut reset_token, &mut dxgi_manager_opt)?;
516            let dxgi_manager = dxgi_manager_opt.ok_or_else(|| windows::core::Error::from(windows::Win32::Foundation::E_FAIL))?;
517
518            // 把我们的 D3D11 设备塞给 manager。`IUnknown::cast` 是 `windows` crate 提供的
519            // 安全转型，内部走 QueryInterface。
520            let unk: IUnknown = _device.cast()?;
521            dxgi_manager.ResetDevice(&unk, reset_token)?;
522
523            // 2. 告诉系统我们要找：输入是 NV12、输出是 H.264 的视频编码器
524            let input_info = MFT_REGISTER_TYPE_INFO { guidMajorType: MFMediaType_Video, guidSubtype: MFVideoFormat_NV12 };
525            let output_info = MFT_REGISTER_TYPE_INFO { guidMajorType: MFMediaType_Video, guidSubtype: MFVideoFormat_H264 };
526
527            // 3. MFTEnumEx 枚举所有匹配的编码器（激活器形式）
528            //    返回的是一个 IMFActivate* 数组，由系统分配在 COM 任务堆上。
529            // [1.11 修复] 顺序：
530            //   1. MFTEnumEx 拿原始数组
531            //   2. 先 clone 所有到 Vec<IMFActivate>（每元素 AddRef，独立的 Rust 所有权）
532            //   3. 立刻 CoTaskMemFree 原始数组（不再 drop_in_place —— 数组内存是 COM 任务堆，
533            //      我们已经把所有权转到 Vec 里了，原数组不需要逐个 Release）
534            //   4. Vec 走正常 Rust Drop，未选中的激活器自动 Release
535            //   原实现的 drop_in_place 顺序和 clone 顺序耦合：
536            //   一旦 clone 之后、drop_in_place 之前发生 panic / 早返回，原数组内存就漏了。
537            let mut pp_activators: *mut Option<IMFActivate> = std::ptr::null_mut();
538            let mut count: u32 = 0;
539            MFTEnumEx(
540                MFT_CATEGORY_VIDEO_ENCODER,
541                MFT_ENUM_FLAG_HARDWARE | MFT_ENUM_FLAG_SORTANDFILTER,
542                Some(&input_info),
543                Some(&output_info),
544                &mut pp_activators,
545                &mut count,
546            )?;
547
548            if count == 0 {
549                if !pp_activators.is_null() {
550                    CoTaskMemFree(Some(pp_activators as *const core::ffi::c_void));
551                }
552                return Err(windows::core::Error::from(windows::Win32::Foundation::E_FAIL));
553            }
554
555            // 把所有激活器 clone 到 Vec（每个都 AddRef，独立的 Rust 所有权）
556            let raw_slice = std::slice::from_raw_parts(pp_activators, count as usize);
557            let activators: Vec<IMFActivate> = raw_slice
558                .iter()
559                .filter_map(|opt| opt.as_ref().cloned())
560                .collect();
561
562            // 立刻释放 CoTaskMemAlloc 的原始数组（每个元素已经被 clone 走了，不需要 drop_in_place）
563            CoTaskMemFree(Some(pp_activators as *const core::ffi::c_void));
564
565            // 4. 尝试按顺序激活，挑第一个成功的
566            //   软硬件混合环境里 MFTEnumEx 可能返回多个候选（NVENC / AMF / 微软软编等），
567            //   第一个未必能 ActivateObject（驱动问题），要 fall back。
568            let mut encoder_opt: Option<IMFTransform> = None;
569            #[cfg(any(debug_assertions, feature = "release-log"))]
570            let mut attempt_count: u32 = 0;
571            for act in &activators {
572                #[cfg(any(debug_assertions, feature = "release-log"))]
573                {
574                    attempt_count += 1;
575                }
576                match act.ActivateObject::<IMFTransform>() {
577                    Ok(t) => {
578                        encoder_opt = Some(t);
579                        #[cfg(any(debug_assertions, feature = "release-log"))]
580                        if attempt_count > 1 {
581                            println!("⚠️ [MFT] 第 {} 个候选硬件编码器 fallback (前 {} 个 ActivateObject 失败)",
582                                attempt_count, attempt_count - 1);
583                        }
584                        break;
585                    }
586                    Err(_e) => {
587                        #[cfg(any(debug_assertions, feature = "release-log"))]
588                        eprintln!("⚠️ [MFT] ActivateObject 失败 (attempt={}, code=0x{:08X}), 试下一个",
589                            attempt_count, _e.code().0 as u32);
590                        // 继续试下一个
591                    }
592                }
593            }
594            let encoder: IMFTransform = encoder_opt
595                .ok_or_else(|| windows::core::Error::from(windows::Win32::Foundation::E_FAIL))?;
596            // activators Vec 离开作用域，未选中的全部自动 Release
597
598            // 5. 硬件 MFT 默认是异步 + 锁定的，必须显式解锁才能用
599            //   MF_TRANSFORM_ASYNC_UNLOCK = 1 表示"我懂异步，我会自己驱动状态机"
600            let attributes = encoder.GetAttributes()?;
601            attributes.SetUINT32(&MF_TRANSFORM_ASYNC_UNLOCK, 1)?;
602
603            // 拿到事件生成器，后面 process_input/output 用它等 MFT 事件
604            let event_gen: IMFMediaEventGenerator = encoder.cast()?;
605
606            // 把 D3D Manager 告诉 MFT（之后 MFT 就能直接读写我们的 GPU 纹理了）
607            encoder.ProcessMessage(MFT_MESSAGE_SET_D3D_MANAGER, dxgi_manager.as_raw() as usize)?;
608
609
610            // 7. 准备输出 MediaType：告诉 MFT 我们要 H.264、分辨率、帧率、码率
611            //    MediaType 用 packed 64 位整数表示 (高 32 位是 width，低 32 位是 height)
612            let frame_size = ((width as u64) << 32) | (height as u64);
613            let frame_rate = ((fps as u64) << 32) | 1;
614
615            let output_type = MFCreateMediaType()?;
616            output_type.SetGUID(&MF_MT_MAJOR_TYPE, &MFMediaType_Video)?;
617            output_type.SetGUID(&MF_MT_SUBTYPE, &MFVideoFormat_H264)?;
618            output_type.SetUINT64(&MF_MT_FRAME_SIZE, frame_size)?;
619            output_type.SetUINT64(&MF_MT_FRAME_RATE, frame_rate)?;
620            output_type.SetUINT32(&MF_MT_AVG_BITRATE, bitrate_bps)?;
621            // 2 = MFVideoInterlace_Progressive，逐行扫描
622            output_type.SetUINT32(&MF_MT_INTERLACE_MODE, 2)?;
623            // [B3] 用传入的 profile 参数替代硬编码 100
624            output_type.SetUINT32(&MF_MT_MPEG2_PROFILE, profile.to_mf_value())?;
625            // [B6 Phase 1] 显式 GOP：每 2 秒一个 IDR（fps * 2 帧）
626            output_type.SetUINT32(&MF_MT_MAX_KEYFRAME_SPACING, fps * 2)?;
627            encoder.SetOutputType(0, &output_type, 0)?;
628
629            // 8. 准备输入 MediaType：告诉 MFT 我们要喂 NV12
630            let input_type = MFCreateMediaType()?;
631            input_type.SetGUID(&MF_MT_MAJOR_TYPE, &MFMediaType_Video)?;
632            input_type.SetGUID(&MF_MT_SUBTYPE, &MFVideoFormat_NV12)?;
633            input_type.SetUINT64(&MF_MT_FRAME_SIZE, frame_size)?;
634            input_type.SetUINT64(&MF_MT_FRAME_RATE, frame_rate)?;
635            input_type.SetUINT32(&MF_MT_INTERLACE_MODE, 2)?;
636            input_type.SetUINT32(&MF_MT_DEFAULT_STRIDE, width)?;
637            encoder.SetInputType(0, &input_type, 0)?;
638
639            // 9. 告诉 MFT 流要开始了
640            encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_BEGIN_STREAMING, 0)?;
641            encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_START_OF_STREAM, 0)?;
642
643            let (output_provides_samples, output_buffer_size) =
644                Self::query_output_stream_info(&encoder, width, height);
645
646            // 10. 创建纹理池
647            //     每张 NV12 纹理 + 对应的 DXGI Surface Buffer（包成 IMFMediaBuffer）
648            //     DXGI Surface Buffer 是一种特殊的 IMFMediaBuffer，
649            //     它内部不存 CPU 字节，而是指向一张 D3D 纹理 —— 给 MFT 直接读写 GPU 显存用。
650            //
651            // [MEMORY_OPTIMIZATION_V2.md §4] 池大小 16 → 5：
652            //   - NVENC/QuickSync/AMF 都是 1:1 输入输出，最多 1~2 帧 in-flight
653            //   - 5 张 = 1 张 VP 正在写 + 2~3 张 MFT in-flight + 1~2 张 buffer
654            //   - iGPU 上 5 张 × 3 MB = 15 MB（vs 16 张 × 3 MB = 48 MB），省 33 MB
655            //   - dGPU 上 NV12 纹理在 VRAM，不计入进程工作集
656            //   - 想确认实际水位：开 release-log 跑 10 分钟，看 [pool] 日志
657            //     （convert_bgra_to_nv12_hw 里加了 release-log hook，详见该函数）
658            // // [原 16 张的池子注释保留] 回归时把下面 4 行注释取消、上面 4 行注释即可
659            // // let mut nv12_texture_pool = VecDeque::new();
660            // // for _ in 0..16 {
661            // //     let tex = Self::create_nv12_texture(_device, width, height)?;
662            // //     let buf = MFCreateDXGISurfaceBuffer(&ID3D11Texture2D::IID, &tex, 0, false)?;
663            // //     nv12_texture_pool.push_back((tex, buf));
664            // // }
665            const HW_TEXTURE_POOL_SIZE: usize = 5;
666            let mut nv12_texture_pool = VecDeque::new();
667            for _ in 0..HW_TEXTURE_POOL_SIZE {
668                let tex = Self::create_nv12_texture(_device, width, height)?;
669                let buf = MFCreateDXGISurfaceBuffer(&ID3D11Texture2D::IID, &tex, 0, false)?;
670                nv12_texture_pool.push_back((tex, buf));
671            }
672
673            let context = _device.GetImmediateContext()?;
674
675            // [修复 §3.4] 缓存 video_device / video_context，避开后续每帧的 QueryInterface。
676            // 硬编必然支持 VideoProcessor（GPU 上才有 VideoProcessor 概念），
677            // cast 不可能失败，这里直接 unwrap。
678            let video_device: ID3D11VideoDevice = _device.cast()?;
679            let video_context: ID3D11VideoContext = context.cast()?;
680
681            Ok(Self {
682                encoder,
683                event_gen: Some(event_gen),
684                pending_need_input: false,
685                device: _device.clone(), context,
686                video_device: Some(video_device),
687                video_context: Some(video_context),
688                nv12_texture: None,
689                nv12_texture_pool,
690                pending_textures: VecDeque::new(),
691                current_hw_texture: None,
692                current_hw_buffer: None,
693                // dxgi_manager: 已传给 MFT（ResetDevice + ProcessMessage SET_D3D_MANAGER），
694                //   MFT 内部会 AddRef，本函数返回时让它自动 Drop 即可。
695                staging_texture: None, cpu_buffer: None,
696                bgra_staging_texture: None,
697                use_cpu_conversion: false,
698                is_hardware: true, width, height,
699                fps, bitrate_bps,
700                recycled_output_sample: None,
701                recycled_input_sample: None,    // 硬编不用输入池
702                // 从 output MediaType 提取 SPS+PPS，转 Annex-B。
703                // 必须在 encoder SetOutputType 之后调（output_type 才有 SPS/PPS）。
704                sps_pps_annex_b: extract_sps_pps_annex_b(&output_type)?,
705                #[cfg(any(debug_assertions, feature = "release-log"))]
706                logged_first_input: false,
707                #[cfg(any(debug_assertions, feature = "release-log"))]
708                logged_first_output: false,
709                output_provides_samples,
710                output_buffer_size,
711                pending_force_idr: false,  // [B6 Phase 2]
712                profile,  // [B3]
713            })
714        }
715    }

717    /// 创建**软件** H.264 编码器实例（微软自带）。
718    ///
719    /// 软件 MFT 是同步的：调 `ProcessInput` 立刻返回，可以直接调 `ProcessOutput`。
720    /// 不需要事件驱动，也不需要 DXGIDeviceManager。
721    fn create_sw(_device: &ID3D11Device, width: u32, height: u32, fps: u32, bitrate_bps: u32, profile: H264Profile) -> Result<Self> {
722        unsafe {
723            // CoCreateInstance: 直接按 CLSID 创建一个 in-process COM 对象
724            // CLSID_MSH264EncoderMFT = 微软自带的 H.264 编码器

730            output_type.SetGUID(&MF_MT_MAJOR_TYPE, &MFMediaType_Video)?;

732            output_type.SetUINT64(&MF_MT_FRAME_SIZE, frame_size)?;
733            output_type.SetUINT64(&MF_MT_FRAME_RATE, frame_rate)?;
734            output_type.SetUINT32(&MF_MT_AVG_BITRATE, bitrate_bps)?;
735            output_type.SetUINT32(&MF_MT_INTERLACE_MODE, 2)?;
736            // [B3] 用传入的 profile 参数替代硬编码 100
737            output_type.SetUINT32(&MF_MT_MPEG2_PROFILE, profile.to_mf_value())?;
738            // [B6 Phase 1] 显式 GOP：每 2 秒一个 IDR（fps * 2 帧）
739            output_type.SetUINT32(&MF_MT_MAX_KEYFRAME_SPACING, fps * 2)?;
740            encoder.SetOutputType(0, &output_type, 0)?;
741
742            let input_type = MFCreateMediaType()?;
743            input_type.SetGUID(&MF_MT_MAJOR_TYPE, &MFMediaType_Video)?;
744            input_type.SetGUID(&MF_MT_SUBTYPE, &MFVideoFormat_NV12)?;
745            input_type.SetUINT64(&MF_MT_FRAME_SIZE, frame_size)?;
746            input_type.SetUINT64(&MF_MT_FRAME_RATE, frame_rate)?;
747            input_type.SetUINT32(&MF_MT_INTERLACE_MODE, 2)?;
748            input_type.SetUINT32(&MF_MT_DEFAULT_STRIDE, width)?;
749            // MF_MT_SAMPLE_SIZE = 一帧 NV12 的总字节数（Y + UV）
750            input_type.SetUINT32(&MF_MT_SAMPLE_SIZE, width * height * 3 / 2)?;
751            encoder.SetInputType(0, &input_type, 0)?;
752
753            encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_BEGIN_STREAMING, 0)?;
754            encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_START_OF_STREAM, 0)?;
755
756            let (output_provides_samples, output_buffer_size) =
757                Self::query_output_stream_info(&encoder, width, height);
758
759            let nv12_texture = Self::create_nv12_texture(_device, width, height)?;
760
761
762            // staging 纹理：GPU 可写 → CPU 可读，专门用来给软件编码器喂数据
763            let staging_desc = D3D11_TEXTURE2D_DESC {
764                Width: width, Height: height, MipLevels: 1, ArraySize: 1,
765                Format: DXGI_FORMAT_NV12,
766                SampleDesc: DXGI_SAMPLE_DESC { Count: 1, Quality: 0 },

826    // ========== 色彩转换：统一入口 ==========
827    /// BGRA → NV12 的总入口。会根据当前模式选硬件或 CPU 路径。
828    pub fn convert_bgra_to_nv12(&mut self, bgra_texture: &ID3D11Texture2D) -> windows::core::Result<()> {
829        if self.use_cpu_conversion {
830            return self.convert_bgra_to_nv12_cpu(bgra_texture);
831        }
832
833        match self.convert_bgra_to_nv12_hw(bgra_texture) {
834            Ok(()) => Ok(()),
835            Err(e) => {
836                // 硬件色彩转换失败，回退到 CPU（仅软件编码器模式）
837                if !self.is_hardware {
838                    // [H2 修复] 加 cfg gate（错误路径但保持 release 0 开销）
839                    #[cfg(any(debug_assertions, feature = "release-log"))]
840                    println!("⚠️ 硬件色彩转换失败: {}，回退到 CPU 色彩转换...", e.message());
841                    self.use_cpu_conversion = true;
842                    return self.convert_bgra_to_nv12_cpu(bgra_texture);
843                }
844                Err(e)
845            }
846        }

849    // ========== 硬件色彩转换（GPU Video Processor）==========
850    /// 用 D3D11 VideoProcessor 把 BGRA 纹理转成 NV12 纹理，全程 GPU 操作，零 CPU 拷贝。
851    ///
852    /// VideoProcessor 本来是用来做颜色空间转换 / 缩放 / deinterlace 的，
853    /// 这里我们只用到它的"BGRA → NV12 颜色空间转换"能力。
854    fn convert_bgra_to_nv12_hw(&mut self, bgra_texture: &ID3D11Texture2D) -> windows::core::Result<()> {
855        unsafe {
856            // 根据模式确定目标 NV12 纹理
857            // 硬件模式：从池里取一张可用的（轮换使用，避免被编码器 in-flight 引用时被覆盖）
858            //
859            // [P0-1 修复] pop 出来后立刻存进 current_hw_*；中间任何 `?` 早返回都必须手动
860            //   回收回池子。否则 GPU 抖动 → CreateVideoProcessor*View 失败 → 池子单调缩水 → 池空崩。
861            let nv12_texture = if self.is_hardware {
862                let (tex, buf) = match self.nv12_texture_pool.pop_front() {
863                    Some(x) => x,
864                    None => {
865                        // 池空 = 编码器流水线深度超过了我们预分配的张数
866                        // [MEMORY_OPTIMIZATION_V2.md §4] 池子从 16 缩到 5 后，
867                        //   如果走这条分支说明 NVENC 实际流水线比 5 深，需要扩大池子。
868                        //   开 release-log 跑 10 分钟，参考 [pool] 水位日志调整。
869                        // [H2 修复] 加 cfg gate（池空 = 异常路径，但要保持 release 0 开销）
870                        #[cfg(any(debug_assertions, feature = "release-log"))]
871                        eprintln!("⚠️ 纹理池耗尽，等待编码器输出...");
872                        return Err(windows::core::Error::from(windows::Win32::Foundation::E_OUTOFMEMORY));
873                    }
874                };
875                // [release-log] 池水位监控：每秒最多打一次（防止 30/60 fps 刷屏）
876                //   跑完后看输出，能直接看出"实际最高水位是多少"，便于调 HW_TEXTURE_POOL_SIZE
877                #[cfg(any(debug_assertions, feature = "release-log"))]
878                {
879                    use std::sync::Mutex;
880                    use std::time::Instant;
881                    static LAST_LOG: Mutex<Option<Instant>> = Mutex::new(None);
882                    let mut last = LAST_LOG.lock().unwrap_or_else(|p| p.into_inner());
883                    let should_log = match *last {
884                        None => true,
885                        Some(t) => t.elapsed().as_secs() >= 1,
886                    };
887                    if should_log {
888                        eprintln!("[pool] after pop: pool={}, pending={}, current={}",
889                            self.nv12_texture_pool.len(),
890                            self.pending_textures.len(),
891                            1);
892                        *last = Some(Instant::now());
893                    }
894                }
895                self.current_hw_texture = Some(tex.clone());
896                self.current_hw_buffer = Some(buf);
897                tex
898            } else {
899                // 软件模式直接使用预先创建好的单一纹理
900                // [M1 修复] 显式清空 current_hw_*：防止任何路径意外污染。
901                //   虽然 build_sample 已经在 is_hardware 分支里 take（line 1167-1168），
902                //   但保险起见软编路径不持有这些字段，让 invariant 永远成立：
903                //   "is_hardware=true ⇔ current_hw_* is Some"
904                self.current_hw_texture = None;
905                self.current_hw_buffer = None;
906                // [1.4 修复] 用 expect 替代 unwrap：nv12_texture 在 create_sw 末尾被赋值为 Some，
907                // [1.4 修复] 用 expect 替代 unwrap：nv12_texture 在 create_sw 末尾被赋值为 Some，
908                //   永不后退。若发现 panic 说明 create_sw 漏写了初始化赋值，需要修。
909                self.nv12_texture
910                    .as_ref()
911                    .expect("convert_bgra_to_nv12_hw: 软件模式 nv12_texture 未初始化 (create_sw bug)")
912                    .clone()
913            };
914
915            // 把 D3D11 设备/上下文升级成视频专用接口
916            // [修复 §3.4] 用 new() 里缓存的字段，不再每帧 cast()：
917            //   此函数进到这里的前提是 use_cpu_conversion=false，
918            //   而 use_cpu_conversion=false ⇔ video_device cast 成功，
919            //   所以这里 unwrap 安全。
920            // [1.4 修复] 用 expect 替代 unwrap：video_device / video_context 在 create_hw
921            //   末尾被赋值为 Some(use_cpu_conversion=false 时)。
922            let video_device = self
923                .video_device
924                .as_ref()
925                .expect("convert_bgra_to_nv12_hw: 硬件模式 video_device 未缓存 (create_hw bug)");
926            let video_context = self
927                .video_context
928                .as_ref()
929                .expect("convert_bgra_to_nv12_hw: 硬件模式 video_context 未缓存 (create_hw bug)");
930
931            // VideoProcessor 创建很贵（几百微秒），用一个 thread_local 缓存住
932            // 注意：这个缓存只在当前线程有效，且永远不会失效（GPU 资源随 D3D device 销毁）
933            thread_local! {
934                static VP_CACHE: RefCell<Option<(ID3D11VideoProcessor, ID3D11VideoProcessorEnumerator)>> = RefCell::new(None);
935            }
936            let vp_setup_result = VP_CACHE.with(|cache| -> windows::core::Result<(ID3D11VideoProcessor, ID3D11VideoProcessorEnumerator)> {
937                let mut cache = cache.borrow_mut();
938                if let Some(ref pair) = *cache { return Ok(pair.clone()); }
939                // ContentDesc 描述这次转换的输入输出格式
940                let content_desc = D3D11_VIDEO_PROCESSOR_CONTENT_DESC {
941                    InputFrameFormat: D3D11_VIDEO_FRAME_FORMAT_PROGRESSIVE,
942                    InputWidth: self.width, InputHeight: self.height,
943                    OutputWidth: self.width, OutputHeight: self.height,
944                    // OPTIMAL_SPEED: 选最快的转换路径，质量略低但 CPU/GPU 都更轻
945                    Usage: D3D11_VIDEO_USAGE_OPTIMAL_SPEED,
946                    ..Default::default()
947                };
948                let enumerator = video_device.CreateVideoProcessorEnumerator(&content_desc)?;
949                let processor = video_device.CreateVideoProcessor(&enumerator, 0)?;
950
951                // [V4-C1] VideoProcessor 状态机守则：
952                //   D3D 文档明确要求显式设色彩空间 / 矩形，否则各 GPU 厂商默认不同
953                //   （NVIDIA 偏 full-range RGB / AMD 偏 limited-range BT.601 / Intel iGPU 经常错）。
954                //   B8G8R8A8_UNORM 是 full-range RGB，NV12 是 limited-range YUV，
955                //   没正确转换 ≈ 整张图偏亮/偏暗 ~10%。
956                //   处理器状态会跨 Blt 持久化，cache 创建时设一次就够（hit 分支不动）。
957                //
958                // windows 0.62 把 D3D11_VIDEO_PROCESSOR_COLOR_SPACE 暴露成 _bitfield: u32。
959                // C 头文件里是 7 位位字段：
960                //   bit 0: Usage               (0=Video, 1=Playback)
961                //   bit 1: RGB_Range           (0=limited, 1=full)
962                //   bit 2: YCbCr_Matrix        (0=BT.709, 1=BT.601, 2=BT.2020)
963                //   bit 3: YCbCr_TransferFunction (0=Linear, 1=BT.709/601 1.x)
964                //   bit 4: YCbCr_TransferMatrix   (0=BT.601, 1=BT.709)
965                //   bit 5-6: Nominal_Range     (0=full, 1=limited, 2=studio)
966                // Set* 方法返回 ()，不返回 Result。
967
968                // 输出色彩空间：NV12 = BT.601 limited range (Y: 16-235, UV: 16-240)
969                // bits = 0 | 0<<1 | 1<<2 | 1<<3 | 1<<4 | 1<<5 = 4+8+16+32 = 60
970                let output_cspace = D3D11_VIDEO_PROCESSOR_COLOR_SPACE { _bitfield: 60 };
971                video_context.VideoProcessorSetOutputColorSpace(&processor, &output_cspace);
972
973                // 输入色彩空间：BGRA = full-range RGB (0-255)
974                // bits = 0 | 1<<1 | 1<<2 | 1<<3 | 1<<4 | 0<<5 = 2+4+8+16 = 30
975                let input_cspace = D3D11_VIDEO_PROCESSOR_COLOR_SPACE { _bitfield: 30 };
976                video_context.VideoProcessorSetStreamColorSpace(&processor, 0, &input_cspace);
977
978                // 输出 target rect
979                let out_rect = RECT {
980                    left: 0, top: 0,
981                    right: self.width as i32, bottom: self.height as i32,
982                };
983                video_context.VideoProcessorSetOutputTargetRect(&processor, true, Some(&out_rect));
984
985                // [V4-H1] SetStreamSourceRect / SetStreamDestRect：
986                //   D3D11 文档规定"必须先调"这两个函数，否则严格驱动（Intel / 部分 AMD）
987                //   会报 E_INVALIDARG → VideoProcessorBlt 失败 → 走软件 fallback。
988                video_context.VideoProcessorSetStreamSourceRect(&processor, 0, true, Some(&out_rect));
989                video_context.VideoProcessorSetStreamDestRect(&processor, 0, true, Some(&out_rect));
990
991                let pair = (processor, enumerator);
992                *cache = Some(pair.clone());
993                Ok(pair)
994            });
995            let (processor, enumerator) = match vp_setup_result {
996                Ok(pair) => pair,
997                Err(e) => {
998                    // [P0-1 修复] CreateVideoProcessorEnumerator/Processor 失败 → 还回池子
999                    if self.is_hardware {
1000                        if let (Some(tex), Some(buf)) = (
1001                            self.current_hw_texture.take(),
1002                            self.current_hw_buffer.take(),
1003                        ) {
1004                            self.nv12_texture_pool.push_back((tex, buf));
1005                        }
1006                    }
1007                    return Err(e);
1008                }
1009            };
1010
1011            // VideoProcessor 不直接吃纹理，要先包成"视图"才能用
1012            // InputView: 把 BGRA 纹理包成 VP 能理解的输入视图
1013            // [P1-1 修复] 改 ok_or_else：某些 AMD 旧驱动 / 虚拟显示驱动 / WSL2 GPU 透传会
1014            //   返回 S_OK 但 ppView=NULL。expect 在 release profile panic=abort 下会整进程 crash。
1015            //   V2 H3 漏修了这里（H3 只修了 create_bgra_texture / create_nv12_texture 两个 case）。
1016            let input_view_result: windows::core::Result<ID3D11VideoProcessorInputView> = (|| {
1017                let desc = D3D11_VIDEO_PROCESSOR_INPUT_VIEW_DESC {
1018                    FourCC: 0, ViewDimension: D3D11_VPIV_DIMENSION_TEXTURE2D,
1019                    Anonymous: D3D11_VIDEO_PROCESSOR_INPUT_VIEW_DESC_0 {
1020                        Texture2D: D3D11_TEX2D_VPIV { MipSlice: 0, ArraySlice: 0 },
1021                    },
1022                };
1023                let mut view_opt: Option<ID3D11VideoProcessorInputView> = None;
1024                video_device.CreateVideoProcessorInputView(
1025                    bgra_texture, &enumerator, &desc,
1026                    Some(&mut view_opt as *mut _ as *mut _),
1027                )?;
1028                view_opt.ok_or_else(|| windows::core::Error::from(windows::Win32::Foundation::E_FAIL))
1029            })();
1030            let input_view = match input_view_result {
1031                Ok(v) => v,
1032                Err(e) => {
1033                    // [P0-1 修复] 任意 CreateVideoProcessor*View 失败 → 还回池子
1034                    if self.is_hardware {
1035                        if let (Some(tex), Some(buf)) = (
1036                            self.current_hw_texture.take(),
1037                            self.current_hw_buffer.take(),
1038                        ) {
1039                            self.nv12_texture_pool.push_back((tex, buf));
1040                        }
1041                    }
1042                    return Err(e);
1043                }
1044            };
1045
1046            // OutputView: 把 NV12 纹理包成 VP 能写出的目标视图
1047            // 注意 CreateVideoProcessorOutputView 是 VideoDevice 的方法（不是 VideoContext）
1048            // [P1-1 修复] 同 InputView：S_OK + ppView=NULL 也要走错误路径
1049            let output_view_result: windows::core::Result<ID3D11VideoProcessorOutputView> = (|| {
1050                let desc = D3D11_VIDEO_PROCESSOR_OUTPUT_VIEW_DESC {
1051                    ViewDimension: D3D11_VPOV_DIMENSION_TEXTURE2D,
1052                    Anonymous: D3D11_VIDEO_PROCESSOR_OUTPUT_VIEW_DESC_0 {
1053                        Texture2D: D3D11_TEX2D_VPOV { MipSlice: 0 },
1054                    },
1055                };
1056                let mut view_opt: Option<ID3D11VideoProcessorOutputView> = None;
1057                video_device.CreateVideoProcessorOutputView(
1058                    &nv12_texture, &enumerator, &desc,
1059                    Some(&mut view_opt as *mut _ as *mut _),
1060                )?;
1061                view_opt.ok_or_else(|| windows::core::Error::from(windows::Win32::Foundation::E_FAIL))
1062            })();
1063            let output_view = match output_view_result {
1064                Ok(v) => v,
1065                Err(e) => {
1066                    // [P0-1 修复] 失败 → 还回池子 + 让 input_view Drop 释放
1067                    drop(input_view);
1068                    if self.is_hardware {
1069                        if let (Some(tex), Some(buf)) = (
1070                            self.current_hw_texture.take(),
1071                            self.current_hw_buffer.take(),
1072                        ) {
1073                            self.nv12_texture_pool.push_back((tex, buf));
1074                        }
1075                    }
1076                    return Err(e);
1077                }
1078            };
1079
1080            // Stream 描述这一次 Blt 用哪些输入/输出视图
1081            // pInputSurface 用 ManuallyDrop 包起来，因为 COM 接口字段是 Option，
1082            // 普通包法在 MFT_OUTPUT_DATA_BUFFER 拷贝时会被多次 Drop，触发多次 Release
1083            let mut stream_data = D3D11_VIDEO_PROCESSOR_STREAM {
1084                Enable: BOOL(1), OutputIndex: 0, InputFrameOrField: 0,
1085                PastFrames: 0, FutureFrames: 0,
1086                ppPastSurfaces: std::ptr::null_mut(),
1087                pInputSurface: ManuallyDrop::new(Some(input_view)),
1088                ppFutureSurfaces: std::ptr::null_mut(),
1089                ..Default::default()
1090            };
1091            // VideoProcessorBlt: 真正执行转换的命令，提交到 GPU 队列
1092            // [修复 §2.1] 不用 ? 直接传播，改为先取结果。
1093            //   错误路径必须把 current_hw_texture / current_hw_buffer 还回 nv12_texture_pool，
1094            //   否则 GPU 出错几次后池子就被吃光，进程崩（详见 GPU_LEAK_REVIEW.md §2.1）。
1095            let blt_result = video_context.VideoProcessorBlt(
1096                &processor,
1097                &output_view,
1098                0,
1099                std::slice::from_ref(&stream_data),
1100            );
1101            // 关键：离开作用域前手动把 ManuallyDrop 里的值 take 出来，
1102            // 那个 Some(input_view) 会跟着本行的 drop 正常 Release。
1103            // 这一步无论 Blt 成功还是失败都要做——失败时 input_view 也得 Release。
1104            let _ = ManuallyDrop::take(&mut stream_data.pInputSurface);
1105
1106            if let Err(e) = blt_result {
1107                // [修复 §2.1] 错误路径：硬件模式把 pop 出来的 (tex, buf) 还回池子
1108                if self.is_hardware {
1109                    if let (Some(tex), Some(buf)) = (
1110                        self.current_hw_texture.take(),
1111                        self.current_hw_buffer.take(),
1112                    ) {
1113                        self.nv12_texture_pool.push_back((tex, buf));
1114                    }
1115                }
1116                return Err(e);
1117            }
1118            Ok(())
1119        }
1120    }

1122    // ========== CPU 色彩转换（BGRA -> NV12，BT.601 limited range）==========
1123    /// 当 GPU 不支持 VideoProcessor 时，CPU 兜底：把 BGRA 读到内存里手算 NV12。
1124    /// **性能极差**，仅作 fallback，能不用就别用。
1125    fn convert_bgra_to_nv12_cpu(&mut self, bgra_texture: &ID3D11Texture2D) -> windows::core::Result<()> {
1126        unsafe {
1127            // 首次调用时创建 BGRA staging 纹理（GPU 复制过来后 CPU 就能读了）
1128            if self.bgra_staging_texture.is_none() {
1129                let mut src_desc = D3D11_TEXTURE2D_DESC::default();
1130                bgra_texture.GetDesc(&mut src_desc);
1131                let staging_desc = D3D11_TEXTURE2D_DESC {
1132                    Width: src_desc.Width,
1133                    Height: src_desc.Height,
1134                    MipLevels: 1,
1135                    ArraySize: 1,
1136                    Format: src_desc.Format, // 通常是 DXGI_FORMAT_B8G8R8A8_UNORM
1137                    SampleDesc: DXGI_SAMPLE_DESC { Count: 1, Quality: 0 },
1138                    Usage: D3D11_USAGE_STAGING,
1139                    BindFlags: 0,
1140                    CPUAccessFlags: D3D11_CPU_ACCESS_READ.0 as u32,
1141                    MiscFlags: 0,
1142                };
1143                let mut staging_opt: Option<ID3D11Texture2D> = None;
1144                self.device.CreateTexture2D(&staging_desc, None, Some(&mut staging_opt as *mut _))?;
1145                self.bgra_staging_texture = staging_opt;
1146            }
1147
1148            // 复制 BGRA 纹理到 staging（GPU -> CPU 可读）
1149            // CopyResource 是异步的，但接下来 Map 会强制等它完成（CPU/GPU 同步点）
1150            let dst_resource: ID3D11Resource = self.bgra_staging_texture.as_ref().unwrap().cast()?;
1151            let src_resource: ID3D11Resource = bgra_texture.cast()?;
1152            self.context.CopyResource(&dst_resource, &src_resource);
1153
1154            // 映射 staging 纹理以读取 BGRA 数据
1155            // 这里产生一个 CPU/GPU 同步：CPU 阻塞直到 GPU 把 CopyResource 做完
1156            let mut mapped = D3D11_MAPPED_SUBRESOURCE::default();
1157            self.context.Map(&dst_resource, 0, D3D11_MAP_READ, 0, Some(&mut mapped as *mut _))?;
1158
1159            let bgra_stride = mapped.RowPitch as usize;
1160            let bgra_data = mapped.pData as *const u8;
1161            let width = self.width as usize;
1162            let height = self.height as usize;
1163            let y_size = width * height;
1164
1165            // 获取 CPU 缓冲区并锁定（拿到裸指针）
1166            let buffer = self.cpu_buffer.as_ref().ok_or_else(|| {
1167                windows::core::Error::from(windows::Win32::Foundation::E_FAIL)
1168            })?;
1169
1170            let mut nv12_data: *mut u8 = std::ptr::null_mut();
1171            let mut max_len = 0u32;
1172            let mut cur_len = 0u32;
1173            // [1.7 修复] Lock 失败时不能直接 ? 早返回：
1174            //   staging texture 此时已被 Map 但还没 Unmap，错误路径会泄露 GPU 资源。
1175            //   这里记日志 + 强制 Unmap + 返回 Err（让 capture 主循环走 recover 路径）
1176            if let Err(e) = buffer.Lock(
1177                &mut nv12_data,
1178                Some(&mut max_len as *mut u32),
1179                Some(&mut cur_len as *mut u32),
1180            ) {
1181                #[cfg(any(debug_assertions, feature = "release-log"))]
1182                eprintln!("❌ [cpu] cpu_buffer.Lock 失败: {:?} (已 Unmap staging)", e);
1183                self.context.Unmap(&dst_resource, 0);
1184                return Err(e);
1185            }
1186            // [1.5 修复] Lock 返回 Ok 也要验证指针 / 长度：
1187            //   理论上 Lock 成功时 MF 一定会返回非空指针 + max_len >= NV12 帧大小 (width*height*3/2)。
1188            //   实际遇到过驱动 bug 返回 Ok 但 max_len=0 / ptr=NULL 的情况，
1189            //   后续 ptr.add()/copy_nonoverlapping 会踩到 NULL deref 或者 size 0 的 malloc 区，
1190            //   都是未定义行为（UB）。这里防御性检查并显式 fail fast。
1191            let expected_size = (width * height * 3 / 2) as u32;
1192            if nv12_data.is_null() || max_len < expected_size {
1193                #[cfg(any(debug_assertions, feature = "release-log"))]
1194                eprintln!(
1195                    "❌ [cpu] cpu_buffer.Lock 返回 Ok 但状态异常: ptr={:?}, max_len={}, 期望 >= {}",
1196                    nv12_data, max_len, expected_size
1197                );
1198                let _ = buffer.Unlock();
1199                self.context.Unmap(&dst_resource, 0);
1200                return Err(windows::core::Error::from(windows::Win32::Foundation::E_FAIL));
1201            }
1202
1203            // NV12 内存布局：前面 width*height 字节是 Y 平面，后面是 UV 交错
1204            let y_plane = nv12_data;
1205            let uv_plane = nv12_data.add(y_size);
1206            let nv12_stride = width; // MF_MT_DEFAULT_STRIDE = width
1207
1208            // ---- Y 平面：BGRA -> Y (BT.601 limited range) ----
1209            // 公式：Y = ((66*R + 129*G + 25*B + 128) >> 8) + 16
1210            // limited range 表示 Y 取值 [16, 235]，对应实际亮度
1211            for row in 0..height {
1212                let bgra_row = bgra_data.add(row * bgra_stride);
1213                let y_row = y_plane.add(row * nv12_stride);
1214                for col in 0..width {
1215                    let b = *bgra_row.add(col * 4) as i32;
1216                    let g = *bgra_row.add(col * 4 + 1) as i32;
1217                    let r = *bgra_row.add(col * 4 + 2) as i32;
1218                    let y = ((66 * r + 129 * g + 25 * b + 128) >> 8) + 16;
1219                    *y_row.add(col) = y.clamp(0, 255) as u8;
1220                }
1221            }
1222
1223            // ---- UV 平面：2x2 子采样，交错 U,V (BT.601 limited range) ----
1224            // 公式：U = ((-38*R - 74*G + 112*B + 128) >> 8) + 128
1225            //       V = ((112*R - 94*G - 18*B + 128) >> 8) + 128
1226            // UV 平面分辨率是 Y 的一半（4:2:0），每 2x2 像素共用一组 UV
1227            for row in 0..(height / 2) {
1228                let bgra_row0 = bgra_data.add(row * 2 * bgra_stride);
1229                let bgra_row1 = bgra_data.add((row * 2 + 1) * bgra_stride);
1230                let uv_row = uv_plane.add(row * nv12_stride);
1231                for col in 0..(width / 2) {
1232                    // 取 2x2 像素块的平均值
1233                    let i00 = col * 2 * 4;
1234                    let i01 = (col * 2 + 1) * 4;
1235
1236                    let b0 = *bgra_row0.add(i00) as i32;
1237                    let g0 = *bgra_row0.add(i00 + 1) as i32;
1238                    let r0 = *bgra_row0.add(i00 + 2) as i32;
1239
1240                    let b1 = *bgra_row0.add(i01) as i32;
1241                    let g1 = *bgra_row0.add(i01 + 1) as i32;
1242                    let r1 = *bgra_row0.add(i01 + 2) as i32;
1243
1244                    let b2 = *bgra_row1.add(i00) as i32;
1245                    let g2 = *bgra_row1.add(i00 + 1) as i32;
1246                    let r2 = *bgra_row1.add(i00 + 2) as i32;
1247
1248                    let b3 = *bgra_row1.add(i01) as i32;
1249                    let g3 = *bgra_row1.add(i01 + 1) as i32;
1250                    let r3 = *bgra_row1.add(i01 + 2) as i32;
1251
1252                    let b = (b0 + b1 + b2 + b3) / 4;
1253                    let g = (g0 + g1 + g2 + g3) / 4;
1254                    let r = (r0 + r1 + r2 + r3) / 4;
1255
1256                    let u = ((-38 * r - 74 * g + 112 * b + 128) >> 8) + 128;
1257                    let v = ((112 * r - 94 * g - 18 * b + 128) >> 8) + 128;
1258
1259                    *uv_row.add(col * 2) = u.clamp(0, 255) as u8;     // U (Cb)
1260                    *uv_row.add(col * 2 + 1) = v.clamp(0, 255) as u8; // V (Cr)
1261                }
1262            }
1263
1264            // 设置 buffer 当前长度（重要，否则编码器不知道这一帧实际有多少字节）
1265            // [1.7 修复] SetCurrentLength/Unlock 失败时也不能直接 ?：
1266            //   buffer 处于 Locked 状态 + staging 处于 Mapped 状态，
1267            //   错误路径必须先 Unlock + Unmap 再返回，避免锁死。
1268            let total_bytes = (y_size + width * height / 2) as u32;
1269            if let Err(e) = buffer.SetCurrentLength(total_bytes) {
1270                #[cfg(any(debug_assertions, feature = "release-log"))]
1271                eprintln!("❌ [cpu] cpu_buffer.SetCurrentLength 失败: {:?}", e);
1272                let _ = buffer.Unlock();
1273                self.context.Unmap(&dst_resource, 0);
1274                return Err(e);
1275            }
1276            if let Err(e) = buffer.Unlock() {
1277                #[cfg(any(debug_assertions, feature = "release-log"))]
1278                eprintln!("❌ [cpu] cpu_buffer.Unlock 失败: {:?}", e);
1279                self.context.Unmap(&dst_resource, 0);
1280                return Err(e);
1281            }
1282            self.context.Unmap(&dst_resource, 0);
1283
1284            Ok(())
1285        }
1286    }
1287
1288    // ========== 构建 IMFSample ==========
1289    /// 把当前帧打包成一个 IMFSample，准备喂给 `ProcessInput`。
1290    ///
1291    /// 三种路径：
1292    /// - 硬件 MFT：直接用 DXGI Surface Buffer（指向刚 VideoProcessor 写过的 NV12 纹理）
1293    /// - 软件 MFT + CPU 转换：直接用 cpu_buffer（数据已经在里面了）
1294    /// - 软件 MFT + 硬件转换：把 NV12 纹理拷到 staging，CPU 拷到 cpu_buffer
1295    pub fn build_sample(&mut self, timestamp: i64, duration: i64) -> Result<IMFSample> {
1296        unsafe {
1297            // MFCreateSample: 创建一个空的"样本容器"
1298            // [修复 §3.3] 软编路径复用输入 sample：
1299            //   - 池空时走 MFCreateSample；命中池则 RemoveAllBuffers 后复用
1300            //   - 硬编异步消费，不能复用（MFT 可能还在 GPU 队列里没读）
1301            //   - 软编同步消费，MFT 不会自己清 sample 的 buffers，必须手动 RemoveAllBuffers
1302            // [P0-2 修复] 硬件路径 MFCreateSample 失败时不需要回收池子（还没拿 tex/buf），
1303            //   但 sample 本身已经分配要让它正常 Drop。改成 match。
1304            let sample = if self.is_hardware {
1305                match MFCreateSample() {
1306                    Ok(s) => s,
1307                    Err(e) => {
1308                        // MFCreateSample 失败但 current_hw_* 还占着 → 还回池子
1309                        if let (Some(tex), Some(buf)) = (
1310                            self.current_hw_texture.take(),
1311                            self.current_hw_buffer.take(),
1312                        ) {
1313                            self.nv12_texture_pool.push_back((tex, buf));
1314                        }
1315                        return Err(e);
1316                    }
1317                }
1318            } else if let Some(s) = self.recycled_input_sample.take() {
1319                let _ = s.RemoveAllBuffers();
1320                s
1321            } else {
1322                MFCreateSample()?
1323            };
1324            // 设置时间戳（PTS）和时长，用于解码端按帧间隔播放
1325            // [P0-2 修复] 失败时 sample 已经构造 + current_hw_* 还占着 → 都得手动回收
1326            if let Err(e) = sample.SetSampleTime(timestamp) {
1327                // sample 直接 Drop（自动 Release）
1328                if self.is_hardware {
1329                    if let (Some(tex), Some(buf)) = (
1330                        self.current_hw_texture.take(),
1331                        self.current_hw_buffer.take(),
1332                    ) {
1333                        self.nv12_texture_pool.push_back((tex, buf));
1334                    }
1335                }
1336                return Err(e);
1337            }
1338            if let Err(e) = sample.SetSampleDuration(duration) {
1339                if self.is_hardware {
1340                    if let (Some(tex), Some(buf)) = (
1341                        self.current_hw_texture.take(),
1342                        self.current_hw_buffer.take(),
1343                    ) {
1344                        self.nv12_texture_pool.push_back((tex, buf));
1345                    }
1346                }
1347                return Err(e);
1348            }
1349
1350            if self.is_hardware {
1351                // 硬件路径：把当前帧的 DXGI Surface Buffer 挂到 sample 上
1352                // 当前帧的纹理和 buffer 是上一帧 `convert_bgra_to_nv12_hw` 设置的
1353                let tex = self.current_hw_texture.take().ok_or_else(|| windows::core::Error::from(windows::Win32::Foundation::E_FAIL))?;
1354                let buf = self.current_hw_buffer.take().ok_or_else(|| windows::core::Error::from(windows::Win32::Foundation::E_FAIL))?;
1355
1356                // [P0-2 修复] AddBuffer 失败时 (tex, buf) 还没进 pending_textures → 还回池子
1357                if let Err(e) = sample.AddBuffer(&buf) {
1358                    self.nv12_texture_pool.push_back((tex, buf));
1359                    return Err(e);
1360                }
1361                // 这张纹理已经交给 MFT，等它消费完才能复用，所以压进 pending_textures
1362                self.pending_textures.push_back((tex, buf));
1363            } else if self.use_cpu_conversion {
1364                // CPU 转换：数据已经直接写入 cpu_buffer 了
1365                sample.AddBuffer(self.cpu_buffer.as_ref().unwrap())?;
1366            } else {
1367                // 软件编码器 + 硬件色彩转换的"半 GPU"路径
1368                // 需要把 NV12 纹理读回 CPU，再写到 cpu_buffer
1369                let dst_resource: ID3D11Resource = self.staging_texture.as_ref().unwrap().cast()?;
1370                let src_resource: ID3D11Resource = self.nv12_texture.as_ref().unwrap().cast()?;
1371                self.context.CopyResource(&dst_resource, &src_resource);
1372
1373                let mut mapped = D3D11_MAPPED_SUBRESOURCE::default();
1374                self.context.Map(&dst_resource, 0, D3D11_MAP_READ, 0, Some(&mut mapped as *mut _))?;
1375
1376                let stride = mapped.RowPitch as usize;
1377                let width = self.width as usize;
1378                let height = self.height as usize;
1379                let y_size = width * height;
1380
1381                let buffer = self.cpu_buffer.as_ref().unwrap();
1382
1383                let mut data: *mut u8 = std::ptr::null_mut();
1384                let mut max_len = 0u32;
1385                let mut cur_len = 0u32;
1386                // [1.7 修复] Lock 失败时不能直接 ?：staging 此时 Mapped，错误路径会泄露 GPU 资源。
1387                //   记录日志 + 强制 Unmap staging + 把 sample 丢掉（不喂给编码器）+ 返回 Err
1388                if let Err(e) = buffer.Lock(
1389                    &mut data,
1390                    Some(&mut max_len as *mut u32),
1391                    Some(&mut cur_len as *mut u32),
1392                ) {
1393                    #[cfg(any(debug_assertions, feature = "release-log"))]
1394                    eprintln!("❌ [build_sample] cpu_buffer.Lock 失败: {:?} (丢 sample + Unmap staging)", e);
1395                    self.context.Unmap(&dst_resource, 0);
1396                    return Err(e);
1397                }
1398                // [1.5 修复] 同 cpu 路径：Lock 返回 Ok 也要验证指针 / 长度。
1399                //   见 convert_bgra_to_nv12_cpu 里的同款注释。
1400                let expected_size2 = (width * height * 3 / 2) as u32;
1401                if data.is_null() || max_len < expected_size2 {
1402                    #[cfg(any(debug_assertions, feature = "release-log"))]
1403                    eprintln!(
1404                        "❌ [build_sample] cpu_buffer.Lock 返回 Ok 但状态异常: ptr={:?}, max_len={}, 期望 >= {}",
1405                        data, max_len, expected_size2
1406                    );
1407                    let _ = buffer.Unlock();
1408                    self.context.Unmap(&dst_resource, 0);
1409                    return Err(windows::core::Error::from(windows::Win32::Foundation::E_FAIL));
1410                }
1411
1412                // 把 staging 里的 NV12 数据（Y 平面 + UV 平面）一行行拷到 cpu_buffer
1413                let y_src = mapped.pData as *const u8;
1414                let uv_src = y_src.add(stride * height);
1415
1416                for y in 0..height {
1417                    std::ptr::copy_nonoverlapping(y_src.add(y * stride), data.add(y * width), width);
1418                }
1419                for y in 0..(height / 2) {
1420                    std::ptr::copy_nonoverlapping(uv_src.add(y * stride), data.add(y_size + y * width), width);
1421                }
1422
1423                // [1.7 修复] SetCurrentLength/Unlock 失败时也不能直接 ?：
1424                //   同样的 staging Mapped + cpu_buffer Locked 资源泄露问题。
1425                //   这里要把 sample 丢掉（已经构造了）不喂给编码器，否则编码器收到残缺数据
1426                //   会输出脏 H.264 流。
1427                let total_bytes2 = (y_size + width * height / 2) as u32;
1428                if let Err(e) = buffer.SetCurrentLength(total_bytes2) {
1429                    #[cfg(any(debug_assertions, feature = "release-log"))]
1430                    eprintln!("❌ [build_sample] cpu_buffer.SetCurrentLength 失败: {:?} (丢 sample)", e);
1431                    let _ = buffer.Unlock();
1432                    self.context.Unmap(&dst_resource, 0);
1433                    return Err(e);
1434                }
1435                if let Err(e) = buffer.Unlock() {
1436                    #[cfg(any(debug_assertions, feature = "release-log"))]
1437                    eprintln!("❌ [build_sample] cpu_buffer.Unlock 失败: {:?} (丢 sample)", e);
1438                    self.context.Unmap(&dst_resource, 0);
1439                    return Err(e);
1440                }
1441                self.context.Unmap(&dst_resource, 0);
1442
1443                sample.AddBuffer(buffer)?;
1444            }
1445            Ok(sample)
1446        }
1447    }
1448    // 当 ProcessOutput 成功输出一帧时，回收最早输入的纹理
1449    /// 把 MFT 已经"消费完"的最早一张 NV12 纹理还回纹理池。
1450    /// 因为 MFT 内部有流水线，输入和输出不是 1:1 的，
1451    /// 所以用 `pending_textures` 队列来追踪 in-flight 的纹理。
1452    pub fn recycle_texture(&mut self) {
1453        if self.is_hardware {
1454            if let Some((tex, buf)) = self.pending_textures.pop_front() {
1455                self.nv12_texture_pool.push_back((tex, buf));
1456            }
1457        }
1458    }

1467    // ========== [B6 Phase 2] 强制 IDR ==========
1468
1469    /// 用户请求"下一帧必须是 IDR"。
1470    ///
1471    /// 典型场景：客户端刚连入 / 客户端断线重连 / 编码器自愈（C2 recover）后。
1472    /// 在下一次 `process_input` 调用时被消费。
1473    ///
1474    /// 实现策略：直接走 flush + restart 路径。`ICodecAPI` 路径在 windows crate
1475    /// 里要手搓 VARIANT（`ICodecAPI::SetValue` 接的是 `*const VARIANT`），
1476    /// 太复杂，性价比不高。flush 会丢 1-2 帧但保证 IDR，远程控制场景可接受。
1477    /// 行为：每次请求 IDR = 丢 1-2 帧 + 下一帧 IDR。
1478    ///
1479    /// 调用方：capture loop 的 proactive 分辨率变化 + MFT STREAM_CHANGE 分支，
1480    /// 以及 `Session::recover` / `Session::recreate` 重建后立即触发。
1481    pub fn request_keyframe(&mut self) {
1482        self.pending_force_idr = true;
1483    }
1484
1485    /// flush MFT + 重启流，强制下一帧为 IDR。
1486    /// 会丢弃所有 pending 帧（可能丢 1-2 帧画面），但保证下一帧是 IDR。
1487    unsafe fn force_idr_via_flush(&mut self) -> windows::core::Result<()> {
1488        // 1. flush —— 清空 MFT 内部缓冲的输入帧
1489        unsafe {
1490            self.encoder.ProcessMessage(MFT_MESSAGE_COMMAND_FLUSH, 0)?;
1491            // 2. 重启流（这些消息已经在 create_hw / create_sw 里发过一次，再发一次 idempotent）
1492            self.encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_BEGIN_STREAMING, 0)?;
1493            self.encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_START_OF_STREAM, 0)?;
1494        }
1495
1496        // 3. [P1-1.1 修复] flush 后 MFT 已经不引用 pending 里的纹理，但 clear() 只是 Drop
1497        //   元素——纹理没回到 nv12_texture_pool。request_keyframe 一旦启用，池子会单调缩水。
1498        //   改成 drain 回池子。
1499        for entry in self.pending_textures.drain(..) {
1500            self.nv12_texture_pool.push_back(entry);
1501        }
1502
1503        // 4. [P1-1.1 修复] 清掉 recycled_output_sample：flush 后它可能引用 MFT 旧内部状态
1504        //   （MFT 内部 sample 缓冲被 flush 清了，再还回去 MFT 行为未定义）。
1505        self.recycled_output_sample = None;
1506
1507        Ok(())
1508    }

1510    /// 把一帧 IMFSample 喂给硬件 MFT 的输入。
1511    ///
1512    /// 硬件 MFT 是异步的：调用 `ProcessInput` 前必须先等 `METransformNeedInput` 事件。
1513    /// （如果上一次 `process_output` 已经缓存了 NeedInput，就直接复用，不需要再等。）
1514    pub fn process_input(&mut self, sample: IMFSample) -> windows::core::Result<()> {
1515        // [B6 Phase 2] 消费强制 IDR 标志。
1516        // 当前实现：直接 flush + restart stream。丢 1-2 帧，保证下一帧是 IDR。
1517        if self.pending_force_idr {
1518            self.pending_force_idr = false;
1519
1520            unsafe {
1521                self.force_idr_via_flush()?;
1522            }
1523
1524            // [release-log] 强制 IDR 日志
1525            #[cfg(any(debug_assertions, feature = "release-log"))]
1526            {
1527                println!("🔑 强制 IDR 请求已处理 (via flush)");
1528            }
1529        }
1530
1531        unsafe {
1532            if self.is_hardware {
1533                let event_gen = self.event_gen.as_ref().unwrap();
1534
1535                // 如果没有待处理的 NeedInput 事件，则阻塞等待一个
1536                if !self.pending_need_input {
1537                    loop {
1538                        // GetEvent 拉一个事件，MF_EVENT_FLAG_NONE = 阻塞直到有事件
1539                        let event = event_gen.GetEvent(MF_EVENT_FLAG_NONE)?;
1540                        let event_type = event.GetType()?;
1541                        if event_type == METransformNeedInput.0 as u32 {
1542                            break;
1543                        }
1544                        // [修复 §2.3] 补 catch-all warn：
1545                        //   其他事件静默丢可能让 MFT 内部状态机错位，至少 log 一下
1546                        //   知道 MFT 实际发了什么。event 局部变量作用域结束自动 Release。
1547                        // [M6 修复] 加 cfg gate（异常事件偶尔触发，release 模式 0 开销）
1548                        #[cfg(any(debug_assertions, feature = "release-log"))]
1549                        eprintln!("[MFT] process_input 收到非 NeedInput 事件: type={}", event_type);
1550                    }
1551                }
1552                self.pending_need_input = false;
1553                // [修复 §3.3] 硬编异步消费，sample 不能复用，让它 Drop
1554                let result = self.encoder.ProcessInput(0, &sample, 0);
1555
1556                // [1.3 修复] ProcessInput 失败时撤销 build_sample 做的 pending_textures push：
1557                //   - 错误路径 sample 被 Drop → buf 的 refcount 减 1（但 pending_textures 里还有 1）
1558                //   - 不撤销的话下次 recycle_texture() 会把 (tex, 半释放的 buf) 还回池子
1559                //   - 之后任何 Lock/cast 操作都可能踩到无效指针或 AV
1560                //   - NVENC 1:1 模型下，pending_textures 队尾就是刚 push 的那个，pop_back 安全
1561                // [P1-5 修复] 1:1 模型假设不再硬编码：
1562                //   - is_hardware=true 不保证是 NVENC；QuickSync/AMF（带 B 帧）是 1:N。
1563                //   - 1:N 模式下 pop_back 会错误回收其他未消费帧的纹理，
1564                //     下次 recycle_texture 真正轮到时双重回收 → 拿空池子。
1565                //   - 加 debug_assert 兜底：pending_textures 队列长度期望 = MFT 已消费帧数；
1566                //     若 >1 说明 MFT 流水线 > 1，pop_back 假定失效。
1567                if result.is_err() {
1568                    if let Some((tex, buf)) = self.pending_textures.pop_back() {
1569                        self.nv12_texture_pool.push_back((tex, buf));
1570                    }
1571                    #[cfg(any(debug_assertions, feature = "release-log"))]
1572                    eprintln!("⚠️ [process_input] ProcessInput 失败, 已回收对应纹理到池");
1573                }
1574
1575                // [release-log] 一次性首帧日志
1576                #[cfg(any(debug_assertions, feature = "release-log"))]
1577                {
1578                    if result.is_ok() && !self.logged_first_input {
1579                        self.logged_first_input = true;
1580                        println!("✅ MFT 首次成功接收输入帧 ({}, 异步硬件路径)", self.width);
1581                    }
1582                }
1583
1584                result
1585            } else {
1586                // 软件 MFT 是同步的，直接喂。ProcessInput 返回时 MFT 已读完数据。
1587                // [修复 §3.3] 把 sample 所有权拿过来：喂完还回输入池，避开下帧 MFCreateSample。
1588                self.encoder.ProcessInput(0, &sample, 0)?;
1589
1590                // [release-log] 一次性首帧日志
1591                #[cfg(any(debug_assertions, feature = "release-log"))]
1592                {
1593                    if !self.logged_first_input {
1594                        self.logged_first_input = true;
1595                        println!("✅ MFT 首次成功接收输入帧 ({}, 同步软件路径)", self.width);
1596                    }
1597                }
1598
1599                // 还回池前清空残留 buffer（虽然 MFT 应该不会自己 AddBuffer，
1600                // 但保险起见手动清，避免下次取出时残留上次的数据指针）
1601                let _ = sample.RemoveAllBuffers();
1602                self.recycled_input_sample = Some(sample);
1603                Ok(())
1604            }
1605        }
1606    }
1607    /// 从硬件 MFT 拉一帧 H.264 输出。
1608    ///
1609    /// 返回 `Ok(Some(sample))` 表示拿到一帧输出，里面装着 H.264 码流字节；
1610    /// 返回 `Err(MF_E_TRANSFORM_NEED_MORE_INPUT)` 表示 MFT 暂时没东西给你；
1611    /// 其他 Err 是真错误。
1612    pub fn process_output(&mut self) -> windows::core::Result<Option<IMFSample>> {
1613        unsafe {
1614            // 准备输出 buffer 数组。pSample 包在 ManuallyDrop 里，
1615            // 因为 MFT_OUTPUT_DATA_BUFFER 是可拷贝结构，rust 自动 Drop 会触发多次 Release。
1616            // 真正的释放靠下面那两行 ManuallyDrop::take。
1617            let mut output_buffers = [MFT_OUTPUT_DATA_BUFFER {
1618                dwStreamID: 0,
1619                pSample: ManuallyDrop::new(self.create_output_sample()?),
1620                dwStatus: 0,
1621                pEvents: ManuallyDrop::new(None),
1622            }];
1623
1624            if self.is_hardware {
1625                // 硬件 MFT 异步：必须先收到 METransformHaveOutput 事件才能调 ProcessOutput
1626                //
1627                // [P0-3 修复] CLAUDE.md §坑 A 守则：output_buffers 是 Copy struct 数组，
1628                //   Drop 时不会自动 take ManuallyDrop 字段 → sample 永久泄漏。
1629                //   任何 `?` 早返回都必须显式 take pSample/pEvents 再 return。
1630                //   把整个事件循环包成 match { Err => 手动清理 + return Err }
1631                let event_gen = self.event_gen.as_ref().unwrap();
1632                let event_loop_result: windows::core::Result<()> = (|| {
1633                    loop {
1634                        let event = event_gen.GetEvent(MF_EVENT_FLAG_NONE)?;
1635                        let event_type = event.GetType()?;
1636                        if event_type == METransformHaveOutput.0 as u32 {
1637                            return Ok(()); // 收到可输出事件，去调用 ProcessOutput
1638                        } else if event_type == METransformNeedInput.0 as u32 {
1639                            // 缓存 NeedInput 事件，下次 process_input 能跳过阻塞等待
1640                            self.pending_need_input = true;
1641                            return Err(windows::core::Error::from(MF_E_TRANSFORM_NEED_MORE_INPUT));
1642                        }
1643                        // [修复 §2.3] 补 catch-all warn：非典型事件静默丢可能让 MFT 状态机
1644                        //   错位，至少 log 一下知道 MFT 发了什么。
1645                        // [M6 修复] 加 cfg gate
1646                        #[cfg(any(debug_assertions, feature = "release-log"))]
1647                        eprintln!("[MFT] process_output 收到非 HaveOutput/NeedInput 事件: type={}", event_type);
1648                    }
1649                    #[allow(unreachable_code)]
1650                    Ok(())
1651                })();
1652                if let Err(e) = event_loop_result {
1653                    // [P0-3 修复] 任意失败路径都手动 take + recycle sample，events 取空即可
1654                    let out_sample = ManuallyDrop::take(&mut output_buffers[0].pSample);
1655                    let _out_events = ManuallyDrop::take(&mut output_buffers[0].pEvents);
1656                    self.recycle_output_sample(out_sample);
1657                    return Err(e);
1658                }
1659            }
1660
1661            let mut status = 0u32;
1662            let process_result = self.encoder.ProcessOutput(0, &mut output_buffers, &mut status);
1663            // 从 ManuallyDrop 里把 sample 拿出来 —— 之后它走正常的 rust Drop，
1664            // 会被回收进 recycled_output_sample 池或者释放。
1665            let out_sample = ManuallyDrop::take(&mut output_buffers[0].pSample);
1666            let _out_events = ManuallyDrop::take(&mut output_buffers[0].pEvents);
1667
1668            match process_result {
1669                Ok(_) => {
1670                    // [release-log] 一次性首帧输出日志
1671                    #[cfg(any(debug_assertions, feature = "release-log"))]
1672                    {
1673                        if !self.logged_first_output {
1674                            self.logged_first_output = true;
1675                            println!("✅ MFT 首次成功产出 H.264 输出帧 (这是关键信号：编码器真的在工作，不是假死)");
1676                        }
1677                    }
1678                    Ok(out_sample)
1679                }
1680                Err(e) if e.code() == MF_E_TRANSFORM_NEED_MORE_INPUT => {
1681                    // MFT 暂时没输出，把 sample 还回池里
1682                    self.recycle_output_sample(out_sample);
1683                    Err(e)
1684                }
1685                Err(e) => {
1686                    // [1.2 修复] 任何非 NEED_MORE_INPUT 的错误也走 recycle 路径：
1687                    //   原实现 Err(e) 直接返回，out_sample 走出作用域被 Release，
1688                    //   下次 create_output_sample 又分配新的。设备抖动时大量 Release 触发 GPU 句柄抖动。
1689                    //   设备真挂了的话 encoder 整个 Drop，池子自动清；这里回收是稳态优化。
1690                    #[cfg(any(debug_assertions, feature = "release-log"))]
1691                    eprintln!("⚠️ [process_output] ProcessOutput 失败 (code=0x{:08X}), 回收 sample 后返回错误", e.code().0 as u32);
1692                    self.recycle_output_sample(out_sample);
1693                    Err(e)
1694                }
1695            }
1696        }
1697    }
1698}

1701// ========== [V6-R1] H.264 Annex-B 字节流转换（模块级自由函数）==========
1702
1703/// [V6-R1] 从输出 MediaType 提取 SPS+PPS AVCC 配置记录，转成 Annex-B 字节流。
1704///
1705/// ## 为什么需要这个函数
1706///
1707/// MFT 用 `MFVideoFormat_H264` 输出时，码流是 **AVCC 格式**（H.264 Annex E）：
1708/// - 每个 NALU 前有 4 字节大端长度前缀
1709/// - SPS/PPS **不在码流里**，单独存在 MediaType 的 `MF_MT_MPEG_SEQUENCE_HEADER` blob
1710///
1711/// 但 ffmpeg / 浏览器 MSE / libavcodec / VideoToolbox 期望 **Annex-B 格式**（H.264 Annex B）：
1712/// - 每个 NALU 前有 `00 00 00 01` 起始码
1713/// - SPS/PPS 必须独立出现在 IDR 帧**之前**
1714///
1715/// 所以本函数：
1716/// 1. 调 `output_type.GetBlob(MF_MT_MPEG_SEQUENCE_HEADER, ...)` 拿 AVCC 配置 blob
1717/// 2. 按 ISO/IEC 14496-15 §5.2.4.1 解析（5 字节头 + num_sps + 每个 sps 长度 + NALU + num_pps + 同理）
1718/// 3. 把每个 SPS/PPS NALU 用 `00 00 00 01` 起始码包起来
1719///
1720/// ## 失败处理
1721///
1722/// 返回 `Err` 而不是 `panic!`：MFT 输出 MediaType **理论上一定** 包含 SPS/PPS，
1723/// 但如果 MFT 实现 bug 或者某种 degenerate 情况 blob 不存在，应该 graceful 失败
1724/// 让调用方 fallback 到空 Vec（不阻塞编码流程）。
1725///
1726/// ## 安全
1727///
1728/// `unsafe` 是必须的（GetBlob 接裸指针）。
1729pub fn extract_sps_pps_annex_b(output_type: &IMFMediaType) -> windows::core::Result<Vec<u8>> {
1730    unsafe {
1731        // 1. 先问 blob 长度（windows 0.62.2 的 GetBlobSize 返回 Result<u32>）
1732        let blob_len = output_type.GetBlobSize(&MF_MT_MPEG_SEQUENCE_HEADER)?;
1733
1734        if blob_len == 0 {
1735            return Err(windows::core::Error::from(windows::Win32::Foundation::E_FAIL));
1736        }
1737
1738        // 2. 分配 + 实际取 blob（windows 0.62.2 GetBlob 签名：*const GUID, &mut [u8], Option<*mut u32>）
1739        let mut blob = vec![0u8; blob_len as usize];
1740        output_type.GetBlob(
1741            &MF_MT_MPEG_SEQUENCE_HEADER,
1742            &mut blob,
1743            None,
1744        )?;
1745
1746        // 3. 解析 AVCC 配置记录
1747        //    layout:
1748        //      [1] configurationVersion (固定 1)
1749        //      [1] AVCProfileIndication
1750        //      [1] profile_compatibility
1751        //      [1] AVCLevelIndication
1752        //      [1] reserved(6 bits=111111) + lengthSizeMinusOne(2 bits)
1753        //      [1] numOfSequenceParameterSets (高 3 bit 是 reserved=111，低 5 bit 是 count)
1754        //      [2] sps_length (BE)
1755        //      [sps_length] sps_data
1756        //      ...重复每个 SPS
1757        //      [1] numOfPictureParameterSets
1758        //      [2] pps_length (BE)
1759        //      [pps_length] pps_data
1760        //      ...重复每个 PPS
1761        let mut out = Vec::with_capacity(blob_len as usize + 8);
1762        let mut c = 5usize; // 跳过 configurationVersion/AVCProfile/profile_compat/AVCLevel/lengthSize
1763
1764        if blob.len() < 6 {
1765            return Err(windows::core::Error::from(windows::Win32::Foundation::E_FAIL));
1766        }
1767
1768        let num_sps = (blob[c] & 0x1f) as usize; // 低 5 bit 是 SPS 数量
1769        c += 1;
1770
1771        for _ in 0..num_sps {
1772            if c + 2 > blob.len() {
1773                break; // 防御性：blob 截断就跳出
1774            }
1775            let sps_len = u16::from_be_bytes([blob[c], blob[c + 1]]) as usize;
1776            c += 2;
1777            if c + sps_len > blob.len() || sps_len == 0 {
1778                break;
1779            }
1780            out.extend_from_slice(&[0, 0, 0, 1]); // Annex-B start code (4 字节)
1781            out.extend_from_slice(&blob[c..c + sps_len]);
1782            c += sps_len;
1783        }
1784
1785        if c >= blob.len() {
1786            return Ok(out);
1787        }
1788
1789        let num_pps = blob[c] as usize;
1790        c += 1;
1791
1792        for _ in 0..num_pps {
1793            if c + 2 > blob.len() {
1794                break;
1795            }
1796            let pps_len = u16::from_be_bytes([blob[c], blob[c + 1]]) as usize;
1797            c += 2;
1798            if c + pps_len > blob.len() || pps_len == 0 {
1799                break;
1800            }
1801            out.extend_from_slice(&[0, 0, 0, 1]);
1802            out.extend_from_slice(&blob[c..c + pps_len]);
1803            c += pps_len;
1804        }
1805
1806        Ok(out)
1807    }
1808}

1810/// [V6-R1] AVCC 字节流转 Annex-B。
1811///
1812/// ## 算法
1813///
1814/// AVCC 格式：
1815/// ```text
1816/// [u32 BE length1][NALU1 bytes][u32 BE length2][NALU2 bytes]...
1817/// ```
1818///
1819/// Annex-B 格式：
1820/// ```text
1821/// [00 00 00 01][NALU1 bytes][00 00 00 01][NALU2 bytes]...
1822/// ```
1823///
1824/// 所以转换就是把每个 4 字节长度前缀换成 `00 00 00 01` 起始码。
1825///
1826/// 第一个 IDR NALU（header byte & 0x1f == 5）**之前**插入 `prepend_sps_pps`（已
1827/// 是 Annex-B 格式的 SPS+PPS 串）。这保证客户端解码器拿到 IDR 时已经有序列头。
1828///
1829/// 非 IDR 帧（P/B slice）不重复插 SPS/PPS（节省带宽，但有些客户端为简化处理每帧都带——可接受，
1830/// 因为 `sps_pps_annex_b` 缓存里只有一次 IDR 触发插）。
1831///
1832/// ## 边界
1833///
1834/// - 返回 `(Vec<u8>, bool)`：第二个元素是 `is_keyframe`，告诉调用方这帧是不是 IDR。
1835/// - `prepend_sps_pps` 为空时不插入（graceful fallback）。
1836/// - 输入字节被 defensive 解析：如果某个 NALU 长度字段错乱导致越界，跳出循环。
1837///
1838/// ## 安全
1839///
1840/// 纯函数，无 unsafe。
1841pub fn avcc_to_annex_b(avcc: &[u8], prepend_sps_pps: &[u8]) -> (Vec<u8>, bool) {
1842    let mut out = Vec::with_capacity(avcc.len() + prepend_sps_pps.len());
1843    let mut c = 0usize;
1844    let mut saw_idr = false;
1845
1846    while c + 4 <= avcc.len() {
1847        // 4 字节大端长度
1848        let nalu_len = u32::from_be_bytes([
1849            avcc[c],
1850            avcc[c + 1],
1851            avcc[c + 2],
1852            avcc[c + 3],
1853        ]) as usize;
1854        c += 4;
1855
1856        // 防御：长度错乱跳出
1857        if nalu_len == 0 || c + nalu_len > avcc.len() {
1858            break;
1859        }
1860
1861        let nal_type = avcc[c] & 0x1f;
1862
1863        // IDR (type 5) 前注入 SPS+PPS（只插一次，处理多 slice IDR 罕见情况）
1864        if nal_type == 5 && !saw_idr {
1865            out.extend_from_slice(prepend_sps_pps);
1866            saw_idr = true;
1867        }
1868
1869        // 4 字节长度 → Annex-B 起始码
1870        out.extend_from_slice(&[0, 0, 0, 1]);
1871        out.extend_from_slice(&avcc[c..c + nalu_len]);
1872        c += nalu_len;
1873    }
1874
1875    (out, saw_idr)
1876}

1954// 内存显存释放
1955//
1956// Drop 是 Rust 的 RAII 析构。结构体里的每个 COM 智能指针字段在 Drop 时会自动 Release
1957//（windows crate 的 `IUnknown::drop` 会调 vtable 的 `Release`），所以本函数只需要做
1958// "业务层"的清理：MFT 流结束 / 解绑 D3D 设备 / 关闭 MF runtime。
1959//
1960// 字段会按声明顺序的逆序自动 drop（→ Release COM）。`MFShutdown` 是 body 的最后一
1961// 个语句，body 返回后才轮到 self 字段 drop —— 满足微软文档"MFShutdown 必须在所有
1962// MF 对象 Release 之后"的要求。
1963impl Drop for H264Encoder {
1964    fn drop(&mut self) {
1965        unsafe {
1966            // [release-log] drop trace
1967            #[cfg(any(debug_assertions, feature = "release-log"))]
1968            println!("[Drop] 正在清理 H264Encoder 资源...");
1969
1970            // 通知 MFT 结束流，并排空内部缓冲的剩余帧
1971            let _ = self.encoder.ProcessMessage(MFT_MESSAGE_NOTIFY_END_STREAMING, 0);
1972            let _ = self.encoder.ProcessMessage(MFT_MESSAGE_COMMAND_DRAIN, 0);
1973
1974            // [V4-C3] 微软文档规定顺序：先 NOTIFY_RELEASE_RESOURCES 再 SET_D3D_MANAGER(0)
1975            // 否则严格 MFT（QuickSync）在解绑时仍持有 GPU 内部资源 → AV / D3D11 内部错。
