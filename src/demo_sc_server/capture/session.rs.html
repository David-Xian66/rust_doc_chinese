1//! [C2] CaptureSession: 把"DXGI 桌面复制 + H.264 编码器"打包成可恢复的会话单元。
2//!
3//! ## 为什么需要这个抽象
4//!
5//! 原来 capture.rs 里 `dup` 和 `h264_impl` 是两个独立的变量。遇到错误时（比如 D3D11
6//! device lost、笔记本睡眠唤醒、显卡驱动崩）我们只能 `?` 退出，整个服务端挂掉。
7//!
8//! 把它们打包成 `CaptureSession` 后，可以：
9//! - 一次性释放所有 GPU 资源（旧 device 上的 texture/encoder 都不能用了）
10//! - 用一组备份参数（屏幕索引 / FPS / 码率 / Profile）整体重建
11//! - 调用 `recover()` 在 500ms ~ 2s 内恢复，输出强制 IDR（配合 B6），客户端无感
12//!
13//! ## 设计取舍
14//!
15//! - **整个 session 重建**（不是局部重建）：device lost 场景下旧 device 上的所有资源
16//!   都失效，局部重建反而要写复杂的"哪些要重建"逻辑。简单优先。
17//! - **节流策略**：30 秒内连续失败 5 次 → 放弃退出。避免驱动持续故障时死循环。
18//! - **错误分类兜底**：未识别的 HRESULT 当 Fatal（宁可退出也不掩盖 bug）。
19//!
20//! ## 不做
21//!
22//! - 不做"GPU OOM 自动降级到软编"——本项目软编不是 fallback（CPU 性能不够），降级没用
23//! - 不做"多客户端共享 session"——那是 D4 的事，每个客户端独立 session
24

42/// 一次"采集 + 编码"会话。
43///
44/// 持有：
45/// - 配置备份（用于 recover / recreate 时重建）
46/// - DXGI 桌面复制（`DxgiDuplicationApi`，内部有 D3D11 device）
47/// - H.264 编码器
48/// - GPU 合成器
49///
50/// ## [L1] release / recreate
51///
52/// 锁屏 / Modern Standby 时调用 `release()` 主动 drop GPU 资源（释放 21 MB VRAM，
53/// 让 GPU 进低功耗）。`recreate()` 从保留的 config 字段重建。
54///
55/// 实现：`dup` / `encoder` / `compositor` 改成 `Option<>`，
56/// `width` / `height` 在 released 时设 0。
57pub struct CaptureSession {
58    // ===== 配置（recover / recreate 时用，不重建） =====
59    enumerate_index: usize,
60    fps: u32,
61    bitrate_bps: u32,
62    profile: H264Profile,
63
64    // ===== 资源（recover / recreate 时整体重建） =====
65    /// 保留 Monitor 是为了将来扩展（比如热切换屏幕），目前 recover 时通过
66    /// `enumerate_index` 重新拿，字段暂时不读。
67    #[allow(dead_code)]
68    monitor: Monitor,
69    /// DXGI 桌面复制会话（None = 已 release）
70    dup: Option<DxgiDuplicationApi>,
71    /// H.264 编码器（None = 已 release）
72    encoder: Option<H264Encoder>,
73    /// [D1 Phase 1] GPU 合成器（None = 已 release）
74    compositor: Option<DirtyCompositor>,
75
76    // ===== 状态（recover / recreate 后刷新） =====
77    /// 当前分辨率（0 = 已 release）。
78    width: u32,
79    height: u32,
80
81    // ===== [V6-R2] 网络层输出缓冲 =====
82    /// 已编码的 H.264 帧队列（Annex-B 格式 + wire format 字段）。
83    ///
84    /// capture loop 每编出一帧 push 一个 `WireFrame`。
85    /// 网络层接上后会用 `mpsc::Sender<WireFrame>` 替换 `VecDeque`，
86    /// sender thread 异步从 receiver 取出序列化发到 socket。
87    ///
88    /// 上限 60 帧 ≈ 1 秒 @60fps，防止网络层没接上时无限增长占内存。
89    /// 超过上限时丢最老的帧（实际场景里 sender 接上后就不会触达上限）。
90    frame_buffer: VecDeque<WireFrame>,
91}
92
93/// [V6-R2] frame_buffer 上限，超过时丢最老的帧。
94const FRAME_BUFFER_MAX: usize = 60;
95
96/// 错误分类结果。
97///
98/// 主循环用这个分类决定要不要 `recover()`：
99/// - `Transient`: 业务流错误（NEED_MORE_INPUT 等），继续循环
100/// - `Recoverable`: 设备暂时不可用（AccessLost / GPU OOM），调 `recover()` 重试
101/// - `Fatal`: 不可恢复，退出主循环
102#[derive(Debug)]
103pub enum SessionError {
104    /// 业务流错误：不是真错误，下一帧重试就好
105    /// 携带的 String 是给日志看的 reason（不影响 match）
106    Transient(#[allow(dead_code)] String),
107    /// 可恢复错误：DXGI AccessLost / D3D device lost / GPU OOM
108    Recoverable(String),
109    /// 不可恢复：MFStartup 失败、参数错误、未知 HRESULT
110    Fatal(String),
111}
112
113impl SessionError {
114    /// 用 HRESULT 分类 Windows 错误。
115    ///
116    /// 注意：
117    /// - `DXGI_ERROR_DEVICE_REMOVED` / `_RESET` / `_DRIVER_INTERNAL_ERROR` 标 Recoverable
118    ///   —— 我们的策略是整个 session 重建 = 重新拿 device，所以能恢复
119    /// - `STREAM_CHANGE` 已经在 capture.rs 里转 renegotiate，不会到这里；兜底当 Recoverable
120    pub fn classify(err: &windows::core::Error) -> Self {
121        let code = err.code();
122        // 取 HRESULT 的原始 u32 码（用于日志和未知错误回退）
123        let raw = code.0 as u32;
124
125        // 业务流
126        if code == windows::Win32::Media::MediaFoundation::MF_E_TRANSFORM_NEED_MORE_INPUT {
127            return Self::Transient("NEED_MORE_INPUT".into());
128        }
129        if code == windows::Win32::Media::MediaFoundation::MF_E_TRANSFORM_STREAM_CHANGE {
130            return Self::Recoverable("STREAM_CHANGE (未在 capture 层处理)".into());
131        }
132
133        // 可恢复（GPU/设备层面）—— 直接比较 HRESULT 常量（PartialEq 比较内部值）
134        if code == DXGI_ERROR_ACCESS_LOST {
135            return Self::Recoverable("DXGI_ACCESS_LOST".into());
136        }
137        if code == E_OUTOFMEMORY {
138            return Self::Recoverable("E_OUTOFMEMORY (GPU OOM)".into());
139        }
140        if code == DXGI_ERROR_DEVICE_REMOVED {
141            return Self::Recoverable("DXGI_DEVICE_REMOVED".into());
142        }
143        if code == DXGI_ERROR_DEVICE_RESET {
144            return Self::Recoverable("DXGI_DEVICE_RESET".into());
145        }
146        if code == DXGI_ERROR_DRIVER_INTERNAL_ERROR {
147            return Self::Recoverable("DXGI_DRIVER_INTERNAL_ERROR".into());
148        }
149
150        // 未知 → Fatal（防御性：宁可退出也不掩盖 bug）
151        Self::Fatal(format!("未知错误 0x{:08X}", raw))
152    }
153}

155impl CaptureSession {
156    /// 首次创建会话。
157    ///
158    /// 流程：
159    /// 1. 拿到屏幕 Monitor
160    /// 2. 创建 DXGI duplication（内部创建 D3D11 device）
161    /// 3. 从 dup 的 device 创建 H.264 编码器
162    /// 4. 读出当前分辨率存到 width/height
163    pub fn new(
164        enumerate_index: usize,
165        fps: u32,
166        bitrate_bps: u32,
167        profile: H264Profile,
168    ) -> Result<Self, Box<dyn std::error::Error>> {
169        let monitor = Monitor::from_index(enumerate_index)?;
170        let dup = DxgiDuplicationApi::new(monitor)?;
171        // DXGI 给的当前屏幕模式描述里有分辨率
172        let desc = dup.duplication_desc();
173        let width = desc.ModeDesc.Width;
174        let height = desc.ModeDesc.Height;
175
176        // 用 dup 内部的 D3D11 device 创建编码器
177        let encoder = H264Encoder::new(dup.device(), width, height, fps, bitrate_bps, profile)?;
178
179        // [D1 Phase 1] 创建 GPU 合成器（2 张 BGRA 纹理，prev + current）
180        let compositor = DirtyCompositor::new(dup.device(), width, height)?;
181        #[cfg(any(debug_assertions, feature = "release-log"))]
182        {
183            println!(
184                "🎨 [compositor] 已创建 {}x{} BGRA 纹理对 (prev + current, 共 ~{} MB VRAM)",
185                width,
186                height,
187                (width as u64 * height as u64 * 4 * 2) / 1_000_000
188            );
189        }
190
191        Ok(Self {
192            enumerate_index,
193            fps,
194            bitrate_bps,
195            profile,
196            monitor,
197            dup: Some(dup),
198            encoder: Some(encoder),
199            compositor: Some(compositor),
200            width,
201            height,
202            frame_buffer: VecDeque::with_capacity(FRAME_BUFFER_MAX),
203        })
204    }
205
206    /// [V6-R2] 把编码好的 H.264 帧推入 frame_buffer。
207    ///
208    /// 超过 [`FRAME_BUFFER_MAX`]（60 帧）时丢最老的帧（FIFO）。
209    /// 当前 sender thread 还没接上，所以这只是占位逻辑——网络层接上后会替换。
210    pub fn push_frame(&mut self, frame: WireFrame) {
211        if self.frame_buffer.len() >= FRAME_BUFFER_MAX {
212            self.frame_buffer.pop_front();
213        }
214        self.frame_buffer.push_back(frame);
215    }
216
217    /// [V6-R2] 当前 frame_buffer 长度（用于 release-log 日志）。
218    pub fn frame_buffer_len(&self) -> usize {
219        self.frame_buffer.len()
220    }
221
222    /// [V6-R2] 取出 frame_buffer 队首（sender 接上后用）。
223    ///
224    /// 当前未使用，编译期保留接口防止 sender 接上时改 capture.rs 调用点。
225    #[allow(dead_code)]
226    pub fn pop_frame(&mut self) -> Option<WireFrame> {
227        self.frame_buffer.pop_front()
228    }
229
230    /// 重建整个会话。
231    ///
232    /// 调用时机：检测到 D3D device removed、MFT 持续错误、GPU OOM 等可恢复错误。
233    ///
234    /// 关键点：
235    /// - **必须完全重建 device**，不能复用旧 device（旧 device 已经失效）
236    /// - 重置前通过 `mem::replace` 让旧资源显式 Drop，触发 MFT Release 等清理
237    /// - 重建后**强制 IDR**（调用 `encoder.request_keyframe()`），配合 B6：
238    ///   客户端解码器在 recover 期间的画面中断后，看到 IDR 就知道"这是新一段开始"
239    pub fn recover(&mut self) -> Result<(), Box<dyn std::error::Error>> {
240        // 1. 显式 Drop 旧资源。Rust 的赋值语义不保证 LHS 在 RHS 构造前 drop，
241        //    所以用 mem::replace 把字段暂时替换成"占位"（实际上 dup/monitor
242        //    没有 Default，我们直接 take+重建）。
243        //
244        //    最简方案：整个 self 重建。
245        //    旧 self 的所有字段会在 reassign 时被 Drop（Rust 默认行为），
246        //    顺序按字段声明顺序的逆序：encoder → dup → monitor → profile → ...
247        //    这正是我们想要的释放顺序（编码器先释放，再释放 dup，monior 最后）。
248        let new_session = Self::new(self.enumerate_index, self.fps, self.bitrate_bps, self.profile)?;
249        // 强制 IDR：让下一帧是关键帧，客户端能立刻恢复画面
250        // [1.4 修复] 把 .unwrap() 改成 .expect()：
251        //   new() 成功返回意味着 encoder/dup/compositor 都被设为 Some()，
252        //   但如果将来 new() 改成"成功但部分资源未创建"会出 bug。
253        //   expect 提供一个明确错误信息，比 .unwrap() 的 panic 信息可读。
254        let mut new_session = new_session;
255        new_session
256            .encoder
257            .as_mut()
258            .expect("recover: new_session 刚成功创建，encoder 必须是 Some")
259            .request_keyframe();
260        *self = new_session;
261        Ok(())
262    }

280    /// [L1] 当前 session 是否已 release（GPU 资源已 drop）
281    ///
282    /// `release()` 一次性 `take()` 掉 `dup` / `encoder` / `compositor` 三个 Option，
283    /// 所以只要任一字段为 None，其余也必为 None。
284    pub fn is_released(&self) -> bool {
285        self.dup.is_none()
286    }
287    /// Mut access to H264Encoder。如果已 release 会 panic，主循环在 paused 分支里不会调用。
288    pub fn encoder(&mut self) -> &mut H264Encoder {
289        self.encoder
290            .as_mut()
291            .expect("session.encoder() called while released")
292    }
293    /// Const access to H264Encoder。
294    pub fn encoder_ref(&self) -> &H264Encoder {
295        self.encoder
296            .as_ref()
297            .expect("session.encoder_ref() called while released")
298    }
299    /// Mut access to DXGI duplication API。
300    pub fn dup(&mut self) -> &mut DxgiDuplicationApi {
301        self.dup
302            .as_mut()
303            .expect("session.dup() called while released")
304    }

306    /// [L1] 主动释放 GPU 资源（锁屏 / Modern Standby / DXGI 持续失败时调用）。
307    ///
308    /// 把 dup / encoder / compositor 三个 Option 字段 take() 出来（drop 释放 GPU 资源），
309    /// width / height 设 0。config 字段（enumerate_index / fps / bitrate_bps / profile / monitor）
310    /// 保留，recreate() 时复用。
311    ///
312    /// 调用前请确认 is_released() == false（重复 release 是 no-op）。
313    pub fn release(&mut self) {
314        if self.is_released() {
315            return;
316        }
317        // take() 把 Option 替换成 None，旧值在这里 drop（GPU 资源 Release）
318        drop(self.dup.take());
319        drop(self.encoder.take());
320        drop(self.compositor.take());
321        self.width = 0;
322        self.height = 0;
323        println!("🗑️ [session] GPU 资源已释放 (VRAM 回收 + GPU 进低功耗)");
324    }

326    /// [L1] 从保留的 config 重建整个 session（解锁 / Modern Standby 唤醒时调用）。
327    ///
328    /// 本质上和 `recover()` 类似，但 `recover()` 用于"设备丢失后的错误恢复"，
329    /// `recreate()` 用于"主动 release 后的重建"。两者功能上等价。
330    ///
331    /// [1.13 修复] 前置检查：必须已 release 才能 recreate，避免在已激活的 session 上重复重建
332    ///   导致 GPU 资源泄漏（旧 session 资源没释放就被覆盖）。
333    pub fn recreate(&mut self) -> Result<(), Box<dyn std::error::Error>> {
334        // [1.13] 前置检查：未 release 时调用 recreate 是 caller bug，直接返回错误而不是重建
335        if !self.is_released() {
336            return Err("recreate() called on a non-released session (caller bug)".into());
337        }
338        let new_session = Self::new(
339            self.enumerate_index,
340            self.fps,
341            self.bitrate_bps,
342            self.profile,
343        )?;
344        let mut new_session = new_session;
345        // 强制 IDR：让下一帧是关键帧，客户端能立刻恢复画面
346        // [1.4 修复] 把 .unwrap() 改成 .expect()：见 recover() 注释
347        new_session
348            .encoder
349            .as_mut()
350            .expect("recreate: new_session 刚成功创建，encoder 必须是 Some")
351            .request_keyframe();
352        *self = new_session;
353        println!("🏗️ [session] 已从保留 config 重建 (force IDR)");
354        Ok(())
355    }

357    /// [D1 Phase 1] GPU 合成：把当前帧 + 脏矩形合并到 current_bgra。
358    ///
359    /// 调用时机：capture.rs 主循环里，dirty_rects 提取之后、`convert_bgra_to_nv12` 之前。
360    /// 返回：合成后的 BGRA 纹理的 owned 句柄（已 AddRef，独立于 session，后续可以自由使用）
361    pub fn composite(
362        &mut self,
363        current_frame: &ID3D11Texture2D,
364        dirty_rects: &[RECT],
365    ) -> windows::core::Result<ID3D11Texture2D> {
366        let device_context: &ID3D11DeviceContext = self
367            .dup
368            .as_ref()
369            .expect("session.composite() called while released")
370            .device_context();
371        self.compositor
372            .as_mut()
373            .expect("session.composite() called while released")
374            .composite(device_context, current_frame, dirty_rects)?;
375        // clone = AddRef，owned 版本独立于 self，后续 mut borrow self.encoder() 不冲突
376        Ok(self
377            .compositor
378            .as_ref()
379            .expect("session.composite() called while released")
380            .current_bgra()
381            .clone())
382    }

404    /// 轻量检查 D3D device 是否还活着。
405    ///
406    /// `GetDeviceRemovedReason` 在 device 正常时返回 S_OK，是 O(1) 调用。
407    /// 在 device lost 后返回相应错误码。
408    ///
409    /// **不调用太频繁**：每帧调一次有点浪费（虽然便宜），
410    /// 推荐每 N 帧（建议 60 ~= 1 秒）调一次。
411    #[allow(dead_code)] // 预留：目前靠错误驱动 recover，没主动轮询
412    pub fn check_device_health(&self) -> Result<(), SessionError> {
413        let device: &ID3D11Device = self
414            .dup
415            .as_ref()
416            .expect("session.check_device_health() called while released")
417            .device();
418        // GetDeviceRemovedReason 返回 HRESULT：S_OK = 健康
419        match unsafe { device.GetDeviceRemovedReason() } {
420            Ok(()) => Ok(()),
421            Err(e) => Err(SessionError::classify(&e)),
422        }
423    }

430/// 失败后的下一步动作。
431///
432/// [H4 修复后] 删了 `RetryImmediate` 变体：
433///   - 原来第 1-2 次失败立即重试（无退避），GPU 抖动时立刻重试可能压垮驱动
434///   - 现在 `record_failure` 永远返回 `Backoff`，最少 50ms 起步
435///   - 3 个 match 分支里的 `RecoverAction::RetryImmediate => { /* 直接重试 */ }` 也一并清理
436#[derive(Debug, Clone, Copy, PartialEq, Eq)]
437pub enum RecoverAction {
438    /// 退避 N 毫秒后重试（避免连续失败打爆 GPU 驱动）
439    Backoff(u64),
440    /// 放弃，shutdown 主循环
441    GiveUp,
442}
443
444/// 失败节流状态机。
445///
446/// 规则：
447/// - 单次失败：立即重试（让驱动 / GPU 抖动自己恢复）
448/// - 连续 3 次失败：开始退避 100/200/300ms...
449/// - 30 秒窗口内 5 次失败：放弃（说明环境真的坏了）
450///
451/// 计数器只在失败时递增，成功一次清零。
452pub struct RecoverState {
453    consecutive_failures: u32,
454    last_failure_at: Option<Instant>,
455    /// 累计成功 recover 次数（只统计成功的）。atomic 是为了让外部 log 线程也能读。
456    total_recoveries: AtomicU64,
457}
458
459impl RecoverState {
460    pub fn new() -> Self {
461        Self {
462            consecutive_failures: 0,
463            last_failure_at: None,
464            total_recoveries: AtomicU64::new(0),
465        }
466    }
467
468    /// 记录一次失败，返回下一步动作。
469    ///
470    /// [H4 修复] 3 个 bug：
471    ///   1. 原 `last_failure_at = Some(now)` 在 GiveUp 检查**之后**才更新 → GiveUp 后下一次失败 `last`
472    ///      还是上次的时间戳，窗口永远从第 1 次失败开始算。改为**每次失败无条件刷新**
473    ///   2. 第 1-2 次失败立即重试（没退避）→ GPU 抖动时立刻重试可能压垮驱动。
474    ///      改为**每次失败至少 50ms 退避**（=consecutive_failures * 50ms 起步）
475    ///   3. 文档/注释说"30 秒/5 次"，但**第一次失败的 last=None** 时 GiveUp 检查
476    ///      `if let Some(last) = ...` 直接跳过，永远不会 GiveUp（counts 不影响）
477    ///      修法：last=None 时 elapsed_since_last = u64::MAX（远大于 30），自然走完
478    pub fn record_failure(&mut self) -> RecoverAction {
479        self.consecutive_failures += 1;
480        let now = Instant::now();
481        let elapsed_since_last = self
482            .last_failure_at
483            .map(|t| now.duration_since(t).as_secs())
484            .unwrap_or(u64::MAX);
485        // [H4.1] 每次失败无条件刷新时间戳（窗口从最近一次失败开始算）
486        self.last_failure_at = Some(now);
487
488        // [H4.3] 30 秒内累计 5 次失败 → 放弃（last=None 走 u64::MAX 路径，不会 GiveUp）
489        if elapsed_since_last < 30 && self.consecutive_failures >= 5 {
490            return RecoverAction::GiveUp;
491        }
492        // [H4.2] 每次失败至少 50ms 退避（之前第 1-2 次是立即重试，抖动时压垮驱动）
493        let backoff_ms = (self.consecutive_failures as u64) * 50;
494        RecoverAction::Backoff(backoff_ms)
495    }
496
497    /// 记录一次成功（recover 成功）。
498    pub fn record_success(&mut self) {
499        self.consecutive_failures = 0;
500        self.total_recoveries.fetch_add(1, Ordering::Relaxed);
501    }
502
503    pub fn total_recoveries(&self) -> u64 {
504        self.total_recoveries.load(Ordering::Relaxed)
505    }
506
507    /// 当前连续失败次数（给日志用）
508    pub fn consecutive_failures(&self) -> u32 {
509        self.consecutive_failures
510    }
511}
512
513impl Default for RecoverState {
514    fn default() -> Self {
515        Self::new()
516    }
517}
