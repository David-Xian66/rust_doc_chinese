1//! 脏矩形（Dirty Rects）提取
2//!
3//! DXGI Desktop Duplication API 有一个非常重要的能力：
4//! 当一帧到来时，它能告诉你"屏幕上哪几个矩形区域和上一帧不同"。
5//! 这对远程控制软件意义重大 —— 只对变化区域编码，CPU 和带宽都能省一大截。
6//!
7//! 本模块只负责把脏矩形从 DXGI 里读出来，不做任何后续处理。
8
9use std::cell::RefCell;
10use windows::Win32::Foundation::RECT;
11use windows_capture::dxgi_duplication_api::DxgiDuplicationFrame;
12
13/// [D1 准备] 计算脏矩形总面积（像素数）。
14///
15/// 注意：
16/// - RECT 是半开区间 [left, top, right, bottom)，面积 = (right - left) * (bottom - top)
17/// - DXGI 不保证脏矩形互斥 —— 重叠区域会被 sum 重复计算。
18///   这是保守估计：偏大的面积 → 偏激进的"跳过"判断（宁愿少编一帧也别浪费带宽）。
19///   V1 不做去重，性价比不高。
20pub fn dirty_total_area(dirty_rects: &[RECT]) -> u64 {
21    dirty_rects
22        .iter()
23        .map(|r| {
24            let w = (r.right - r.left).max(0) as u64;
25            let h = (r.bottom - r.top).max(0) as u64;
26            w * h
27        })
28        .sum()
29}

31// 线程局部的 RECT 缓冲区（每帧复用，避开 `Vec::with_capacity` + `resize` 的零填充开销）。
32//
33// 之所以能缓存：`RECT` 是 16 字节的 POD（4×i32），drop 时不释放任何资源，单纯是堆内存。
34// 线程结束（或进程退出）时由 `thread_local!` 的析构自动释放。
35//
36// 注：这里不能写 `///`（rustdoc 文档注释），因为 `thread_local!` 是宏，rustdoc 不会为宏
37//     调用生成文档，`///` 会被识别成未使用的文档注释并触发 warning。普通 `//` 注释无此问题。
38thread_local! {
39    static DIRTY_RECT_BUF: RefCell<Vec<RECT>> = RefCell::new(Vec::new());
40}
41
42/// 从一帧 DXGI 桌面复制结果中提取脏矩形列表（屏幕坐标系，左上为原点）。
43///
44/// # DXGI 内部机制
45/// `IDXGIOutputDuplication::GetFrameDirtyRects` 是 DXGI 1.2 提供的接口。
46/// 调用方需要：
47///   1. 准备一个**足够大**的缓冲区（按 `TotalMetadataBufferSize` 推算最大矩形数）；
48///   2. 传入缓冲区的字节数 + 裸指针；
49///   3. DXGI 把脏矩形写进缓冲区，并通过 `required_size` 告诉你实际写了多少字节。
50///
51/// 返回 `Vec<RECT>`，每个 RECT = (left, top, right, bottom)。
52pub fn extract_dirty_reacts(frame: &DxgiDuplicationFrame) -> Vec<RECT> {
53    // `frame_info()` 返回的是 DXGI 的 `DXGI_FRAME_INFO` 结构体，
54    // 这里只关心 `TotalMetadataBufferSize`：DXGI 告诉你"这一帧的元数据最多占多少字节"。
55    let frame_info = frame.frame_info();
56    if frame_info.TotalMetadataBufferSize == 0 {
57        // 没有元数据 = 桌面和上一帧完全相同，连一个脏矩形都没有
58        return Vec::new();
59    }
60
61    // `duplication()` 是当前帧背后那个 `IDXGIOutputDuplication` 接口的封装，
62    // 它的 `GetFrameDirtyRects` 是真正的 Win32 调用入口
63    let duplication = frame.duplication();
64
65    // 用 RECT 的大小（16 字节）反推这个缓冲区最多能装多少个矩形
66    let rect_size = std::mem::size_of::<RECT>();
67    let max_rect_count = (frame_info.TotalMetadataBufferSize as usize) / rect_size;
68
69    if max_rect_count == 0 {
70        return Vec::new();
71    }
72
73    // [修复 §3.1] 用线程局部缓存的 Vec 替换每帧 `Vec::with_capacity` + `resize`：
74    //   - `clear()` 把 len 清零但保留已分配的容量
75    //   - `reserve()` 在容量不足时按需扩容（策略与原版一致：够装 max_rect_count）
76    //   - `unsafe { set_len(max_rect_count) }` 跳过 `resize` 的零填充（DXGI 会立刻写入）
77    //   - 最后 `clone()` 把结果 Vec 拷出去（不能 move out of RefCell；RECT 是 16B POD，开销可忽略）
78    DIRTY_RECT_BUF.with(|buf| {
79        let mut buf = buf.borrow_mut();
80        buf.clear();
81        if buf.capacity() < max_rect_count {
82            // 先算好要补的量，避免 reserve + len 同时借用 buf
83            let additional = max_rect_count - buf.len();
84            buf.reserve(additional);
85        }
86        // SAFETY: 我们刚刚 reserve 了足够容量，set_len 之后紧接着 GetFrameDirtyRects
87        // 会写入完整的 `max_rect_count * 16` 字节；如果它失败，下面的 truncate(0)
88        // 会把 len 归零，所以哪怕未初始化也无所谓（不会有人读到）。
89        unsafe { buf.set_len(max_rect_count) };
90
91        let mut required_size = 0u32;
92        // [M8 修复] 检测 GetFrameDirtyRects 失败 → 回退到全屏 rect（不要静默返回空 Vec）：
93        //   旧实现用 `let _ = ...` 静默吃掉错误。
94        //   GetFrameDirtyRects 失败时如果返回空 Vec → 上层 capture.rs 看到 dirty_rects.is_empty()
95        //   → 走 `continue` 跳过这一帧 → 屏幕有变化但编码器没收到任何输入 → 静默卡死。
96        //   修复：失败时构造一个覆盖整个屏幕的 rect（保守但保正确性）。
97        let rects_result = unsafe {
98            duplication.GetFrameDirtyRects(
99                (max_rect_count * rect_size) as u32,
100                buf.as_mut_ptr(),
101                &mut required_size,
102            )
103        };
104        if rects_result.is_err() {
105            // 失败 → 用 texture 尺寸构造全屏 rect
106            let desc = frame.texture_desc();
107            // [release-log] 偶尔失败打 warn，避免刷屏
108            #[cfg(any(debug_assertions, feature = "release-log"))]
109            {
110                use std::sync::Mutex;
111                use std::time::Instant;
112                static LAST_LOG: Mutex<Option<Instant>> = Mutex::new(None);
113                let mut last = LAST_LOG.lock().unwrap_or_else(|p| p.into_inner());
114                let should_log = match *last {
115                    None => true,
116                    Some(t) => t.elapsed().as_secs() >= 5,
117                };
118                if should_log {
119                    *last = Some(Instant::now());
120                    eprintln!(
121                        "⚠️ [dirty] GetFrameDirtyRects 失败 → 回退全屏 rect ({}x{})",
122                        desc.Width, desc.Height
123                    );
124                }
125            }
126            return vec![RECT {
127                left: 0,
128                top: 0,
129                right: desc.Width as i32,
130                bottom: desc.Height as i32,
131            }];
132        }
133
134        // 根据 DXGI 实际写的字节数，截掉多余的元素
135        let actual_count = (required_size as usize) / rect_size;
136        buf.truncate(actual_count);
137
138        // 拷贝返回（RefCell 不能 move out）
139        buf.clone()
140    })
141}
