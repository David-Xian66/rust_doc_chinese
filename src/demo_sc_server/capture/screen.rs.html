1//! 显示器枚举与信息收集
2//!
3//! 这里不直接碰 COM/DXGI，而是封装了 `windows-capture` 这个 crate 的 `Monitor` API。
4//! `windows-capture` 内部通过 DXGI Output / Direct3D 拿到每个显示器的属性，
5//! 对外只暴露纯 Rust 的 `Monitor` 结构体和方法。
6//!
7//! 本模块只关心一件事：列出当前机器上所有的物理显示器，并标记哪个是主屏。
8
9use std::collections::HashMap;
10use windows_capture::monitor::Monitor;
11
12/// 单个显示器的对外信息（不持有任何 COM 对象，纯数据，方便跨线程传）
13pub struct ScreenData {
14    /// 自增序号，从 1 开始，作为本程序内部的"屏幕索引"。
15    /// [2.7 修复说明] 这个 ID 直接传给 `windows-capture` crate 的 `Monitor::from_index()`，
16    ///   那个函数期望 1-based index（内部用 `index - 1` 索引 Vec，`index < 1` 直接拒绝）。
17    ///   所以这里必须保持 1-based，否则 `Monitor::from_index(0)` 会报 `IndexIsLowerThanOne`。
18    pub(crate) id: usize,
19    /// 显示器友好名称（如 "DELL U2719D"）
20    pub(crate) name: String,
21    /// 系统给显示器的内部设备名（如 `\\.\DISPLAY1`），用于唯一标识
22    pub(crate) device_name: String,
23    /// 刷新率（Hz），用于和目标 FPS 取较小值，避免无效帧
24    pub(crate) refresh_rate: u32,
25    /// 是否是主显示器
26    pub(crate) main_screen: bool
27}
28
29/// 获取主显示器的 device_name。
30///
31/// `Monitor::primary()` 内部通过 DXGI 枚举适配器 + 输出，
32/// 找到 GDI 主屏幕对应的那个 `Monitor`。
33/// `device_name()` 返回它的内部设备名（`\\.\DISPLAY1` 这种）。
34fn screen_main() -> Result<String, windows_capture::monitor::Error>{
35    let monitors = Monitor::primary()?;
36    Ok(monitors.device_name()?)
37}
38
39/// 枚举所有显示器，返回以"本程序内部序号（从 1 开始）"为键的 HashMap。
40///
41/// 注意：这里返回的 `id` **不是** DXGI 的 AdapterIndex，也不是 Windows 的 GDI 序号，
42/// 而是一个临时分配的递增序号，仅供本程序内部使用。
43/// 真正传给 DXGI Desktop Duplication 时，需要再用 `Monitor::from_index(...)` 反查。
44///
45/// [2.7 修复说明] 历史上 ID 是 1-based，配合 `windows-capture` crate 的 `Monitor::from_index()`
46///   语义（内部用 `index - 1`，且 `index < 1` 直接拒绝）。改成 0-based 会让
47///   `Monitor::from_index(0)` 报 `IndexIsLowerThanOne` 错误。
48///   所以保持 1-based，但需要修 main.rs 的 bounds check（之前用 `0..len-1`）。
49pub fn screen_list_all() -> Result<HashMap<usize, ScreenData>, windows_capture::monitor::Error>{
50    // 枚举所有物理显示器。`Monitor::enumerate` 内部走 DXGI EnumOutputs
51    let monitors = Monitor::enumerate()?;
52
53    // 先拿到主屏的 device_name，后面用来比对
54    let main_index = screen_main()?;
55
56    let mut hashmap_: HashMap<usize, ScreenData> = HashMap::new();
57    let mut index_: usize = 0;
58
59    // `Monitor` 内部持有 DXGI Output 的引用，所以这里遍历时它一直活着
60    for i in monitors {
61        index_ += 1;  // 1-based：和 Monitor::from_index() 的 1-based 语义对齐
62
63        // 取一次 device_name 既用来比对主屏，又用来存到结果里
64        let device_name = i.device_name()?;
65        let main_screen_ = {device_name == main_index};   // 判断是不是主屏幕
66
67        hashmap_.insert(index_, ScreenData{
68            id: index_,
69            name: i.name()?,
70            device_name: i.device_name()?,
71            refresh_rate: i.refresh_rate()?,
72            main_screen: main_screen_
73        });
74    }
75    Ok(hashmap_)
76}
