1//! 服务端入口
2//!
3//! 整个程序只做三件事：
4//!   1. 列出当前机器的所有显示器
5//!   2. 让用户选一个（demo 阶段硬编码 index 1）
6//!   3. 把目标帧率裁到屏幕实际刷新率以内，进入捕获 + 编码主循环
7//!
8//! Media Foundation / DXGI / D3D11 的初始化都在 capture::capture_main 内部。
9//!
10//! 错误处理（[E3]）：`main` 是 Result 风格，错误用 `?` 传播，统一在 `print_error`
11//! 打印错误链 + 调试提示 + 友好提示（按错误码给针对性建议）。
12//! 退出码：0 = 正常退出，非 0 = 异常退出。
13//!
14//! 信号处理（[C1]）：注册 Ctrl+C handler，第一次按设 shutdown flag 触发优雅退出，
15//! 第二次按（5 秒内）立即强制退出。
16

39/// 把 Mbps 转成 bps 的常量函数。
40/// 用法：`const BITRATE: u32 = mbps(8);` —— 一眼能看出"8 Mbps"，
41/// 不必读者心算 `8_000_000` 是几个零。
42const fn mbps(x: u32) -> u32 {
43    x * 1_000_000
44}
45
46/// 编码器目标平均码率（demo 阶段写死，不暴露给用户调节）。
47///
48/// 8 Mbps 是 1080p 桌面共享（带文字）的合理上限：
49///   - 文字可读、窗口边框清晰
50///   - 30 fps 下 8 Mbps 留给控制端的解码端余量充足
51///   - 局域网单流占带宽 < 4 MB/s
52///   - 调到 4 Mbps 可视场景：纯视频（无密集文字）/ 带宽紧张
53///   - 调到 12+ Mbps：4K / 高帧率 / 极致清晰度
54const TARGET_BITRATE_BPS: u32 = mbps(8);
55
56/// [SLICE] H.264 Slice 切分配置（写死到 main，暂不暴露 CLI）。
57///
58/// 字段含义（详见 `H264SliceConfig` 文档 / `H264_SLICE_PARAMS.md`）：
59/// - `mode = MbBased`:        按宏块切（H.264 标准 Slice 切分模式）
60/// - `control_mode = MbRows`: "count" 字段按 MB 行数解读
61/// - `count = 16`:            每片 16 MB 行
62///   → 1080p 屏幕（120 MB 行 / 帧）= 7.5 片/帧
63///   → 1440p（170 MB 行）= 10.6 片/帧
64///   → 4K（240 MB 行）= 15 片/帧
65/// - `max_bytes / max_count`: 故意不开（许多 GPU 不支持，按"硬错 + 继续跑"策略会
66///   在 log 里看到 `⚠️ [slice] SetValue(...) 失败`，不影响编码器继续运行）。
67///   调试时手动开 → 把 None 改成 Some(N) 即可。
68const SLICE_CONFIG: H264SliceConfig = H264SliceConfig {
69    mode: Some(SliceMode::MbBased),
70    control_mode: Some(SliceControlUnit::MbRows),
71    count: Some(16),
72    max_bytes: None,
73    max_count: None,
74};

82fn main() -> ExitCode {
83    match run_server() {
84        Ok(()) => {
85            println!("👋 服务端正常退出");
86            ExitCode::SUCCESS
87        }
88        Err(e) => {
89            print_error(&*e);  // Box<dyn Error> deref 到 dyn Error
90            ExitCode::FAILURE
91        }
92    }
93}

95/// 实际的服务端逻辑：注册 Ctrl+C handler → 枚举屏幕 → 启动主循环。
96/// 错误用 `?` 传播，调用方（main）统一打印。
97fn run_server() -> Result<(), Box<dyn Error>> {
98    // [C1] 注册 Ctrl+C handler
99    let shutdown = Arc::new(AtomicBool::new(false));
100    let ctrl_c_count = Arc::new(AtomicI32::new(0));
101    {
102        let s = shutdown.clone();
103        let c = ctrl_c_count.clone();
104        ctrlc::set_handler(move || {
105            let n = c.fetch_add(1, Ordering::SeqCst);
106            if n >= 1 {
107                // 第二次按 Ctrl+C：立即强退，不等 cleanup
108                eprintln!("\n⚠️ 第二次 Ctrl+C, 强制退出");
109                std::process::exit(2);
110            }
111            eprintln!("\n⚠️ 收到 Ctrl+C, 正在优雅退出...");
112            s.store(true, Ordering::SeqCst);
113        })
114        .map_err(|e| format!("无法注册 Ctrl+C handler: {}", e))?;
115    }
116
117    // 列出所有屏幕（内部走 DXGI EnumOutputs）
118    let list_ = screen::screen_list_all()?;
119    println!("存在以下显示屏");
120    for a in &list_ {
121        println!("是否为主屏幕: {}, id: {}, name: {}, device name: {},FPS: {}",
122            &a.1.main_screen, &a.1.id, &a.1.name, &a.1.device_name, &a.1.refresh_rate);
123    }
124
125    // if list_.len() == 1 {
126    // TODO: 这里要可以选择屏幕，因为demo，所以先不写
127    // }
128
129    // [B3] 解析 CLI 参数（手写 std::env::args，避免 clap 依赖）
130    let config = parse_args()?;
131
132    // [2.7 修复] CLI 范围校验：
133    //   screen_list_all 返回的 ID 是 1-based（和 windows-capture crate 的 Monitor::from_index 语义对齐，
134    //   那个函数内部用 index-1 索引 Vec，且 index < 1 直接拒绝）。
135    //   之前的校验用 `config.screen >= list_.len()` + `id 范围 0..{}` 是 0-based 假设 → 用户报"id 1 越界"。
136    //   修正：
137    //     - 单屏幕时 id 范围是 1..1（合法只有 1）
138    //     - N 屏幕时 id 范围是 1..N
139    //     - 越界条件：config.screen < 1 || config.screen > list_.len()
140    if config.screen < 1 || config.screen > list_.len() {
141        return Err(format!(
142            "--screen {} 越界: 系统中只有 {} 块屏幕 (id 范围 1..={})",
143            config.screen,
144            list_.len(),
145            list_.len()
146        ).into());
147    }
148    if config.fps == 0 {
149        return Err("--fps 必须 > 0 (合法: 1..=屏幕刷新率)".into());
150    }
151    if config.fps > 1000 {
152        return Err(format!("--fps {} 超过硬上限 1000 (防止误填)", config.fps).into());
153    }
154
155    // demo 阶段直接硬编码选第 1 块屏幕
156    let screen_choose = config.screen;
157    let fps_choose_hope: u32 = config.fps;
158
159    // 屏幕刷新率作为上限，避免请求比屏幕能给的更高的帧率
160    let fps_screen = list_[&screen_choose].refresh_rate;
161
162    // 取用户期望值和屏幕刷新率的较小者
163    let fps_choose = min(fps_choose_hope, fps_screen);
164
165    // 启动捕获 + 编码主循环
166    // 内部会调 CoInitializeEx、MFStartup、创建 D3D11 设备等
167    println!(
168        "🚀 启动服务端: 屏幕 {} ({}), 目标 {} fps, 码率 {} bps ({} Mbps), Profile={}",
169        screen_choose,
170        list_[&screen_choose].device_name,
171        fps_choose,
172        TARGET_BITRATE_BPS,
173        TARGET_BITRATE_BPS / 1_000_000,
174        config.profile.as_str(),
175    );
176    println!("按 Ctrl+C 优雅退出（双击 Ctrl+C 强制终止）");
177
178    // [L1] 启动锁屏监听器。
179    //   失败时打日志但不阻塞程序启动（捕获功能仍然可用，只是锁屏时会浪费 CPU）。
180    //   默认不启用 user switch pause（UAC 弹窗 / 用户注销场景较少）。
181    let lockscreen = match LockscreenWatcher::new(config.enable_user_switch_pause) {
182        Ok(w) => {
183            println!("🔐 [L1] 锁屏监听器已启动 (会话状态变化时自动暂停 + 释放 session)");
184            Some(w)
185        }
186        Err(e) => {
187            eprintln!("⚠️ [L1] 锁屏监听器启动失败 ({}), 锁屏时会浪费 CPU 但不影响基本捕获", e);
188            None
189        }
190    };
191
192    let _server = ServerInfo {
193
194        ip: IpAddr::V4(Ipv4Addr::new(127, 0, 0, 1)),
195        port: 8080,
196    };
197    let host_ = SocketAddr::new(_server.ip, _server.port);
198
199
200    println!("开始配置网络线程");
201    let server_certs = quic_cert("server_cert.crt")?;     // 返回 Vec<CertificateDer<'a>>
202    let mut roots = quinn::rustls::RootCertStore::empty();
203    roots.add(server_certs[0].clone())?;
204
205    let rustls_crypto = quinn::rustls::ClientConfig::builder()
206        .with_root_certificates(roots)
207        .with_no_client_auth();
208
209    let quic_crypto = QuicClientConfig::try_from(rustls_crypto)?;
210
211    let client_config = quinn::ClientConfig::new(Arc::new(quic_crypto));
212
213    let (tx, mut rx) = mpsc::channel::<EncodedFrame>(frame_buffer_for(60, &fps_choose));
214
215    let quic_thread = thread::spawn(move || {
216        let rt = tokio::runtime::Builder::new_multi_thread()
217            .worker_threads(8)  // 8个工作线程
218            .enable_io()        // 可在runtime中使用异步IO
219            .enable_time()      // 可在runtime中使用异步计时器(timer)
220            .build().expect("// TODO: 进程创建错误处理");
221
222        let mut endpoint = Endpoint::client("[::]:0".parse().unwrap()).expect("// TODO: 进程创建错误处理");  // 哪怕是quic的客户端，是要独占一个端口的，:0是让系统自己选
223        endpoint.set_default_client_config(client_config);
224
225        let endpoint_async = endpoint.connect(
226            host_,
227            "torchol.com"  // TODO: 这个后期要改,最好做到每个被控端走单独的
228        ).expect("// TODO: 进程创建错误处理");
229
230
231        rt.spawn(async move {
232            let conn = endpoint_async.await.expect("// TODO: 进程创建错误处理");
233            let max_datagram_size: usize = conn.max_datagram_size().expect("// TODO: 进程创建错误处理");
234            while let Some(res) = rx.recv().await {
235                // 发送数据
236                // conn.send_datagram()
237                let data_size = res.annex_b.len();
238                let mut quicdata = QuicSendData::new(
239                    res.annex_b,
240                    Vec::new(),
241                    res.keyframe_index,
242                    res.delta_index,
243                    max_datagram_size,
244                    data_size,
245                    conn.clone(),
246                    res.is_keyframe
247                );
248                quicdata.data_slices();
249                quicdata.send().await;
250            }
251        })
252    });
253
254
255
256    capture::capture::capture_main(
257        screen_choose,
258        fps_choose,
259        fps_screen,
260        TARGET_BITRATE_BPS,
261        config.profile,
262        SLICE_CONFIG,
263        shutdown,
264        lockscreen,
265        tx
266    )?;
267    Ok(())
268}

270/// [B3] 服务端配置（CLI 参数聚合）
271struct ServerConfig {
272    /// 屏幕索引（1-based，默认 1，对应 `screen_list_all` 的 HashMap key）
273    screen: usize,
274    /// 目标 FPS（默认 130，会和屏幕刷新率取较小值）
275    fps: u32,
276    /// H.264 Profile（默认 High）
277    profile: H264Profile,
278    /// [L1] 是否在用户注销 / 切换时也暂停（默认 false，避免 UAC 弹窗误触发）
279    enable_user_switch_pause: bool,
280}
281
282impl Default for ServerConfig {
283    fn default() -> Self {
284        Self {
285            screen: 1,
286            fps: 130,
287            profile: H264Profile::default(),
288            enable_user_switch_pause: false,
289        }
290    }
291}

293/// [B3] 解析命令行参数。
294///
295/// 支持：
296///   -s, --screen <N>        屏幕索引
297///   -f, --fps <N>           目标 FPS
298///   -p, --profile <name>    H.264 profile (high/main/baseline)
299///   -u, --user-switch-pause 启用用户切换暂停（L1，默认 false）
300///   -h, --help              打印帮助
301fn parse_args() -> Result<ServerConfig, Box<dyn Error>> {
302    let mut config = ServerConfig::default();
303    let args: Vec<String> = std::env::args().collect();
304    let mut i = 1;
305    while i < args.len() {
306        match args[i].as_str() {
307            "--screen" | "-s" => {
308                config.screen = args.get(i + 1)
309                    .ok_or_else(|| format!("缺少 --screen 参数值"))?
310                    .parse()
311                    .map_err(|e| format!("--screen 参数值非法: {}", e))?;
312                i += 2;
313            }
314            "--fps" | "-f" => {
315                config.fps = args.get(i + 1)
316                    .ok_or_else(|| format!("缺少 --fps 参数值"))?
317                    .parse()
318                    .map_err(|e| format!("--fps 参数值非法: {}", e))?;
319                i += 2;
320            }
321            "--profile" | "-p" => {
322                let val = args.get(i + 1)
323                    .ok_or_else(|| format!("缺少 --profile 参数值"))?;
324                config.profile = H264Profile::from_str(val)
325                    .ok_or_else(|| format!("未知 --profile 值 '{}' (合法: high|main|baseline)", val))?;
326                i += 2;
327            }
328            "--user-switch-pause" | "-u" => {
329                config.enable_user_switch_pause = true;
330                i += 1;
331            }
332            "--help" | "-h" => {
333                print_help();
334                std::process::exit(0);
335            }
336            other => {
337                return Err(format!("未知参数: {}", other).into());
338            }
339        }
340    }
341    Ok(config)
342}

355/// 打印错误链 + 调试提示 + 友好提示。
356///
357/// 错误链最多打 5 层（避免循环引用或过深的 wrap 撑爆终端）。
358fn print_error(e: &dyn Error) {
359    eprintln!();
360    eprintln!("❌ 服务端异常退出");
361    eprintln!("   原因: {}", e);
362
363    // 错误链：递归打印 cause chain
364    let mut source = e.source();
365    let mut depth = 1;
366    while let Some(s) = source {
367        eprintln!("   {}↳ {}", "  ".repeat(depth), s);
368        source = s.source();
369        depth += 1;
370        if depth > 5 {
371            eprintln!("   {}↳ ... (链太长，截断)", "  ".repeat(depth));
372            break;
373        }
374    }
375
376    // 通用调试提示
377    eprintln!();
378    eprintln!("💡 调试提示:");
379    eprintln!("   - 用 release-log 重跑: cargo build --release --features release-log");
380    eprintln!("   - 确认显卡驱动是最新版");
381    eprintln!("   - 确认 Windows 版本 >= 1903 (DXGI 1.4 要求)");
382    print_friendly_hints(e);
383}

385/// 按错误消息里的关键字打印针对性提示。
386///
387/// 字符串匹配是脆弱的（V1 可接受），V2 可以走结构化错误（custom Error enum）。
388fn print_friendly_hints(e: &dyn Error) {
389    // 把整个错误链展平成一个字符串，方便 substring 匹配
390    let mut chain_str = e.to_string();
391    let mut src = e.source();
392    while let Some(x) = src {
393        chain_str.push_str(&format!(" | {}", x));
394        src = x.source();
395    }
396
397    if chain_str.contains("DXGI_ERROR_ACCESS_LOST") || chain_str.contains("0x887A0006") {
398        eprintln!("   - 看到 DXGI_ERROR_ACCESS_LOST？通常是锁屏 / 用户切换导致，正常现象");
399    }
400    if chain_str.contains("DXGI_ERROR_DEVICE_REMOVED") || chain_str.contains("0x887A0005") {
401        eprintln!("   - 看到 DXGI_ERROR_DEVICE_REMOVED？显卡驱动可能崩溃，重启可恢复");
402    }
403    if chain_str.contains("E_OUTOFMEMORY") || chain_str.contains("0x8007000E") {
404        eprintln!("   - 看到 E_OUTOFMEMORY？GPU 显存不足，关掉其他吃显存的程序");
405    }
406    if chain_str.contains("CLSID_MSH264EncoderMFT") || chain_str.contains("0xC00D5212") {
407        eprintln!("   - 找不到编码器？安装 Windows Media Feature: 控制面板 → 程序 → 启用或关闭 Windows 功能");
408    }
