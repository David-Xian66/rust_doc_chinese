1//! # demo_sc_proto
2//!
3//! 服务端 ↔ 客户端共享的**纯数据**协议层 crate。
4//!
5//! ## 范围
6//!
7//! - **包含**：纯数据结构 + 序列化函数（不依赖 Windows / MF / tokio / QUIC）
8//! - **不包含**：网络层抽象。demo 阶段 server / client 在各自 `network/quic.rs`
9//!   里直接定义 `QuicSendData` / `QuicDataFragment` 和 wire-format header，
10//!   暂不引入 transport trait。等协议稳定再上提到本 crate。
11//!
12//! ## WireFrame
13//!
14//! 服务端 capture loop 编出一帧 H.264，构造一个 `WireFrame` 推到
15//! `session.frame_buffer`（当前类型 `VecDeque<WireFrame>`）。后续网络层接走后
16//! 会替换为 `QuicSendData`（见 `crates/server/src/network/quic.rs`）；本结构
17//! 暂留作单元测试和占位通路使用。
18//!
19//! ## WireFormat
20//!
21//! 服务端字节流格式（22 字节头 + payload）：
22//!
23//! ```text
24//! +--------+-----------------+----------+----------------------+----------------------+-------------+
25//! |  u32 LE |  u8             |  u8      |  i64 LE              |  u64 LE              | payload ... |
26//! | length |  is_keyframe    |  reserved|  timestamp_hns       |  frame_index         |             |
27//! +--------+-----------------+----------+----------------------+----------------------+-------------+
28//!   4 bytes   1 byte            1 byte     8 bytes                8 bytes                N bytes
29//! ```
30//!
31//! - `length`：整个包的总字节数（含头和 payload），方便客户端先读 4 字节再 alloc
32//! - `is_keyframe`：NALU type == 5（IDR）；客户端据此决定是否需要"清空参考帧缓存"
33//! - `reserved`：留作未来协议扩展，**当前必须为 0**
34//! - `timestamp_hns`：MF HNS（100ns 单位）；客户端可换算为 PTS（毫秒 = hns / 10_000）
35//! - `frame_index`：0..N 自增；客户端用来检测丢帧 / 重排序
36//! - `payload`：Annex-B 格式 H.264 字节流：
37//!     - 每个 NALU 前有 `00 00 00 01` 起始码（4 字节）
38//!     - IDR 帧前自动注入 SPS NALU + PPS NALU（也是 4 字节起始码）
39

42#[derive(Debug, Clone, PartialEq, Eq)]
43pub struct WireFrame {
44    /// 帧序号：0..N 自增。
45    ///
46    /// 服务端在 capture loop 里递增（`frame_count: u64`）。
47    /// 客户端可以用来：
48    ///   - 检测丢帧（收到的 index 不连续）
49    ///   - 检测重排序（罕见，但 TCP 通常不会）
50    pub frame_index: u64,
51
52    /// 时间戳（MF HNS，100ns 单位）。
53    ///
54    /// 服务端在 `MFCreateSample` 时设到 sample 上的 `MF_PT_SAMPLE_TIMESTAMP`。
55    /// 客户端换算为毫秒：`timestamp_ms = timestamp_hns / 10_000`。
56    pub timestamp_hns: i64,
57
58    /// 是否关键帧（IDR）。
59    ///
60    /// **判定**：NALU header byte（payload 第一个字节）& 0x1f == 5。
61    /// 服务端在 `avcc_to_annex_b` 时已经检测过，存到本字段直接传。
62    ///
63    /// 客户端可以用来：
64    ///   - 解码器初始化（第一个 IDR）
65    ///   - 解码器 recover（中途收到 IDR 时清参考帧缓存）
66    ///   - UI 显示"正在缓冲..."
67    pub is_keyframe: bool,
68
69    /// payload 字节数（冗余字段，方便客户端先看长度再 alloc）。
70    pub payload_len: u32,
71
72    /// Annex-B 格式 H.264 字节流。
73    ///
74    /// 已被 `avcc_to_annex_b` 转好：
75    ///   - 每个 NALU 前 4 字节 `00 00 00 01` 起始码
76    ///   - IDR 帧前自动注入 SPS NALU + PPS NALU（`sps_pps_annex_b`）
77    ///
78    /// 客户端可以用 ffmpeg / libavcodec / 浏览器 MSE / VideoToolbox 直接解码。
79    pub payload: Vec<u8>,
80}
81
82impl WireFrame {
83    /// 序列化为字节流（22 字节头 + payload）。
84    ///
85    /// **仅用于 wire-format 单元测试**。**热路径不要用**——
86    /// 热路径直接拿 `payload: Vec<u8>` 发 QUIC，不要走 encode。
87    ///
88    /// 格式见 module 文档。
89    pub fn encode(&self) -> Vec<u8> {
90        let total_len = (Self::HEADER_LEN + self.payload.len()) as u32;
91        let mut out = Vec::with_capacity(total_len as usize);
92        out.extend_from_slice(&total_len.to_le_bytes()); // 4 bytes
93        out.push(if self.is_keyframe { 1 } else { 0 }); // 1 byte
94        out.push(0u8); // reserved, must be 0
95        out.extend_from_slice(&self.timestamp_hns.to_le_bytes()); // 8 bytes
96        out.extend_from_slice(&self.frame_index.to_le_bytes()); // 8 bytes
97        out.extend_from_slice(&self.payload); // N bytes
98        out
99    }
100
101    /// 从字节流反序列化（供客户端测试 / mock 用）。
102    ///
103    /// 当前未使用，等客户端实际解码时再调。
104    /// 失败返回 `Err(String)`：长度不足 / 长度字段不一致。
105    pub fn decode(bytes: &[u8]) -> Result<Self, String> {
106        if bytes.len() < Self::HEADER_LEN {
107            return Err(format!("bytes 长度 {} < HEADER_LEN {}", bytes.len(), Self::HEADER_LEN));
108        }
109        let total_len = u32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]) as usize;
110        if bytes.len() != total_len {
111            return Err(format!("声明长度 {} != 实际 {}", total_len, bytes.len()));
112        }
113        let is_keyframe = bytes[4] != 0;
114        let _reserved = bytes[5];
115        let timestamp_hns = i64::from_le_bytes([
116            bytes[6], bytes[7], bytes[8], bytes[9],
117            bytes[10], bytes[11], bytes[12], bytes[13],
118        ]);
119        let frame_index = u64::from_le_bytes([
120            bytes[14], bytes[15], bytes[16], bytes[17],
121            bytes[18], bytes[19], bytes[20], bytes[21],
122        ]);
123        let payload = bytes[Self::HEADER_LEN..].to_vec();
124        let payload_len = payload.len() as u32;
125
126        Ok(Self {
127            frame_index,
128            timestamp_hns,
129            is_keyframe,
130            payload_len,
131            payload,
132        })
133    }
134
135    /// 头部长度（22 字节），用于 [`encode`] 和 [`decode`] 共享。
136    pub const HEADER_LEN: usize = 4 + 1 + 1 + 8 + 8;
137}

139#[cfg(test)]
140mod tests {
141    use super::*;
142
143    #[test]
144    fn encode_decode_roundtrip_empty_payload() {
145        let wire = WireFrame {
146            frame_index: 42,
147            timestamp_hns: 16666,
148            is_keyframe: true,
149            payload_len: 0,
150            payload: Vec::new(),
151        };
152        let bytes = wire.encode();
153        assert_eq!(bytes.len(), WireFrame::HEADER_LEN);
154        let decoded = WireFrame::decode(&bytes).unwrap();
155        assert_eq!(decoded, wire);
156    }
157
158    #[test]
159    fn encode_decode_roundtrip_with_payload() {
160        let payload_bytes = vec![0x00, 0x00, 0x00, 0x01, 0x67, 0x64, 0x00, 0x28, 0xAC, 0xD9];
161        let wire = WireFrame {
162            frame_index: 12345,
163            timestamp_hns: -9876,
164            is_keyframe: false,
165            payload_len: payload_bytes.len() as u32,
166            payload: payload_bytes,
167        };
168        let bytes = wire.encode();
169        assert_eq!(bytes.len(), WireFrame::HEADER_LEN + 10);
170        let decoded = WireFrame::decode(&bytes).unwrap();
171        assert_eq!(decoded, wire);
172    }
173
174    #[test]
175    fn decode_rejects_short_buffer() {
176        let bytes = vec![0u8; 10];
177        assert!(WireFrame::decode(&bytes).is_err());
178    }
179
180    #[test]
181    fn decode_rejects_length_mismatch() {
182        // 声明 total_len=100 但实际只有 22 字节
183        let mut bytes = vec![0u8; WireFrame::HEADER_LEN];
184        bytes[0..4].copy_from_slice(&100u32.to_le_bytes());
185        assert!(WireFrame::decode(&bytes).is_err());
186    }
