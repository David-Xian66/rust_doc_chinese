1//! D3D11 device + IMFDXGIDeviceManager 封装。
2//!
3//! # 为什么这个
4//!
5//! MF 编码器 / 解码器要访问 GPU 显存时需要 [`IMFDXGIDeviceManager`]：
6//!
7//! 1. 调 `MFCreateDXGIDeviceManager` 拿到 manager
8//! 2. 调 `manager.ResetDevice(d3d_device, token)` 把 D3D11 设备"借"给它
9//! 3. MFT 内部用这个设备创建 / 读写纹理
10//!
11//! 不走这条路 = MFT 拿到的是 CPU 内存 = GPU→CPU→GPU 回读 = 性能崩。
12//!
13//! # 谁需要
14//!
15//! - 服务端：DXGI 桌面复制 → D3D11 设备 → `MediaDevice::new(d3d_device)` → MFT
16//! - 客户端：自己创建 D3D11 设备 → `MediaDevice::new(d3d_device)` → 解码 MFT
17//!
18//! # 用法
19//!
20//! ```ignore
21//! use demo_sc_media::MediaDevice;
22//! use windows::Win32::Graphics::Direct3D11::ID3D11Device;
23//!
24//! // 1. 拿到 D3D11 设备（自己创建 / 来自 DXGI duplication）
25//! let d3d_device: ID3D11Device = ...;
26//!
27//! // 2. 包装成 MediaDevice（创建 DXGI manager + ResetDevice）
28//! let media_device = MediaDevice::new(&d3d_device)?;
29//!
30//! // 3. 把 manager 喂给 MFT
31//! // mft.ProcessMessage(MFT_MESSAGE_SET_D3D_MANAGER, ...)?;
32//! # Ok::<(), Box<dyn std::error::Error>>(())
33//! ```
34
35use windows::core::IUnknown;
36use windows::Win32::Graphics::Direct3D11::ID3D11Device;
37use windows::Win32::Media::MediaFoundation::{
38    MFCreateDXGIDeviceManager, IMFDXGIDeviceManager,
39};
40use windows_core::Interface; // 给 .cast() 用
41
42use crate::error::{MediaError, MediaResult};
43
44/// 包装 IMFDXGIDeviceManager 的 RAII 容器。
45///
46/// **不持有** D3D11 设备的所有权——调用方创建 / 释放设备，`MediaDevice` 只接管 DXGI manager。
47///
48/// 持有 D3D11 设备会让 D3D11 设备生命周期和 MediaDevice 绑定，调用方使用起来很麻烦
49/// （server 的设备来自 `DxgiDuplicationApi`，client 的设备来自自己的渲染栈）。
50pub struct MediaDevice {
51    /// MF 的 DXGI 设备管理器（MFT 借这个访问 GPU 显存）
52    pub dxgi_manager: IMFDXGIDeviceManager,
53
54    /// `ResetDevice` 用的 token（MFT 通过这个 token 找到对应的 D3D11 设备）
55    pub device_token: u32,
56}
57
58impl MediaDevice {
59    /// 用现有的 D3D11 设备创建一个 [`MediaDevice`]。
60    ///
61    /// 内部会调 `MFCreateDXGIDeviceManager` 拿到 manager，再用传入的设备调 `ResetDevice`。
62    ///
63    /// # Errors
64    ///
65    /// - [`MediaError::DxgiDeviceManager`]：`MFCreateDXGIDeviceManager` 失败
66    /// - [`MediaError::DxgiResetDevice`]：`ResetDevice` 失败（设备不兼容 DXGI 1.2）
67    ///
68    /// # Examples
69    ///
70    /// ```ignore
71    /// use demo_sc_media::MediaDevice;
72    ///
73    /// let d3d_device = ...; // 自己创建 / 来自 DxgiDuplicationApi
74    /// let media_device = MediaDevice::new(&d3d_device)?;
75    /// # Ok::<(), Box<dyn std::error::Error>>(())
76    /// ```
77    pub fn new(d3d_device: &ID3D11Device) -> MediaResult<Self> {
78        // 1. 创建 DXGI device manager
79        //    `windows` 0.62.2 的 API：MFCreateDXGIDeviceManager(&mut token, &mut Option<IMFDXGIDeviceManager>)
80        let mut token: u32 = 0;
81        let mut manager_opt: Option<IMFDXGIDeviceManager> = None;
82        unsafe { MFCreateDXGIDeviceManager(&mut token, &mut manager_opt) }
83            .map_err(|e| MediaError::DxgiDeviceManager(e.to_string()))?;
84        let dxgi_manager = manager_opt
85            .ok_or_else(|| MediaError::DxgiDeviceManager("manager is None".into()))?;
86
87        // 2. ResetDevice：把 D3D11 设备"借"给 manager
88        //    IUnknown::cast 是 windows crate 提供的安全转型
89        let unk: IUnknown = d3d_device
90            .cast()
91            .map_err(|e| MediaError::DxgiResetDevice(format!("IUnknown cast: {e}")))?;
92        unsafe { dxgi_manager.ResetDevice(&unk, token) }
93            .map_err(|e| MediaError::DxgiResetDevice(e.to_string()))?;
94
95        Ok(MediaDevice {
96            dxgi_manager,
97            device_token: token,
98        })
99    }
100}
101
102impl Drop for MediaDevice {
103    fn drop(&mut self) {
104        // IMFDXGIDeviceManager 是 COM 智能指针，靠 windows crate 自动 Release
105        // 不需要手动调 Release
106        eprintln!(
107            "[media::device] MediaDevice drop, device_token={}",
108            self.device_token
109        );
110    }
111}
