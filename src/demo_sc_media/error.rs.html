1//! Media Foundation / D3D11 相关错误的统一封装。
2//!
3//! 不引入 `thiserror` —— 错误类型简单，手写 `Display` / `Error` impl 就够了，
4//! 少一个依赖。
5
6use std::fmt;
7
8/// `media` crate 的 [`Result`] 类型。
9pub type MediaResult<T> = Result<T, MediaError>;
10
11/// `media` crate 的统一错误类型。
12///
13/// 各变体对应 MF / D3D11 / DXGI 栈的不同失败点。原始 `windows::core::Error`
14/// 已经被 `to_string()` 化进消息里，方便日志 / 诊断但保留为不可恢复（不重新抛出）。
15#[derive(Debug)]
16pub enum MediaError {
17    /// `MFStartup` 失败
18    MfStartup(String),
19
20    /// D3D11 设备创建失败
21    D3D11Device(String),
22
23    /// `MFCreateDXGIDeviceManager` 失败
24    DxgiDeviceManager(String),
25
26    /// `IMFDXGIDeviceManager::ResetDevice` 失败
27    DxgiResetDevice(String),
28
29    /// `MFTEnumEx` 枚举失败
30    MftEnum(String),
31
32    /// `IMFActivate::ActivateObject` 失败
33    MftActivation(String),
34
35    /// 找不到任何可用的 MFT（HW / SW 都没有）
36    NoMftFound,
37
38    /// 读取 MFT 属性失败（名字 / vendor / 标志位）
39    MftAttribute(String),
40
41    /// 其它未分类错误
42    Other(String),
43}
44
45impl fmt::Display for MediaError {
46    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
47        match self {
48            MediaError::MfStartup(e) => write!(f, "MFStartup 失败: {e}"),
49            MediaError::D3D11Device(e) => write!(f, "D3D11 设备创建失败: {e}"),
50            MediaError::DxgiDeviceManager(e) => {
51                write!(f, "MFCreateDXGIDeviceManager 失败: {e}")
52            }
53            MediaError::DxgiResetDevice(e) => {
54                write!(f, "IMFDXGIDeviceManager::ResetDevice 失败: {e}")
55            }
56            MediaError::MftEnum(e) => write!(f, "MFTEnumEx 失败: {e}"),
57            MediaError::MftActivation(e) => write!(f, "MFT ActivateObject 失败: {e}"),
58            MediaError::NoMftFound => write!(f, "找不到任何可用的 MFT（HW / SW 都没有）"),
59            MediaError::MftAttribute(e) => write!(f, "读取 MFT 属性失败: {e}"),
60            MediaError::Other(e) => write!(f, "{e}"),
61        }
62    }
63}
64
65impl std::error::Error for MediaError {}
66
67/// 方便把 `windows::core::Error` 转成 [`MediaError::Other`]
68impl From<windows::core::Error> for MediaError {
69    fn from(e: windows::core::Error) -> Self {
70        MediaError::Other(e.to_string())
71    }
72}
