1//! `Verifier` implementation for Windows targets.
2//!
3//! The design of the rustls-native-certs crate for Windows doesn't work
4//! completely enough. In general it is hard to emulate enough of what
5//! Windows does to be compatible with all users' configurations, especially
6//! when corporate MitM proxies or custom CAs or complex trust policies are
7//! used. Instead, delegate to Windows's own certificate validation engine
8//! directly.
9//!
10//! This implementation was modeled on:
11//! * Chromium's [cert_verify_proc_win.cc] and [x509_util_win.cc]
12//! * Golang's [root_windows.go]
13//! * [Microsoft's Documentation] and [Microsoft's Example]
14//!
15//! [cert_verify_proc_win.cc]: <https://chromium.googlesource.com/chromium/src/net/+/refs/heads/main/cert/cert_verify_proc_win.cc>
16//! [x509_util_win.cc]: <https://chromium.googlesource.com/chromium/src/net/+/refs/heads/main/cert/x509_util_win.cc>
17//! [root_windows.go]: <https://github.com/golang/go/blob/master/src/crypto/x509/root_windows.go>
18//! [Microsoft's Documentation]: <https://docs.microsoft.com/en-us/windows/win32/api/wincrypt/nf-wincrypt-certgetcertificatechain>
19//! [Microsoft's Example]: <https://docs.microsoft.com/en-us/windows/win32/seccrypto/example-c-program-creating-a-certificate-chain>

127impl CertChain {
128    fn verify_chain_policy(
129        &self,
130        mut server_null_terminated: Vec<u16>,
131    ) -> Result<CERT_CHAIN_POLICY_STATUS, TlsError> {
132        let mut extra_params = HTTPSPolicyCallbackData::zeroed_with_size();
133        extra_params.dwAuthType = AUTHTYPE_SERVER;
134        // `server_null_terminated` outlives `extra_params`.
135        extra_params.pwszServerName = server_null_terminated.as_mut_ptr();
136
137        let mut params = CERT_CHAIN_POLICY_PARA::zeroed_with_size();
138        // Ignore any errors when trying to obtain OCSP revocation information.
139        // This is also done in OpenSSL, Secure Transport from Apple, etc.
140        params.dwFlags = CERT_CHAIN_POLICY_IGNORE_ALL_REV_UNKNOWN_FLAGS;
141        // `extra_params` outlives `params`.
142        params.pvExtraPolicyPara = NonNull::from(&mut extra_params).cast::<c_void>().as_ptr();
143
144        let mut status: MaybeUninit<CERT_CHAIN_POLICY_STATUS> = MaybeUninit::uninit();
145
146        // SAFETY: The certificate chain is non-null, `params` is valid for reads, and its valid to write to `status`.
147        let res = unsafe {
148            CertVerifyCertificateChainPolicy(
149                CERT_CHAIN_POLICY_SSL,
150                self.inner.as_ptr(),
151                &params,
152                status.as_mut_ptr(),
153            )
154        };
155
156        // This should rarely, if ever, be false since it would imply no TLS verification
157        // is currently possible on the system: https://docs.microsoft.com/en-us/windows/win32/api/wincrypt/nf-wincrypt-certverifycertificatechainpolicy#return-value
158        if res != TRUE {
159            return Err(TlsError::General(String::from(
160                "TLS certificate verification was unavailable on the system!",
161            )));
162        }
163
164        // SAFETY: The verification call was checked to have succeeded, so the status
165        // is written correctly and initialized.
166        let status = unsafe { status.assume_init() };
167        Ok(status)
168    }
169}

178/// A representation of a certificate.
179///
180/// The `CertificateStore` must be opened with the correct flags to ensure the
181/// certificate may outlive it; see the `CertificateStore` documentation.
182struct Certificate {
183    inner: NonNull<CERT_CONTEXT>,
184}
185
186impl Certificate {
187    /// Sets the specified property of this certificate context.
188    ///
189    /// ### Safety
190    /// `prop_data` must be a valid pointer for the property type.
191    unsafe fn set_property(
192        &mut self,
193        prop_id: u32,
194        prop_data: *const c_void,
195    ) -> Result<(), TlsError> {
196        // SAFETY: `cert` points to a valid certificate context and the OCSP data is valid to read.
197        call_with_last_error(|| {
198            (CertSetCertificateContextProperty(
199                self.inner.as_ptr(),
200                prop_id,
201                CERT_SET_PROPERTY_IGNORE_PERSIST_ERROR_FLAG,
202                prop_data,
203            ) == TRUE)
204                .then_some(())
205        })
206    }
207}

213    }

216#[derive(Debug)]
217struct CertEngine {
218    inner: NonNull<c_void>, // HCERTENGINECONTEXT
219}
220
221impl CertEngine {
222    fn new_with_extra_roots(
223        roots: impl IntoIterator<Item = pki_types::CertificateDer<'static>>,
224    ) -> Result<Self, TlsError> {
225        let mut exclusive_store = CertificateStore::new()?;
226        for root in roots {
227            exclusive_store.add_cert(&root)?;
228        }
229
230        let mut config = CERT_CHAIN_ENGINE_CONFIG::zeroed_with_size();
231        config.hExclusiveRoot = exclusive_store.inner.as_ptr();
232
233        let mut engine = EnginePtr::NULL;
234        // SAFETY: `engine` is valid to be written to and the config is valid to be read.
235        let res = unsafe { CertCreateCertificateChainEngine(&config, &mut engine) };
236
237        #[allow(clippy::as_conversions)]
238        let engine = call_with_last_error(|| match NonNull::new(engine as *mut c_void) {
239            Some(c) if res == TRUE => Some(c),
240            _ => None,
241        })?;
242        Ok(Self { inner: engine })
243    }
244
245    #[cfg(any(test, feature = "ffi-testing", feature = "dbg"))]
246    fn new_with_fake_root(root: &[u8]) -> Result<Self, TlsError> {
247        use windows_sys::Win32::Security::Cryptography::{
248            CERT_CHAIN_CACHE_ONLY_URL_RETRIEVAL, CERT_CHAIN_ENABLE_CACHE_AUTO_UPDATE,
249        };
250
251        let mut root_store = CertificateStore::new()?;
252        root_store.add_cert(root)?;
253
254        let mut config = CERT_CHAIN_ENGINE_CONFIG::zeroed_with_size();
255        // We use these flags for the following reasons:
256        //
257        // - CERT_CHAIN_CACHE_ONLY_URL_RETRIEVAL is used in an attempt to stop Windows from using the internet to
258        // fetch anything during the tests, regardless of what test data is used.
259        //
260        // - CERT_CHAIN_ENABLE_CACHE_AUTO_UPDATE is used as a minor performance optimization to allow Windows to reuse
261        // data inside of a test and avoid any extra parsing, etc, it might need to do pulling directly from the store each time.
262        //
263        // Ref: https://docs.microsoft.com/en-us/windows/win32/api/wincrypt/ns-wincrypt-cert_chain_engine_config
264        config.dwFlags = CERT_CHAIN_CACHE_ONLY_URL_RETRIEVAL | CERT_CHAIN_ENABLE_CACHE_AUTO_UPDATE;
265        config.hExclusiveRoot = root_store.inner.as_ptr();
266
267        let mut engine = EnginePtr::NULL;
268        // SAFETY: `engine` is valid to be written to and the config is valid to be read.
269        let res = unsafe { CertCreateCertificateChainEngine(&config, &mut engine) };
270
271        #[allow(clippy::as_conversions)]
272        let engine = call_with_last_error(|| match NonNull::new(engine as *mut c_void) {
273            Some(c) if res == TRUE => Some(c),
274            _ => None,
275        })?;
276
277        Ok(Self { inner: engine })
278    }
279}

295/// An in-memory Windows certificate store.
296///
297/// # Safety
298///
299/// `CertificateStore` creates `Certificate` objects that may outlive the
300/// `CertificateStore`. This is only safe to do if the certificate store is
301/// constructed with `CERT_STORE_DEFER_CLOSE_UNTIL_LAST_FREE_FLAG`.
302struct CertificateStore {
303    inner: NonNull<c_void>, // HCERTSTORE
304    // In production code, this is always `None`.
305    //
306    // During tests, we set this to `Some` as the tests use a
307    // custom verification engine that only uses specific roots.
308    engine: Option<CertEngine>, // HCERTENGINECONTEXT
309}

311impl Drop for CertificateStore {
312    fn drop(&mut self) {
313        // SAFETY: See the `CertificateStore` documentation.
314        unsafe { CertCloseStore(self.inner.as_ptr(), 0) };
315    }
316}

318impl CertificateStore {
319    /// Creates a new, in-memory certificate store.
320    fn new() -> Result<Self, TlsError> {
321        let store = call_with_last_error(|| {
322            // SAFETY: Called with valid constants and result is checked to be non-null.
323            // The `CERT_STORE_DEFER_CLOSE_UNTIL_LAST_FREE_FLAG` flag is critical;
324            // see the `CertificateStore` documentation for more info.
325            NonNull::new(unsafe {
326                CertOpenStore(
327                    CERT_STORE_PROV_MEMORY,
328                    0, // Set to zero since this uses `PROV_MEMORY`.
329                    0, // This field shouldn't be used.
330                    CERT_STORE_DEFER_CLOSE_UNTIL_LAST_FREE_FLAG,
331                    ptr::null(),
332                )
333            })
334        })?;
335
336        // Use the system's default root store and rules.
337        Ok(Self {
338            inner: store,
339            engine: None,
340        })
341    }
342
343    #[cfg(any(test, feature = "ffi-testing", feature = "dbg"))]
344    fn new_with_fake_root(root: &[u8]) -> Result<Self, TlsError> {
345        let mut inner = Self::new()?;
346
347        let mut root_store = CertificateStore::new()?;
348        root_store.add_cert(root)?;
349
350        let engine = CertEngine::new_with_fake_root(root)?;
351        inner.engine = Some(engine);
352
353        Ok(inner)
354    }
355
356    /// Adds the provided certificate to the store.
357    ///
358    /// The certificate must be encoded as ASN.1 DER.
359    ///
360    /// Errors if the certificate was malformed and couldn't be added.
361    fn add_cert(&mut self, cert: &[u8]) -> Result<Certificate, TlsError> {
362        let mut cert_context: *mut CERT_CONTEXT = ptr::null_mut();
363
364        // SAFETY: `inner` is a valid certificate store, and `cert` is a valid a byte array valid
365        // for reads, the correct length is being provided, and `cert_context` is valid to write to.
366        let res = unsafe {
367            CertAddEncodedCertificateToStore(
368                self.inner.as_ptr(),
369                X509_ASN_ENCODING,
370                cert.as_ptr(),
371                cert.len()
372                    .try_into()
373                    .map_err(|_| InvalidCertificate(CertificateError::BadEncoding))?,
374                CERT_STORE_ADD_ALWAYS,
375                &mut cert_context,
376            )
377        };
378
379        // SAFETY: Constructing a `Certificate` is only safe if the store was
380        // created with the right flags; see the `CertificateStore` docs.
381        match (res, NonNull::new(cert_context)) {
382            (TRUE, Some(cert)) => Ok(Certificate { inner: cert }),
383            _ => Err(InvalidCertificate(CertificateError::BadEncoding)),
384        }
385    }

387    fn new_chain_in(
388        &self,
389        certificate: &Certificate,
390        now: pki_types::UnixTime,
391        engine: Option<&CertEngine>,
392    ) -> Result<CertChain, TlsError> {
393        let mut cert_chain = ptr::null_mut();
394
395        let mut parameters = CERT_CHAIN_PARA::zeroed_with_size();
396
397        #[allow(clippy::as_conversions)]
398        // https://docs.microsoft.com/en-us/windows/win32/api/wincrypt/ns-wincrypt-cert_usage_match
399        let usage = CERT_USAGE_MATCH {
400            dwType: USAGE_MATCH_TYPE_AND,
401            Usage: CTL_USAGE {
402                cUsageIdentifier: ALLOWED_EKUS.len() as u32,
403                rgpszUsageIdentifier: ALLOWED_EKUS.as_ptr() as *mut windows_sys::core::PSTR,
404            },
405        };
406        parameters.RequestedUsage = usage;
407
408        #[allow(clippy::as_conversions)]
409        let time = {
410            /// Seconds between Jan 1st, 1601 and Jan 1, 1970.
411            const UNIX_ADJUSTMENT: std::time::Duration =
412                std::time::Duration::from_secs(11_644_473_600);
413
414            let since_unix_epoch = now.as_secs();
415
416            // Convert the duration from the UNIX epoch to the Window one, and then convert
417            // the result into a `FILETIME` structure.
418
419            let since_windows_epoch = since_unix_epoch + UNIX_ADJUSTMENT.as_secs();
420            let intervals = (since_windows_epoch * 1_000_000_000) / 100;
421
422            FILETIME {
423                dwLowDateTime: (intervals & u32::MAX as u64) as u32,
424                dwHighDateTime: (intervals >> 32) as u32,
425            }
426        };
427
428        // `CERT_CHAIN_REVOCATION_CHECK_END_CERT` only checks revocation for end cert. See the crate's revocation documentation
429        // for more details.
430        // `CERT_CHAIN_REVOCATION_ACCUMULATIVE_TIMEOUT` accumulates network retrievals timeouts
431        // to limit network time and improve performance.
432        // `CERT_CHAIN_CACHE_END_CERT` speeds up the common case of multiple connections to same server.
433        const FLAGS: u32 = CERT_CHAIN_REVOCATION_CHECK_END_CERT
434            | CERT_CHAIN_REVOCATION_ACCUMULATIVE_TIMEOUT
435            | CERT_CHAIN_CACHE_END_CERT;
436
437        // Lowering URL retrieval timeout from default 15s to 10s to account for higher internet speeds
438        parameters.dwUrlRetrievalTimeout = 10 * 1000; // milliseconds
439
440        // SAFETY: `cert` points to a valid certificate context, parameters is valid for reads, `cert_chain` is valid
441        // for writes, and the certificate store is valid and initialized.
442        let res = unsafe {
443            // XXX: Due to the redefinition of `CERT_CHAIN_PARA`, we need to do pointer casts
444            // in order to pass our expanded structure into `CertGetCertificateChain`.
445            // This is safe because the OS uses `cbSize` to know if the extra parameters
446            // are present or not. As we set `cbSize` correctly, the fields can be read from correctly.
447            let parameters = NonNull::from(&parameters).cast().as_ptr();
448
449            CertGetCertificateChain(
450                match engine {
451                    Some(eng) => EnginePtr::from_raw(eng.inner),
452                    None => EnginePtr::NULL,
453                },
454                certificate.inner.as_ptr(),
455                &time,
456                self.inner.as_ptr(),
457                parameters,
458                FLAGS,
459                ptr::null_mut(),
460                &mut cert_chain,
461            )
462        };
463
464        // XXX: Windows will internally map the chain's `TrustStatus.dwErrorStatus` to a `dwError` when
465        // a chain policy is verified, so we only check for errors there.
466        call_with_last_error(|| match NonNull::new(cert_chain) {
467            Some(c) if res == TRUE => Some(CertChain { inner: c }),
468            _ => None,
469        })
470    }
471}

500fn call_with_last_error<T, F: FnMut() -> Option<T>>(mut call: F) -> Result<T, TlsError> {
501    if let Some(res) = call() {
502        Ok(res)
503    } else {
504        Err(TlsError::General(
505            std::io::Error::last_os_error().to_string(),
506        ))
507    }

562    /// Verifies a certificate and its chain for the specified `server`.
563    ///
564    /// Return `Ok(())` if the certificate was valid.
565    fn verify_certificate(
566        &self,
567        primary_cert: &[u8],
568        intermediate_certs: &[&[u8]],
569        server: &[u8],
570        ocsp_data: Option<&[u8]>,
571        now: pki_types::UnixTime,
572    ) -> Result<(), TlsError> {
573        #[cfg(any(test, feature = "ffi-testing", feature = "dbg"))]
574        let mut store = match self.test_only_root_ca_override.as_ref() {
575            Some(test_only_root_ca_override) => {
576                CertificateStore::new_with_fake_root(test_only_root_ca_override)?
577            }
578            None => CertificateStore::new()?,
579        };
580
581        #[cfg(not(any(test, feature = "ffi-testing", feature = "dbg")))]
582        let mut store = CertificateStore::new()?;
583
584        let mut primary_cert = store.add_cert(primary_cert)?;
585
586        for cert in intermediate_certs.iter().copied() {
587            store.add_cert(cert)?;
588        }
589
590        if let Some(ocsp_data) = ocsp_data {
591            #[allow(clippy::as_conversions)]
592            let data = CRYPT_INTEGER_BLOB {
593                cbData: ocsp_data.len().try_into().map_err(|_| {
594                    invalid_certificate("Malformed OCSP response stapled to server certificate")
595                })?,
596                pbData: ocsp_data.as_ptr() as *mut u8,
597            };
598
599            // SAFETY: `data` is a valid pointer and matches the property ID.
600            unsafe {
601                primary_cert.set_property(
602                    CERT_OCSP_RESPONSE_PROP_ID,
603                    NonNull::from(&data).cast::<c_void>().as_ptr(),
604                )?;
605            }
606        }
607
608        // Encode UTF-16, null-terminated
609        let server: Vec<u16> = server
610            .iter()
611            .map(|c| u16::from(*c))
612            .chain(Some(0))
613            .collect();
614
615        let mut cert_chain = store.new_chain_in(&primary_cert, now, store.engine.as_ref())?;
616
617        // We only use `TrustStatus` here because it hasn't had verification performed on it.
618        // SAFETY: The pointer is guaranteed to be non-null.
619        let is_partial_chain = unsafe { *cert_chain.inner.as_ptr() }
620            .TrustStatus
621            .dwErrorStatus
622            & CERT_TRUST_IS_PARTIAL_CHAIN
623            != 0;
624
625        // If we have extra roots and building the chain gave us an error, we try to build a
626        // new one with the extra roots.
627        if is_partial_chain && self.extra_roots.is_some() {
628            let mut store = CertificateStore::new()?;
629
630            for cert in intermediate_certs.iter().copied() {
631                store.add_cert(cert)?;
632            }
633
634            cert_chain = store.new_chain_in(&primary_cert, now, self.extra_roots.as_ref())?;
635        }
636
637        let status = cert_chain.verify_chain_policy(server)?;
638
639        if status.dwError == 0 {
640            return Ok(());
641        }
642
643        // Only map the errors we have tests for.
644        #[allow(clippy::as_conversions)]
645        let win_error = status.dwError as i32;
646        Err(match win_error {
647            CERT_E_CN_NO_MATCH | CERT_E_INVALID_NAME => {
648                InvalidCertificate(CertificateError::NotValidForName)
649            }
650            CRYPT_E_REVOKED => InvalidCertificate(CertificateError::Revoked),
651            CERT_E_EXPIRED => InvalidCertificate(CertificateError::Expired),
652            CERT_E_UNTRUSTEDROOT => InvalidCertificate(CertificateError::UnknownIssuer),
653            CERT_E_WRONG_USAGE => InvalidCertificate(CertificateError::InvalidPurpose),
654            error_num => {
655                let err = std::io::Error::from_raw_os_error(error_num);
656                // The included error message has both the description and raw OS error code.
657                invalid_certificate(err.to_string())
658            }
659        })
660    }
661}

663impl ServerCertVerifier for Verifier {
664    fn verify_server_cert(
665        &self,
666        end_entity: &pki_types::CertificateDer<'_>,
667        intermediates: &[pki_types::CertificateDer<'_>],
668        server_name: &pki_types::ServerName,
669        ocsp_response: &[u8],
670        now: pki_types::UnixTime,
671    ) -> Result<rustls::client::danger::ServerCertVerified, TlsError> {
672        log_server_cert(end_entity);
673
674        let name = server_name.to_str();
675
676        let intermediate_certs: Vec<&[u8]> = intermediates.iter().map(|c| c.as_ref()).collect();
677
678        let ocsp_data = if !ocsp_response.is_empty() {
679            Some(ocsp_response)
680        } else {
681            None
682        };
683
684        match self.verify_certificate(
685            end_entity.as_ref(),
686            &intermediate_certs,
687            name.as_bytes(),
688            ocsp_data,
689            now,
690        ) {
691            Ok(()) => Ok(rustls::client::danger::ServerCertVerified::assertion()),
692            Err(e) => {
693                // SAFETY:
694                // Errors are our own custom errors, WinAPI errors, or static strings.
695                log::error!("failed to verify TLS certificate: {}", e);
696                Err(e)
697            }
698        }
699    }
700
701    fn verify_tls12_signature(
702        &self,
703        message: &[u8],
704        cert: &pki_types::CertificateDer<'_>,
705        dss: &DigitallySignedStruct,
706    ) -> Result<HandshakeSignatureValid, TlsError> {
707        verify_tls12_signature(
708            message,
709            cert,
710            dss,
711            &self.crypto_provider.signature_verification_algorithms,
712        )
713    }
714
715    fn verify_tls13_signature(
716        &self,
717        message: &[u8],
718        cert: &pki_types::CertificateDer<'_>,
719        dss: &DigitallySignedStruct,
720    ) -> Result<HandshakeSignatureValid, TlsError> {
721        verify_tls13_signature(
722            message,
723            cert,
724            dss,
725            &self.crypto_provider.signature_verification_algorithms,
726        )
727    }
728
729    fn supported_verify_schemes(&self) -> Vec<SignatureScheme> {
730        self.crypto_provider
731            .signature_verification_algorithms
732            .supported_schemes()
733    }
734}

736/// A trait to represent an object that can be safely created with all zero values
737/// and have a size assigned to it.
738///
739/// # Safety
740///
741/// This has the same safety requirements as [std::mem::zeroed].
742unsafe trait ZeroedWithSize: Sized {
743    const SIZE: u32 = {
744        let size = core::mem::size_of::<Self>();
745
746        // NB: `TryInto` isn't stable in const yet.
747        #[allow(clippy::as_conversions)]
748        if size <= u32::MAX as usize {
749            size as u32
750        } else {
751            panic!("structure was larger then DWORD")
752        }
753    };
754
755    /// Returns a zeroed structure with its structure size (`cbSize`) field set to the correct value.
756    fn zeroed_with_size() -> Self;
757}
