25/// demo 阶段的最小 EventHandler 实现：只处理 WM_DESTROY，其它交给默认处理
26struct DemoHandler {
27    // is_connected: bool,
28}
29
30impl EventHandler for DemoHandler {
31    fn on_message(&mut self, msg: &Message) -> Option<LRESULT> {
32        if msg.msg == WM_DESTROY {
33            // 业务方清理资源：socket 关闭、文件 flush 等
34            // self.is_connected = false;
35            // 不用手动 PostQuitMessage - 封装层会处理
36            Some(LRESULT(0))
37        } else {
38            // 其它消息：让系统走默认处理
39            None
40        }
41    }
42}

44fn main() -> anyhow::Result<()> {
45    let _server = ServerInfo {
46        ip: IpAddr::V4(Ipv4Addr::new(127, 0, 0, 1)),
47        port: 8080,
48    };
49
50    let host_ = SocketAddr::new(_server.ip, _server.port);
51
52    let rt = tokio::runtime::Builder::new_multi_thread()
53        .worker_threads(8)  // 8个工作线程
54        .enable_io()        // 可在runtime中使用异步IO
55        .enable_time()      // 可在runtime中使用异步计时器(timer)
56        .build()?;
57
58    let CertifiedKey { cert, signing_key } = rcgen::generate_simple_self_signed(vec!["localhost".into()])?;
59    let cert_der = cert.der().clone();
60
61    let server_config = ServerConfig::with_single_cert(
62        vec![CertificateDer::from(cert_der)],
63        PrivateKeyDer::from(signing_key),
64    )?;
65
66
67    rt.spawn(async move {
68        let endpoint = Endpoint::server(server_config, host_);
69
70        match endpoint {
71            Ok(endpoint_) => {
72                while let Some(incoming) =  endpoint_.accept().await {
73                    match incoming.await {
74                        Ok(K) => {
75
76                        }
77                        Err(_) => {
78                            // TODO: 这里发个信号给主程序
79                        }
80                    }
81                }
82            }
83            Err(_) => {
84                // TODO: 这里发个信号给主程序
85            }
86        }
87
88    });
89
90
91
92
93    // 1. 用 builder 创建窗口
94    //    风格位直接给标准三键窗口：WS_OVERLAPPEDWINDOW = CAPTION + SYSMENU + THICKFRAME + MIN/MAXIMIZEBOX
95    //    用 WindowClassPreset::MainControl 预设窗口类（昵称查找），
96    //    业务方不用记 WindowClassConfig 的细节
97    let main_hwnd = WindowBuilder::new()
98        .preset_class(WindowClassPreset::MainControl)
99        .title("Demo_火炬远控")
100        .ex_style(WS_EX_APPWINDOW)
101        .style(WS_OVERLAPPEDWINDOW | WS_VISIBLE)
102        // .build(DemoHandler { is_connected: false })?;
103        .build(DemoHandler {})?;
104
105
106    // 2. 解码器窗口：owner = main_hwnd
107    //    - 不出现在任务栏
108    //    - 永远在主窗口之上
109    //    - 主窗口关闭时自动销毁
110    //    - 主窗口最小化时自动最小化
111    let _media_window_hwnd = WindowBuilder::new()
112        .preset_class(WindowClassPreset::VideoDecoder)
113        .title("Demo_火炬远控_画面")
114        .ex_style(WS_EX_APPWINDOW)
115        .style(WS_OVERLAPPEDWINDOW | WS_VISIBLE)
116        .parent(main_hwnd)
117        .build(DemoHandler {})?;
118
119    // 3. H.264 解码器初始化 + 预留 input 接口
120    //    `H264Decoder::feed(bytes)` 是网络线程喂 H.264 字节的入口。
121    //    当前 demo 不接 socket、也不把 NV12 渲染到上面的 media 窗口 —— 仅构造一次展示 API 形状。
122    let mut decoder = H264Decoder::new(1920, 1080)?;
123    let _ = decoder.stats(); // 摸一下字段，证明 init 路径跑通
124
125    // 2. 主线程消息循环
126    //    GetMessageW 的 hwnd 参数传 None 表示"收本线程所有窗口的消息"
127    //    WM_DESTROY 时封装层会自动 PostQuitMessage，循环会自动退出
128    unsafe {
129        let mut msg = MSG::default();
130        while GetMessageW(&mut msg, None, 0, 0).as_bool() {
131            let _ = TranslateMessage(&msg);
132            DispatchMessageW(&msg);
133        }
134    }
135
136    Ok(())
